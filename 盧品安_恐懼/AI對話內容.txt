    const canvas = document.getElementById('canvas');
    const ctx = canvas.getContext('2d');
    let width, height;
    let audioCtx = null;

    // 自適應調整大小
    function resize() {
        width = canvas.width = window.innerWidth;
        height = canvas.height = window.innerHeight;
    }
    window.addEventListener('resize', resize);
    resize();

    // 隨機整數
    const rand = (min, max) => Math.random() * (max - min) + min;

    // 繪製背景質感 (雜點)
    function drawGrain() {
        ctx.fillStyle = 'rgba(255, 255, 255, 0.03)';
        for (let i = 0; i < 5000; i++) {
            ctx.fillRect(rand(0, width), rand(0, height), 1, 1);
        }
    }

    // 繪製尖銳三角形 (恐懼的防禦)
    function drawTriangle() {
        ctx.strokeStyle = '#333';
        ctx.lineWidth = rand(0.5, 2);
        ctx.beginPath();
        ctx.moveTo(rand(0, width), rand(0, height));
        ctx.lineTo(rand(0, width), rand(0, height));
        ctx.lineTo(rand(0, width), rand(0, height));
        ctx.closePath();
        ctx.stroke();
    }

    // 初始化背景
    function init() {
        ctx.fillStyle = '#0a0a0a';
        ctx.fillRect(0, 0, width, height);
        drawGrain();
        for (let i = 0; i < 30; i++) drawTriangle();
    }

    // 混亂線條繪製 (互動觸發)
    function drawChaos(x, y) {
        ctx.strokeStyle = `rgba(${rand(100, 200)}, 0, 0, 0.5)`; // 暗紅/灰色線條
        ctx.lineWidth = rand(1, 3);
        ctx.beginPath();
        ctx.moveTo(x, y);
        for (let i = 0; i < 15; i++) {
            ctx.lineTo(x + rand(-100, 100), y + rand(-100, 100));
        }
        ctx.stroke();
    }

    // 初始化音訊 (產生煩躁的雜訊)
    function initAudio() {
        if (audioCtx) return;
        audioCtx = new (window.AudioContext || window.webkitAudioContext)();
        const bufferSize = 2 * audioCtx.sampleRate,
        noiseBuffer = audioCtx.createBuffer(1, bufferSize, audioCtx.sampleRate),
        output = noiseBuffer.getChannelData(0);
        for (let i = 0; i < bufferSize; i++) {
            output[i] = Math.random() * 2 - 1;
        }

        const whiteNoise = audioCtx.createBufferSource();
        whiteNoise.buffer = noiseBuffer;
        whiteNoise.loop = true;

        const filter = audioCtx.createBiquadFilter();
        filter.type = 'lowpass';
        filter.frequency.value = 400;

        const gainNode = audioCtx.createGain();
        gainNode.gain.value = 0.05; // 初始音量較小

        whiteNoise.connect(filter);
        filter.connect(gainNode);
        gainNode.connect(audioCtx.destination);
        whiteNoise.start();

        return { filter, gainNode };
    }

    let audio;

    // 滑鼠點擊互動
    window.addEventListener('mousedown', (e) => {
        if (!audio) audio = initAudio();
        
        // 增加音量與頻率使人煩躁
        if (audio) {
            audio.gainNode.gain.setTargetAtTime(0.15, audioCtx.currentTime, 0.1);
            audio.filter.frequency.setTargetAtTime(rand(1000, 3000), audioCtx.currentTime, 0.1);
        }

        // 視覺混亂與震動
        document.body.style.transform = `translate(${rand(-10, 10)}px, ${rand(-10, 10)}px)`;
        for(let i=0; i<5; i++) drawChaos(e.clientX, e.clientY);
    });

    window.addEventListener('mouseup', () => {
        if (audio) {
            audio.gainNode.gain.setTargetAtTime(0.05, audioCtx.currentTime, 0.5);
            audio.filter.frequency.setTargetAtTime(400, audioCtx.currentTime, 0.5);
        }
        document.body.style.transform = 'translate(0, 0)';
    });

    init();
</script>

<script>
    const canvas = document.getElementById('canvas');
    const ctx = canvas.getContext('2d');
    let width, height;
    let audioCtx = null;
    let audioFilter, audioGain;
    let mouse = { x: 0, y: 0, clicked: false, lastX: 0, lastY: 0 };
    let triangles = [];
    let chaosLines = [];
    let frameCount = 0;

    // 自適應調整大小
    function resize() {
        width = canvas.width = window.innerWidth;
        height = canvas.height = window.innerHeight;
    }
    window.addEventListener('resize', resize);
    resize();

    // 隨機數生成
    const rand = (min, max) => Math.random() * (max - min) + min;

    // 三角形物件
    class Triangle {
        constructor(x, y) {
            this.x = x;
            this.y = y;
            this.baseSize = rand(15, 40);
            this.size = this.baseSize;
            this.color = `rgba(${rand(30, 80)}, ${rand(0, 20)}, ${rand(0, 20)}, ${rand(0.2, 0.5)})`;
            this.rotation = rand(0, Math.PI * 2);
            this.speedX = rand(-0.5, 0.5);
            this.speedY = rand(-0.5, 0.5);
            this.targetX = this.x;
            this.targetY = this.y;
            this.avoiding = false;
        }

        draw() {
            ctx.save();
            ctx.translate(this.x, this.y);
            ctx.rotate(this.rotation + Math.sin(frameCount * 0.05 + this.x) * 0.1); // 微小抖動
            ctx.strokeStyle = this.color;
            ctx.lineWidth = rand(0.5, 2);
            ctx.beginPath();
            ctx.moveTo(0, -this.size / 2);
            ctx.lineTo(-this.size / 2, this.size / 2);
            ctx.lineTo(this.size / 2, this.size / 2);
            ctx.closePath();
            ctx.stroke();
            ctx.restore();
        }

        update() {
            if (this.avoiding) {
                this.x += this.speedX * 5; // 閃避時加速
                this.y += this.speedY * 5;
                // 逐漸回到原位 (減速)
                this.speedX *= 0.95;
                this.speedY *= 0.95;
                if (Math.abs(this.speedX) < 0.1 && Math.abs(this.speedY) < 0.1) {
                    this.avoiding = false;
                    this.speedX = rand(-0.5, 0.5);
                    this.speedY = rand(-0.5, 0.5);
                }
            } else {
                // 緩慢漂移
                this.x += this.speedX;
                this.y += this.speedY;

                // 邊界反彈
                if (this.x < 0 || this.x > width) this.speedX *= -1;
                if (this.y < 0 || this.y > height) this.speedY *= -1;
            }
        }

        avoid(targetX, targetY) {
            this.avoiding = true;
            const dx = this.x - targetX;
            const dy = this.y - targetY;
            const dist = Math.sqrt(dx * dx + dy * dy);
            const force = 100 / (dist + 1); // 距離越近，閃避力越大
            this.speedX = dx / dist * force;
            this.speedY = dy / dist * force;
        }
    }

    // 混亂線條物件
    class ChaosLine {
        constructor(x1, y1, x2, y2, opacity) {
            this.x1 = x1;
            this.y1 = y1;
            this.x2 = x2;
            this.y2 = y2;
            this.opacity = opacity;
            this.life = 1;
            this.color = `rgba(${rand(50, 150)}, ${rand(0, 50)}, ${rand(0, 50)}, `;
        }

        draw() {
            ctx.strokeStyle = this.color + (this.opacity * this.life) + ')';
            ctx.lineWidth = rand(0.5, 2);
            ctx.beginPath();
            ctx.moveTo(this.x1, this.y1);
            ctx.lineTo(this.x2, this.y2);
            ctx.stroke();
        }

        update() {
            this.life -= 0.01; // 逐漸消失
            return this.life > 0;
        }
    }

    // 初始化音訊 (產生煩躁的雜訊)
    function initAudio() {
        if (audioCtx) return;
        audioCtx = new (window.AudioContext || window.webkitAudioContext)();
        const bufferSize = 2 * audioCtx.sampleRate,
        noiseBuffer = audioCtx.createBuffer(1, bufferSize, audioCtx.sampleRate),
        output = noiseBuffer.getChannelData(0);
        for (let i = 0; i < bufferSize; i++) {
            output[i] = Math.random() * 2 - 1;
        }

        const whiteNoise = audioCtx.createBufferSource();
        whiteNoise.buffer = noiseBuffer;
        whiteNoise.loop = true;

        audioFilter = audioCtx.createBiquadFilter();
        audioFilter.type = 'lowpass';
        audioFilter.frequency.value = 200; // 初始低頻雜訊

        audioGain = audioCtx.createGain();
        audioGain.gain.value = 0.03; // 初始音量極小

        whiteNoise.connect(audioFilter);
        audioFilter.connect(audioGain);
        audioGain.connect(audioCtx.destination);
        whiteNoise.start();
    }

    // 初始化三角形
    for (let i = 0; i < 50; i++) {
        triangles.push(new Triangle(rand(0, width), rand(0, height)));
    }

    // 主要繪圖迴圈
    function animate() {
        frameCount++;
        ctx.fillStyle = 'rgba(10, 10, 10, 0.2)'; // 輕微殘影效果
        ctx.fillRect(0, 0, width, height);

        // 更新與繪製混亂線條
        if (frameCount % 3 === 0) { // 每隔幾幀生成新的背景混亂線條
            chaosLines.push(new ChaosLine(rand(0, width), rand(0, height), rand(0, width), rand(0, height), rand(0.05, 0.15)));
        }
        chaosLines = chaosLines.filter(line => line.update());
        chaosLines.forEach(line => line.draw());


        // 更新與繪製三角形
        triangles.forEach(tri => {
            tri.update();
            tri.draw();
        });

        // 滑鼠殘影
        if (mouse.lastX !== mouse.x || mouse.lastY !== mouse.y) {
            chaosLines.push(new ChaosLine(mouse.lastX, mouse.lastY, mouse.x + rand(-20, 20), mouse.y + rand(-20, 20), 0.3));
            mouse.lastX = mouse.x;
            mouse.lastY = mouse.y;
        }

        requestAnimationFrame(animate);
    }

    // 滑鼠移動事件
    window.addEventListener('mousemove', (e) => {
        if (!audioCtx) initAudio();
        mouse.x = e.clientX;
        mouse.y = e.clientY;

        // 隨滑鼠移動調整音訊 (輕微煩躁感)
        const distToCenter = Math.sqrt(Math.pow(e.clientX - width / 2, 2) + Math.pow(e.clientY - height / 2, 2));
        const normalizedDist = Math.min(1, distToCenter / (Math.max(width, height) / 2));
        audioFilter.frequency.value = 200 + normalizedDist * 800; // 越遠雜訊越刺耳
        audioGain.gain.value = 0.03 + normalizedDist * 0.05; // 音量略增

        document.body.style.transform = `translate(${rand(-1, 1)}px, ${rand(-1, 1)}px)`; // 輕微抖動
    });

    // 滑鼠點擊事件 (閃避與強烈音效)
    window.addEventListener('mousedown', (e) => {
        mouse.clicked = true;
        if (!audioCtx) initAudio();

        // 點擊時，所有三角形閃避
        triangles.forEach(tri => tri.avoid(e.clientX, e.clientY));

        // 強烈音效變化
        audioGain.gain.setTargetAtTime(0.2, audioCtx.currentTime, 0.1);
        audioFilter.frequency.setTargetAtTime(rand(1500, 4000), audioCtx.currentTime, 0.1);
        document.body.style.transform = `translate(${rand(-10, 10)}px, ${rand(-10, 10)}px)`; // 強烈震動
    });

    window.addEventListener('mouseup', () => {
        mouse.clicked = false;
        if (audioCtx) {
            audioGain.gain.setTargetAtTime(0.03, audioCtx.currentTime, 0.5); // 恢復背景音量
            audioFilter.frequency.setTargetAtTime(200, audioCtx.currentTime, 0.5); // 恢復背景頻率
        }
        document.body.style.transform = 'translate(0, 0)';
    });

    // 啟動動畫
    animate();
</script>
</body>

    let audioCtx = null, audioFilter, audioGain;
    let mouse = { x: 0, y: 0, lastX: 0, lastY: 0, active: false };
    let triangles = [];
    let fragments = [];
    let frameCount = 0;

    function resize() {
        width = canvas.width = window.innerWidth;
        height = canvas.height = window.innerHeight;
    }
    window.addEventListener('resize', resize);
    resize();

    const rand = (min, max) => Math.random() * (max - min) + min;

    class Triangle {
        constructor(isPlayer = false) {
            this.isPlayer = isPlayer;
            this.reset();
        }

        reset() {
            this.x = rand(0, width);
            this.y = rand(0, height);
            this.size = this.isPlayer ? 60 : rand(20, 40);
            this.rotation = rand(0, Math.PI * 2);
            this.vx = rand(-1, 1);
            this.vy = rand(-1, 1);
            this.color = this.isPlayer ? "#ffffff" : `rgba(${rand(50, 100)}, ${rand(50, 100)}, ${rand(50, 100)}, 0.4)`;
            this.lineWidth = this.isPlayer ? 8 : 2; // 白色三角形加粗
        }

        draw(targetX, targetY) {
            let posX = this.isPlayer ? targetX : this.x;
            let posY = this.isPlayer ? targetY : this.y;

            ctx.save();
            ctx.translate(posX, posY);
            ctx.rotate(this.rotation + (this.isPlayer ? frameCount * 0.1 : 0));
            ctx.strokeStyle = this.color;
            ctx.lineWidth = this.lineWidth;
            ctx.beginPath();
            ctx.moveTo(0, -this.size / 1.5);
            ctx.lineTo(-this.size / 1.5, this.size / 1.5);
            ctx.lineTo(this.size / 1.5, this.size / 1.5);
            ctx.closePath();
            ctx.stroke();
            ctx.restore();
        }

        update(tx, ty) {
            if (!this.isPlayer) {
                this.x += this.vx;
                this.y += this.vy;
                if (this.x < 0 || this.x > width) this.vx *= -1;
                if (this.y < 0 || this.y > height) this.vy *= -1;

                // 點擊閃避邏輯
                if (mouse.active) {
                    const dx = this.x - tx;
                    const dy = this.y - ty;
                    const dist = Math.sqrt(dx * dx + dy * dy);
                    if (dist < 250) {
                        this.vx = (dx / dist) * 15;
                        this.vy = (dy / dist) * 15;
                    }
                }
                this.vx *= 0.95; // 摩擦力
                this.vy *= 0.95;
            }
        }
    }

    class Fragment {
        constructor(x, y, isWhite = false) {
            this.x = x;
            this.y = y;
            this.vx = rand(-10, 10);
            this.vy = rand(-10, 10);
            this.life = 1.0;
            this.color = isWhite ? `rgba(255,255,255,${rand(0.5, 1)})` : `rgba(${rand(20, 50)},0,0,${rand(0.2, 0.5)})`;
            this.w = rand(1, 4);
        }
        draw() {
            ctx.strokeStyle = this.color;
            ctx.lineWidth = this.w;
            ctx.beginPath();
            ctx.moveTo(this.x, this.y);
            ctx.lineTo(this.x + this.vx * 2, this.y + this.vy * 2);
            ctx.stroke();
        }
        update() {
            this.x += this.vx;
            this.y += this.vy;
            this.life -= 0.02;
            return this.life > 0;
        }
    }

    function initAudio() {
        if (audioCtx) return;
        audioCtx = new (window.AudioContext || window.webkitAudioContext)();
        const bufferSize = audioCtx.sampleRate * 2;
        const noiseBuffer = audioCtx.createBuffer(1, bufferSize, audioCtx.sampleRate);
        const output = noiseBuffer.getChannelData(0);
        for (let i = 0; i < bufferSize; i++) output[i] = Math.random() * 2 - 1;

        const whiteNoise = audioCtx.createBufferSource();
        whiteNoise.buffer = noiseBuffer;
        whiteNoise.loop = true;

        audioFilter = audioCtx.createBiquadFilter();
        audioFilter.type = 'lowpass';
        audioGain = audioCtx.createGain();
        audioGain.gain.value = 0.05;

        whiteNoise.connect(audioFilter);
        audioFilter.connect(audioGain);
        audioGain.connect(audioCtx.destination);
        whiteNoise.start();
    }

    const playerTri = new Triangle(true);
    for (let i = 0; i < 60; i++) triangles.push(new Triangle(false));

    function animate() {
        frameCount++;
        // 隨機背景閃爍增加焦慮感
        if (Math.random() > 0.98) {
            ctx.fillStyle = 'rgba(255,255,255,0.05)';
        } else {
            ctx.fillStyle = 'rgba(0, 0, 0, 0.25)';
        }
        ctx.fillRect(0, 0, width, height);

        // 自動生成背景混亂線條
        for (let i = 0; i < 3; i++) {
            fragments.push(new Fragment(rand(0, width), rand(0, height), false));
        }

        fragments = fragments.filter(f => f.update());
        fragments.forEach(f => f.draw());

        triangles.forEach(t => {
            t.update(mouse.x, mouse.y);
            t.draw();
        });

        playerTri.draw(mouse.x, mouse.y);

        // 滑鼠移動殘影
        if (Math.abs(mouse.x - mouse.lastX) > 5) {
            for (let i = 0; i < 5; i++) {
                fragments.push(new Fragment(mouse.x, mouse.y, true));
            }
            mouse.lastX = mouse.x;
        }

        requestAnimationFrame(animate);
    }

    window.addEventListener('mousemove', (e) => {
        if (!audioCtx) initAudio();
        mouse.x = e.clientX;
        mouse.y = e.clientY;
        audioFilter.frequency.setValueAtTime(rand(200, 1000), audioCtx.currentTime);
    });

    window.addEventListener('mousedown', () => {
        mouse.active = true;
        if (audioCtx) {
            audioGain.gain.setTargetAtTime(0.3, audioCtx.currentTime, 0.05);
            audioFilter.frequency.setTargetAtTime(5000, audioCtx.currentTime, 0.05);
        }
        // 點擊爆發大量碎片
        for (let i = 0; i < 50; i++) fragments.push(new Fragment(mouse.x, mouse.y, true));
    });

    window.addEventListener('mouseup', () => {
        mouse.active = false;
        if (audioCtx) {
            audioGain.gain.setTargetAtTime(0.05, audioCtx.currentTime, 0.2);
            audioFilter.frequency.setTargetAtTime(400, audioCtx.currentTime, 0.2);
        }
    });

    animate();
</script>
</body>
