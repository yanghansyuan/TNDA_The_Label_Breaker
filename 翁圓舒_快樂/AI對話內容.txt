const colors = [
    "#FFD84D", // 黃色爆炸
    "#6EC5FF", // 藍氣泡
    "#58D68D", // 綠流動
    "#FF7A59", // 橘雲
    "#4A90E2"
];


// 粒子類型
class Particle {
    constructor(x, y, type){
        this.x = x;
        this.y = y;
        this.type = type;
        this.size = Math.random() * 20 + 10;
        this.speedX = (Math.random() - 0.5) * 3;
        this.speedY = (Math.random() - 0.5) * 3;
        this.life = 100;
        this.color = colors[Math.floor(Math.random() * colors.length)];
    }

    update(){
        this.x += this.speedX;
        this.y += this.speedY;
        this.life--;
    }

    draw(){
        ctx.save();
        ctx.globalAlpha = this.life / 100;

        if(this.type === "bubble"){
            ctx.beginPath();
            ctx.arc(this.x, this.y, this.size, 0, Math.PI*2);
            ctx.strokeStyle = this.color;
            ctx.lineWidth = 3;
            ctx.stroke();
        }

        if(this.type === "burst"){
            ctx.beginPath();
            for(let i=0;i<10;i++){
                let angle = (Math.PI*2/10)*i;
                let r = this.size + Math.random()*10;
                ctx.lineTo(this.x + Math.cos(angle)*r,
                           this.y + Math.sin(angle)*r);
            }
            ctx.closePath();
            ctx.fillStyle = this.color;
            ctx.fill();
        }

        if(this.type === "line"){
            ctx.beginPath();
            ctx.moveTo(this.x, this.y);
            ctx.lineTo(this.x + this.size*2, this.y + this.size);
            ctx.strokeStyle = this.color;
            ctx.lineWidth = 2;
            ctx.stroke();
        }

        if(this.type === "cloud"){
            ctx.beginPath();
            ctx.ellipse(this.x, this.y, this.size*2, this.size, 0, 0, Math.PI*2);
            ctx.fillStyle = this.color;
            ctx.fill();
        }

        ctx.restore();
    }
}

function createHappy(x, y){

        p.update();
        p.draw();
        if(p.life <= 0){
            particles.splice(index,1);
        }
    });

    requestAnimationFrame(animate);
}

// 修正殘影（讓畫面更柔）
function preventTrailing(){
    ctx.fillStyle = "rgba(255,250,240,0.3)";
    ctx.fillRect(0,0,width,height);
}

animate();
