        class WaveLine {
            constructor(x, y, color) {
                this.x = x;
                this.y = y;
                this.color = color;
                this.angle = Math.random() * Math.PI * 2;
                this.velocity = {
                    x: (Math.random() - 0.5) * 4,
                    y: (Math.random() - 0.5) * 4

                this.isLinked = false;
                this.linkedTo = [];

                // 1. 輕飄飄與天馬行空的移動
                this.x += this.velocity.x;
                this.y += this.velocity.y;
                this.phase += this.frequency;

                // 邊界反彈 (驚喜反彈)

                if (this.y < 0 || this.y > canvas.height) this.velocity.y *= -1;

                        if (other !== this && !other.isLinked && this.linkedTo.length < 2) {
                            const dx = this.x - other.x;
                            const dy = this.y - other.y;
                            const dist = Math.sqrt(dx*dx + dy*dy);
                            
                            if (dist < 50) { // 磁吸距離
                                this.linkedTo.push(other);
                                other.isLinked = true;
                                if (this.linkedTo.length === 2) this.isLinked = true;
                            }
                        }
                    });
                }
            }

            draw() {
                ctx.save();
                ctx.shadowBlur = 15;
                ctx.shadowColor = this.color;
                ctx.strokeStyle = this.color;
                ctx.lineWidth = 8;
                ctx.lineCap = 'round';

                    ctx.stroke();

        function initBackgroundDots() {
            backgroundDots = [];
            for(let i=0; i<30; i++) {
                backgroundDots.push({
                    x: Math.random() * canvas.width,
                    y: Math.random() * canvas.height,
                    vx: (Math.random()-0.5) * 1,
                    vy: (Math.random()-0.5) * 1,
                    size: Math.random() * 3 + 1

        class LiquidLine {
            constructor(x, y, color) {
                this.x = x;
                this.y = y;
                this.color = color;
                this.vx = (Math.random() - 0.5) * 3;
                this.vy = (Math.random() - 0.5) * 3;
                
                // 4. 線條長度增加到約 5 公分 (約 180px)
                this.maxLength = 25; 
                this.points = [];
                
                // 5. 線條粗細隨總數增加
                this.baseWidth = 6 + (particles.length * 0.2); 
                
                this.angle = Math.random() * Math.PI * 2;
                this.waveSpeed = 0.05 + Math.random() * 0.1;
                this.amplitude = 10 + Math.random() * 10;
                
                this.group = null; // 用於儲存吸附後的群組
                this.groupAngleOffset = 0;
            }

            update() {
                if (!this.group) {
                    this.x += this.vx;
                    this.y += this.vy;

                    // 邊界反彈
                    if (this.x < 0 || this.x > canvas.width) this.vx *= -1;
                    if (this.y < 0 || this.y > canvas.height) this.vy *= -1;
                } else {
                    // 吸附後的運動邏輯：繞中心點轉動
                    this.groupAngleOffset += 0.02;
                    const finalAngle = this.group.baseAngle + this.groupAngleOffset;
                    this.x = this.group.center.x;
                    this.y = this.group.center.y;
                    this.angle = finalAngle;
                }

                this.points.unshift({ x: this.x, y: this.y });
                if (this.points.length > this.maxLength) this.points.pop();

                // 檢測吸附邏輯
                if (!this.group) {
                    this.checkCollision();
                }
            }

            checkCollision() {
                let nearby = particles.filter(p => p !== this && !p.group && 
                    Math.hypot(this.x - p.x, this.y - p.y) < 60);

                if (nearby.length >= 2) {
                    // 湊齊三條線，形成驚喜球
                    const groupCenter = { x: this.x, y: this.y };
                    const groupObj = { center: groupCenter, baseAngle: 0 };
                    
                    this.group = { ...groupObj, baseAngle: 0 };
                    nearby[0].group = { ...groupObj, baseAngle: Math.PI * 2 / 3 };
                    nearby[1].group = { ...groupObj, baseAngle: Math.PI * 4 / 3 };
                }

            draw() {

                ctx.shadowColor = 'rgba(0,0,0,0.1)';

                ctx.lineWidth = this.baseWidth;
                ctx.lineCap = 'round';
                ctx.lineJoin = 'round';

                ctx.beginPath();

                    const wave = Math.sin(frameCount * this.waveSpeed + i * 0.3) * this.amplitude;
                    // 如果有群組，朝向固定角度
                    const drawX = p.x + (this.group ? Math.cos(this.angle) * i * 8 : wave);
                    const drawY = p.y + (this.group ? Math.sin(this.angle) * i * 8 : wave);
                    
                    if (i === 0) ctx.moveTo(drawX, drawY);
                    else ctx.lineTo(drawX, drawY);
                });
                ctx.stroke();

                // 加上一點玻璃高光
                ctx.strokeStyle = 'rgba(255,255,255,0.4)';
                ctx.lineWidth = this.baseWidth / 3;
                ctx.stroke();
                
                ctx.restore();
            }
        }

            const x = e.clientX || (e.touches && e.touches[0].clientX);
            const y = e.clientY || (e.touches && e.touches[0].clientY);
            
            if (isDrawing) {

                }

        }

        window.addEventListener('mouseup', () => isDrawing = false);
        window.addEventListener('mousemove', handleInput);
        window.addEventListener('touchstart', (e) => { isDrawing = true; });
        window.addEventListener('touchmove', handleInput);

        function animate() {
            ctx.clearRect(0, 0, canvas.width, canvas.height);
            frameCount++;

            // 繪製背景蝴蝶點點
            backgroundDots.forEach(dot => {
                dot.x += dot.vx + Math.sin(frameCount * 0.02); // 蝴蝶扇動感

消消樂鎖定：

我加入了一個 isEliminationMode 布林值。

當它為 true 時，程式碼會跳過 checkCollision()（磁吸功能），這樣剩下的單線就不會再結合成新球。

同時開啟 checkGroupElimination()，讓現有的球開始「互碰消失」。

歸零重啟機制：animate() 函式每一幀都會檢查 currentGroupCount。只有當數量精確回到 0 時，才會將模式切換回磁吸模式，確保循環正確。

這是在開發複雜 Canvas 動畫時常見的「邏輯衝突」與「效能瓶頸」。

問題診斷：

抽動原因：之前的代碼在 update 階段同時進行了「座標更新」與「碰撞檢查」，且在消消樂模式下，多個點同時偵測到碰撞會反覆標記 markForDeletion，造成運算量瞬間暴增，導致畫面卡頓（微秒級的當機感）。

線條跳動：因為「風力」與「中點旋轉」的邏輯在座標計算上沒有進行平滑處理（Smoothing），當粒子數過多時，物理引擎的反應速度跟不上渲染速度。

優化策略：

物理分層：將「移動」與「碰撞偵測」完全分開，減少重複運算。

預處理寬度：不再每一幀對每一條線重新計算寬度，改用全域變數快取，解決視覺上的抽動。

效能優化：增加 particles.length 的嚴格限制，並使用 offscreen-canvas 的概念優化繪圖。

        class LiquidLine {
            constructor(x, y, color, isSpread = false) {
                const spread = isSpread ? 65 : 0;
                this.x = x + (Math.random() - 0.5) * spread;
                this.y = y + (Math.random() - 0.5) * spread;
                
                this.color = color;
                const speedBase = 2 + Math.random() * 1.5;
                this.vx = (Math.random() - 0.5) * speedBase; 
                this.vy = (Math.random() - 0.5) * speedBase;
                
                this.points = [];
                this.maxLength = 22; 
                this.group = null;
                this.groupAngle = 0;
                this.markForDeletion = false;
            }

            updatePhysics(currentGroups) {
                const windX = Math.sin(frameCount * 0.008) * 0.45;
                const windY = Math.cos(frameCount * 0.01) * 0.45;
                const margin = isEliminationMode ? 60 : 35;

                if (!this.group) {
                    this.x += this.vx + windX;
                    this.y += this.vy + windY;
                    if (this.x < margin || this.x > canvas.width - margin) this.vx *= -1;
                    if (this.y < margin || this.y > canvas.height - margin) this.vy *= -1;
                } else {
                    // --- 彩蛋剩 2 個時的互相吸引邏輯 ---
                    if (isEliminationMode && currentGroups.length === 2) {
                        let otherGroup = currentGroups.find(g => g.id !== this.group.id);
                        if (otherGroup) {
                            const dx = otherGroup.center.x - this.group.center.x;
                            const dy = otherGroup.center.y - this.group.center.y;
                            const dist = Math.hypot(dx, dy);
                            
                            // 產生向心加速度
                            const pullForce = 0.5; 
                            this.group.vx += (dx / dist) * pullForce;
                            this.group.vy += (dy / dist) * pullForce;
                            
                            // 限制最高衝撞速度
                            const maxChaseSpeed = 12;
                            const speed = Math.hypot(this.group.vx, this.group.vy);
                            if (speed > maxChaseSpeed) {
                                this.group.vx = (this.group.vx / speed) * maxChaseSpeed;
                                this.group.vy = (this.group.vy / speed) * maxChaseSpeed;
                            }
                        }
                    }

                    this.group.center.x += this.group.vx + windX;
                    this.group.center.y += this.group.vy + windY;

                    if (this.group.center.x < margin || this.group.center.x > canvas.width - margin) this.group.vx *= -1;
                    if (this.group.center.y < margin || this.group.center.y > canvas.height - margin) this.group.vy *= -1;
                    
                    this.x = this.group.center.x;
                    this.y = this.group.center.y;

                                id: Math.random()
                            };

                            this.groupAngle = 0;
                            nearby[0].group = sharedGroup;
                            nearby[0].groupAngle = Math.PI / 3; 
                            nearby[1].group = sharedGroup;
                            nearby[1].groupAngle = Math.PI * 2 / 3;
                        }
                    }
                } else if (this.group) {
                    for (let other of allParticles) {
                        if (other.group && other.group.id !== this.group.id) {
                            const dist = Math.hypot(this.group.center.x - other.group.center.x, this.group.center.y - other.group.center.y);
                            if (dist < 135) { 
                                this.markGroupDeleted(this.group.id, allParticles);
                                this.markGroupDeleted(other.group.id, allParticles);
                                break;
                            }
                        }
                    }
                }
            }

            markGroupDeleted(id, allParticles) {
                allParticles.forEach(p => {
                    if (p.group && p.group.id === id) p.markForDeletion = true;
                });
            }

            draw() {
                if (this.markForDeletion || this.points.length < 2) return;

                ctx.save();
                ctx.strokeStyle = this.color;
                ctx.lineWidth = globalWidth;
                ctx.lineCap = 'round';
                ctx.lineJoin = 'round';
                ctx.shadowColor = 'rgba(0,0,0,0.1)';
                ctx.shadowBlur = 5;
                ctx.shadowOffsetY = 4;

                ctx.beginPath();
                for(let i=0; i<this.points.length; i++) {
                    const p = this.points[i];
                    let drawX, drawY;

                    if (this.group) {
                        const offset = (i - this.maxLength / 2) * 10; 
                        drawX = p.x + Math.cos(this.groupAngle) * offset;
                        drawY = p.y + Math.sin(this.groupAngle) * offset;
                    } else {
                        const wave = Math.sin(frameCount * 0.1 + i * 0.3) * 10;
                        drawX = p.x + wave;
                        drawY = p.y + wave;
                    }
                    
                    if (i === 0) ctx.moveTo(drawX, drawY);
                    else ctx.lineTo(drawX, drawY);
                }
                ctx.stroke();

                ctx.strokeStyle = 'rgba(255,255,255,0.42)';
                ctx.lineWidth = globalWidth / 2.6;
                ctx.stroke();
                ctx.restore();
            }
        }

        function handleInput(e) {
            if (!isDrawing) return;
            const x = e.clientX || (e.touches && e.touches[0].clientX);
            const y = e.clientY || (e.touches && e.touches[0].clientY);
            
            const throttleRate = isEliminationMode ? 16 : 8;

        }

            particles.forEach(p => p.draw());

        }

                ctx.strokeStyle = this.color;
                const breathingWidth = baseGlobalWidth + Math.sin(frameCount * 0.15 + this.jitterSeed) * 1.5;
                ctx.lineWidth = Math.max(2, breathingWidth);
                ctx.lineCap = 'round';
                ctx.lineJoin = 'round';
                ctx.shadowColor = 'rgba(0,0,0,0.1)';
                ctx.shadowBlur = 4;
                ctx.shadowOffsetY = 3;

                ctx.beginPath();
                for(let i=0; i<this.points.length; i++) {
                    const p = this.points[i];
                    // 3. 全時快樂抖動：幅度微增，頻率變快
                    const jitterX = Math.sin(frameCount * this.wobbleSpeed + i * 0.4 + this.jitterSeed) * 3.5;
                    const jitterY = Math.cos(frameCount * (this.wobbleSpeed * 1.3) + i * 0.3 + this.jitterSeed) * 3.5;

                    let drawX, drawY;

                    if (this.group) {
                        const offset = (i - this.maxLength / 2) * 11; 
                        drawX = p.x + Math.cos(this.groupAngle) * offset + jitterX;
                        drawY = p.y + Math.sin(this.groupAngle) * offset + jitterY;
                    } else {
                        const wave = Math.sin(frameCount * 0.15 + i * 0.4 + this.jitterSeed) * 12;
                        drawX = p.x + wave + jitterX;
                        drawY = p.y + wave + jitterY;
                    }
                    
                    if (i === 0) ctx.moveTo(drawX, drawY);
                    else ctx.lineTo(drawX, drawY);
                }
                ctx.stroke();

                ctx.strokeStyle = 'rgba(255,255,255,0.48)';
                ctx.lineWidth = Math.max(1, ctx.lineWidth / 2.5);
                ctx.stroke();
                ctx.restore();
            }
        }

        function handleInput(e) {
            if (!isDrawing) return;
            const x = e.clientX || (e.touches && e.touches[0].clientX);
            const y = e.clientY || (e.touches && e.touches[0].clientY);
            
            const throttleRate = isEliminationMode ? 14 : 7; // 生成頻率稍微變快
            if (frameCount % throttleRate === 0) {
                particles.push(new LiquidLine(x, y, PANTONE_COLORS[Math.floor(Math.random() * PANTONE_COLORS.length)], true));
            }
        }

        window.addEventListener('mousedown', () => isDrawing = true);
        window.addEventListener('mouseup', () => isDrawing = false);
        window.addEventListener('mousemove', handleInput);
        window.addEventListener('touchstart', (e) => { isDrawing = true; handleInput(e); });
        window.addEventListener('touchend', () => isDrawing = false);
        window.addEventListener('touchmove', handleInput);

        function animate() {
            ctx.clearRect(0, 0, canvas.width, canvas.height);
            frameCount++;

            backgroundDots.forEach(dot => {
                dot.x += dot.vx + Math.sin(frameCount * 0.03 + dot.size) * 0.6;
                dot.y += dot.vy;
                if (dot.x < 0) dot.x = canvas.width; if (dot.x > canvas.width) dot.x = 0;
                if (dot.y < 0) dot.y = canvas.height; if (dot.y > canvas.height) dot.y = 0;
                ctx.fillStyle = 'rgba(255, 255, 255, 0.4)';
                ctx.beginPath(); ctx.arc(dot.x, dot.y, dot.size, 0, Math.PI * 2); ctx.fill();
            });

            const groupMap = new Map();
            particles.forEach(p => { if(p.group && !p.markForDeletion) groupMap.set(p.group.id, p.group); });
            const currentGroups = Array.from(groupMap.values());
            const currentGroupCount = currentGroups.length;

            if (!isEliminationMode && currentGroupCount >= 10) isEliminationMode = true;
            else if (isEliminationMode && currentGroupCount === 0) isEliminationMode = false;

            const targetWidth = 8 + (currentGroupCount * 3.2);
            baseGlobalWidth += (targetWidth - baseGlobalWidth) * 0.15;

            particles.forEach(p => p.updatePhysics(currentGroups));
            particles.forEach(p => p.checkCollision(particles));
            
            particles = particles.filter(p => !p.markForDeletion);
            particles.forEach(p => p.draw());

            if (particles.length > 150) particles.splice(0, particles.length - 150);

            requestAnimationFrame(animate);
        }
        animate();
    </script>
