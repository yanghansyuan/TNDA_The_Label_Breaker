(() => {
  const canvas = document.getElementById('c');
  const ctx = canvas.getContext('2d', { alpha: false });

  // ===== Responsive sizing (CSS pixels -> device pixels) =====
  function resize() {
    const dpr = Math.max(1, window.devicePixelRatio || 1);
    const w = Math.floor(window.innerWidth * dpr);
    const h = Math.floor(window.innerHeight * dpr);
    canvas.width = w; canvas.height = h;
    canvas._dpr = dpr;
  }
  window.addEventListener('resize', resize, { passive:true });
  resize();

  // ===== State =====
  let darkMode = false;

  const pointer = {
    x: 0, y: 0, down: false,
    lastAdd: 0
  };

  // Ripples: created by click/drag; warp the contour lines
  const ripples = []; // {x,y,t0,strength,decay,radSpeed}
  const now = () => performance.now() / 1000;

  function addRipple(x, y, strength=1) {
    ripples.push({
      x, y,
      t0: now(),
      strength,
      decay: 1.35 + Math.random()*0.65,
      radSpeed: 260 + Math.random()*120
    });
    // Keep memory bounded
    if (ripples.length > 64) ripples.splice(0, ripples.length - 64);
  }

  // Mouse interaction (left button)
  canvas.addEventListener('pointerdown', (e) => {
    if (e.button !== 0) return;
    canvas.setPointerCapture(e.pointerId);
    pointer.down = true;
    const {x,y} = toCanvas(e.clientX, e.clientY);
    pointer.x = x; pointer.y = y;
    addRipple(x, y, 1.25);
  });

  canvas.addEventListener('pointermove', (e) => {
    const {x,y} = toCanvas(e.clientX, e.clientY);
    pointer.x = x; pointer.y = y;
    if (!pointer.down) return;

    // While dragging, add ripples at a gentle rate
    const t = now();
    if (t - pointer.lastAdd > 0.06) {
      pointer.lastAdd = t;
      addRipple(x, y, 0.8);
    }
  }, { passive:true });

  canvas.addEventListener('pointerup', (e) => {
    pointer.down = false;
    try { canvas.releasePointerCapture(e.pointerId); } catch {}
  });

  // Keyboard
  window.addEventListener('keydown', (e) => {
    if (e.key === ' '){ e.preventDefault(); darkMode = !darkMode; }
    if (e.key.toLowerCase() === 'r'){ ripples.length = 0; }
    if (e.key.toLowerCase() === 's'){
      const a = document.createElement('a');
      a.download = 'emotion-slice.png';
      a.href = canvas.toDataURL('image/png');
      a.click();
    }
  });

  function toCanvas(clientX, clientY){
    const dpr = canvas._dpr || 1;
    return { x: clientX * dpr, y: clientY * dpr };
  }

  // ===== Drawing helpers =====
  function lerp(a,b,t){ return a + (b-a)*t; }
  function clamp(v,a,b){ return Math.max(a, Math.min(b, v)); }
  function smoothstep(a,b,x){
    const t = clamp((x-a)/(b-a), 0, 1);
    return t*t*(3-2*t);
  }

  // Pseudo-noise (deterministic-ish, no libs)
  function hash(n){
    return (Math.sin(n*127.1 + 311.7) * 43758.5453123) % 1;
  }
  function noise1(x){
    const i = Math.floor(x);
    const f = x - i;
    const a = hash(i);
    const b = hash(i+1);
    const u = f*f*(3-2*f);
    return a + (b-a)*u;
  }

  function warpAt(x, y, t){
    let dx = 0, dy = 0;

    for (let i = ripples.length - 1; i >= 0; i--){
      const r = ripples[i];
      const age = t - r.t0;
      const life = Math.exp(-age * r.decay);
      if (life < 0.015) { ripples.splice(i,1); continue; }

      const vx = x - r.x;
      const vy = y - r.y;
      const dist = Math.sqrt(vx*vx + vy*vy) + 1e-6;

      const wave = Math.sin(dist*0.035 - age*4.2) * 0.5 + 0.5; // 0..1
      const falloff = Math.exp(-dist / (240 + 80*noise1(i*3.1))) * life;
      const amp = r.strength * falloff * (0.55 + 0.45*wave);

      const nx = vx / dist, ny = vy / dist;
      const swirl = Math.sin(age*2.3 + dist*0.01) * 0.35;
      dx += (nx - ny*swirl) * amp * 70;
      dy += (ny + nx*swirl) * amp * 70;
    }

    const wob = (noise1(x*0.002 + t*0.4) - 0.5) * 8;
    dx += wob;
    dy += (noise1(y*0.002 + 12 + t*0.35) - 0.5) * 8;

    return {dx, dy};
  }

  // ===== Main scene geometry (inspired by your sketch) =====
  function draw(t){
    const W = canvas.width, H = canvas.height;

    const bg = darkMode ? '#0f1014' : '#f4f1ea';
    ctx.fillStyle = bg;
    ctx.fillRect(0,0,W,H);

    const margin = Math.min(W,H) * 0.08;
    const boxW = W - margin*2;
    const boxH = H - margin*2;

    const cx = W*0.52, cy = H*0.52;
    const mainW = Math.min(boxW, boxH) * 0.95;
    const mainH = mainW * 0.62;

    const capW = mainW * 0.34;
    const capR = mainH * 0.48;

    const rightW = mainW - capW;
    const contourCount = 10;

    const ink = darkMode ? '#e7e2d8' : '#121215';
    const inkSoft = darkMode ? 'rgba(231,226,216,.7)' : 'rgba(18,18,21,.7)';
    const fillInk = darkMode ? '#e7e2d8' : '#121215';

    ctx.lineCap = 'round';
    ctx.lineJoin = 'round';

    ctx.save();
    ctx.translate(cx, cy);
    ctx.rotate(-0.06);
    ctx.translate(-cx, -cy);

    const leftX = cx - mainW/2;
    const rightX = cx + mainW/2;
    const topY0 = cy - mainH/2;
    const botY0 = cy + mainH/2;

    function topY(x){
      const u = (x - leftX) / (rightX - leftX);
      const arch = Math.sin(u*Math.PI) * 0.18;
      const droopRight = Math.pow(u, 2.2) * 0.22;
      return topY0 + mainH*(arch*0.25 + droopRight*0.35);
    }
    function botY(x){
      const u = (x - leftX) / (rightX - leftX);
      const bulge = Math.sin(u*Math.PI) * 0.10;
      const liftRight = Math.pow(u, 2.0) * 0.08;
      return botY0 + mainH*(bulge*0.15 - liftRight*0.10);
    }

    // === Draw solid cap ===
    ctx.save();
    ctx.fillStyle = fillInk;
    ctx.beginPath();
    const capLeft = leftX;
    const capRight = leftX + capW;
    const capTop = topY(capLeft) + mainH*0.02;
    const capBot = botY(capLeft) - mainH*0.02;

    const r = Math.min(capR, (capBot-capTop)/2, capW/2);
    ctx.moveTo(capLeft + r, capTop);
    ctx.lineTo(capRight, capTop);
    ctx.lineTo(capRight, capBot);
    ctx.lineTo(capLeft + r, capBot);
    ctx.arcTo(capLeft, capBot, capLeft, capBot - r, r);
    ctx.arcTo(capLeft, capTop, capLeft + r, capTop, r);
    ctx.closePath();

    ctx.fill();
    ctx.globalCompositeOperation = 'source-atop';
    ctx.strokeStyle = darkMode ? 'rgba(15,16,20,.25)' : 'rgba(244,241,234,.18)';
    ctx.lineWidth = 2.2 * (canvas._dpr||1);
    for (let i=0;i<28;i++){
      const y = lerp(capTop+6, capBot-6, i/27);
      ctx.beginPath();
      const x1 = capLeft + 10 + noise1(i*1.7)*18;
      const x2 = capRight - 12 - noise1(i*2.9)*22;
      ctx.moveTo(x1, y);
      ctx.lineTo(x2, y + (noise1(i*4.3)-0.5)*6);
      ctx.stroke();
    }
    ctx.restore();

    // === Draw contour lines on the right ===
    const startX = leftX + capW*0.92;
    const endX = rightX;

    for (let k=0; k<contourCount; k++){
      const kk = k/(contourCount-1);
      const inset = lerp(0, rightW*0.52, kk);

      const lw = lerp(6.0, 3.2, kk) * (canvas._dpr||1);
      ctx.lineWidth = lw;
      ctx.strokeStyle = ink;

      ctx.beginPath();

      const steps = 140;
      for (let i=0;i<=steps;i++){
        const u = i/steps;
        const x = lerp(startX + inset*0.12, endX - inset, u);

        let yTop = topY(x) + inset*0.18 + mainH*0.02;
        let yBot = botY(x) - inset*0.12 - mainH*0.02;

        const hook = smoothstep(0.72, 1.0, u);
        yTop += hook * inset*0.18;
        yBot += hook * inset*0.38;

        const w = warpAt(x, lerp(yTop,yBot, 0.5), t);
        const bend = (1-kk) * 0.95 + 0.05;

        const jit = (noise1(i*0.08 + k*2.1 + t*0.2) - 0.5) * 1.6 * (canvas._dpr||1);

        const y = yTop + (yBot - yTop) * 0.03 + jit;
        if (i === 0) ctx.moveTo(x + w.dx*bend, y + w.dy*bend);
        else ctx.lineTo(x + w.dx*bend, y + w.dy*bend);
      }

      for (let i=steps;i>=0;i--){
        const u = i/steps;
        const x = lerp(startX + inset*0.12, endX - inset, u);

        let yTop = topY(x) + inset*0.18 + mainH*0.02;
        let yBot = botY(x) - inset*0.12 - mainH*0.02;

        const hook = smoothstep(0.72, 1.0, u);
        yTop += hook * inset*0.18;
        yBot += hook * inset*0.38;

        const w = warpAt(x, lerp(yTop,yBot, 0.5), t);
        const bend = (1-kk) * 0.95 + 0.05;
        const jit = (noise1(i*0.08 + 33 + k*2.1 + t*0.2) - 0.5) * 1.6 * (canvas._dpr||1);

        const y = yBot - (yBot - yTop) * 0.03 + jit;
        ctx.lineTo(x + w.dx*bend, y + w.dy*bend);
      }

      ctx.closePath();
      ctx.stroke();
    }

    // Outer silhouette stroke
    ctx.lineWidth = 7.5 * (canvas._dpr||1);
    ctx.strokeStyle = ink;
    ctx.beginPath();

    const outlineSteps = 180;
    for (let i=0;i<=outlineSteps;i++){
      const u = i/outlineSteps;
      const x = lerp(leftX, rightX, u);
      const y = topY(x);
      const w = warpAt(x, y, t);
      if (i===0) ctx.moveTo(x + w.dx*0.8, y + w.dy*0.8);
      else ctx.lineTo(x + w.dx*0.8, y + w.dy*0.8);
    }
    for (let i=0;i<=outlineSteps*0.35;i++){
      const v = i/(outlineSteps*0.35);
      const x = rightX - v*mainW*0.02;
      const y = lerp(topY(rightX), botY(rightX), v);
      const w = warpAt(x, y, t);
      ctx.lineTo(x + w.dx*0.8, y + w.dy*0.8);
    }
    for (let i=outlineSteps;i>=0;i--){
      const u = i/outlineSteps;
      const x = lerp(leftX, rightX, u);
      const y = botY(x);
      const w = warpAt(x, y, t);
      ctx.lineTo(x + w.dx*0.8, y + w.dy*0.8);
    }
    ctx.closePath();
    ctx.stroke();

    // Shadow
    ctx.globalAlpha = darkMode ? 0.25 : 0.18;
    ctx.fillStyle = '#000';
    ctx.beginPath();
    ctx.ellipse(cx + mainW*0.02, cy + mainH*0.48, mainW*0.40, mainH*0.10, -0.03, 0, Math.PI*2);
    ctx.fill();
    ctx.globalAlpha = 1;

    // Right-edge stitch marks
    ctx.strokeStyle = inkSoft;
    ctx.lineWidth = 2.4 * (canvas._dpr||1);
    for (let i=0;i<10;i++){
      const u = i/9;
      const x = rightX - 6*(canvas._dpr||1) - noise1(i*3.7)*12*(canvas._dpr||1);
      const y = lerp(topY(rightX)+mainH*0.22, botY(rightX)-mainH*0.10, u);
      ctx.beginPath();
      ctx.moveTo(x-2*(canvas._dpr||1), y);
      ctx.lineTo(x+2*(canvas._dpr||1), y);
      ctx.stroke();
    }

    ctx.restore();

    requestAnimationFrame(() => draw(now()));
  }

  requestAnimationFrame(() => draw(now()));
})();
