    const field = document.getElementById('field');
    const stage = document.getElementById('stage');

    // 依螢幕大小決定形狀數量（自適應）
    const count = Math.round(Math.max(16, Math.min(42, window.innerWidth / 28)));

    // 三種情緒層：冷靜(藍) / 光(橘) / 壓力滲出(粉)
    const palette = [
      { c: getCSS('--calm'),   w: 0.44 },
      { c: getCSS('--warm'),   w: 0.38 },
      { c: getCSS('--stress'), w: 0.18 },
    ];

    function getCSS(v){
      return getComputedStyle(document.documentElement).getPropertyValue(v).trim();
    }

    function pickColor(){
      const r = Math.random();
      let acc = 0;
      for (const p of palette){
        acc += p.w;
        if (r <= acc) return p.c;
      }
      return palette[0].c;
    }

    function rnd(min,max){ return min + Math.random()*(max-min); }

    function makeShape(i){
      const typeRoll = Math.random();
      const el = document.createElement('div');
      el.classList.add('shape');

      if (typeRoll < 0.5) el.classList.add('dot');
      else if (typeRoll < 0.78) el.classList.add('line');
      else el.classList.add('tri');

      const s = rnd(16, 64) * (window.innerWidth < 720 ? 0.85 : 1);
      const x = rnd(6, 84);             // 留右邊杯子空間
      const y = rnd(10, 92);
      const r = rnd(-28, 28) + 'deg';
      const o = rnd(0.25, 0.85).toFixed(2);
      const b = rnd(0.0, 1.6).toFixed(2) + 'px';
      const t = rnd(3.2, 8.5).toFixed(2) + 's';
      const c = pickColor();

      el.style.setProperty('--s', s + 'px');
      el.style.setProperty('--x', x + 'vw');
      el.style.setProperty('--y', y + 'vh');
      el.style.setProperty('--r', r);
      el.style.setProperty('--o', o);
      el.style.setProperty('--b', b);
      el.style.setProperty('--t', t);
      el.style.setProperty('--c', c);

      // 小小的互動：滑過時，讓「壓力色」更容易被看見
      el.addEventListener('mouseenter', () => {
        if (Math.random() < 0.22) el.style.setProperty('--c', getCSS('--stress'));
      });

      return el;
    }

    function seed(){
      field.innerHTML = '';
      for(let i=0;i<count;i++){
        field.appendChild(makeShape(i));
      }
    }

    // 點擊：丟一顆「壓力泡泡」往下沉
    stage.addEventListener('pointerdown', (e) => {
      const d = document.createElement('div');
      d.className = 'drop';
      d.style.left = e.clientX + 'px';
      d.style.top  = e.clientY + 'px';
      stage.appendChild(d);
      d.addEventListener('animationend', ()=> d.remove());
    });

    let resizeTimer = null;
    window.addEventListener('resize', () => {
      clearTimeout(resizeTimer);
      resizeTimer = setTimeout(seed, 160);
    });

    seed();
  </script>

    const canvas = document.getElementById('c');
    const ctx = canvas.getContext('2d');

    // ====== 自適應：真正改 canvas 像素尺寸（避免模糊）======
    function resize(){
      const dpr = Math.max(1, window.devicePixelRatio || 1);
      canvas.width  = Math.floor(window.innerWidth  * dpr);
      canvas.height = Math.floor(window.innerHeight * dpr);
      canvas.style.width  = window.innerWidth + "px";
      canvas.style.height = window.innerHeight + "px";
      ctx.setTransform(dpr,0,0,dpr,0,0); // 讓繪圖用 CSS 像素座標
    }
    window.addEventListener('resize', resize);
    resize();

    // ====== 情緒色盤（可自己改權重）======
    const COLORS = [
      { c: getCSS('--calm'),   w: 0.45 },
      { c: getCSS('--warm'),   w: 0.35 },
      { c: getCSS('--stress'), w: 0.20 },
    ];
    function getCSS(v){ return getComputedStyle(document.documentElement).getPropertyValue(v).trim(); }
    function pickColor(){
      const r = Math.random();
      let acc = 0;
      for(const p of COLORS){ acc += p.w; if(r <= acc) return p.c; }
      return COLORS[0].c;
    }
    const rnd = (a,b)=> a + Math.random()*(b-a);

    // ====== 物件（點/線/三角形）======
    const shapes = [];
    const MAX = 220; // 避免太多卡

    function spawn(x,y, intensity=1){
      if(shapes.length > MAX) shapes.splice(0, shapes.length - MAX);

      const typeRoll = Math.random();
      const type = (typeRoll < 0.52) ? 'dot' : (typeRoll < 0.80) ? 'line' : 'tri';

      // 依螢幕寬度自動調整大小
      const scale = Math.min(1.0, Math.max(0.75, window.innerWidth / 1200));
      const sizeBase = (type === 'line') ? rnd(26, 90) : rnd(16, 64);
      const s = sizeBase * scale * rnd(0.9, 1.15);

      shapes.push({
        type,
        x, y,
        vx: rnd(-0.25,0.25) * intensity,
        vy: rnd(-0.22,0.22) * intensity,
        r: rnd(-Math.PI, Math.PI),
        vr: rnd(-0.012, 0.012),
        s,
        o: rnd(0.25, 0.9),
        blur: rnd(0, 1.4),
        c: pickColor(),
        life: 1, // 會慢慢淡出（但不會一下消失）
      });
    }

    // 初始鋪一些
    function seed(){
      shapes.length = 0;
      const n = Math.round(Math.max(18, Math.min(46, window.innerWidth / 26)));
      for(let i=0;i<n;i++){
        spawn(rnd(40, window.innerWidth * 0.82), rnd(60, window.innerHeight - 60), 0.8);
      }
    }
    seed();

        if(s.x < 10) { s.x = 10; s.vx *= -1; }
        if(s.x > rightLimit) { s.x = rightLimit; s.vx *= -1; }
        if(s.y < 10) { s.y = 10; s.vy *= -1; }
        if(s.y > h-10) { s.y = h-10; s.vy *= -1; }

        const dx = s.x - mouse.x;
        const dy = s.y - mouse.y;
        const dist = Math.hypot(dx,dy);
        const radius = Math.max(90, s.s * 2.2);

        if(dist < radius){
          const f = (1 - dist / radius);
          const nx = dx / (dist || 1);
          const ny = dy / (dist || 1);

          // 推開 + 輕微抖動
          s.vx += nx * f * 0.65;
          s.vy += ny * f * 0.65;

          // 互動時變亮（life > 1 的感覺）
          s.life = Math.min(1.25, s.life + f * 0.06);

          // 偶爾把顏色拉向壓力色（像滲出）
          if(Math.random() < 0.010 * f){
            s.c = getCSS('--stress');
          }
        } else {
          // 慢慢回到正常亮度
          s.life += (1 - s.life) * 0.03;
        }

        // 阻尼（避免越推越飛）
        s.vx *= 0.94;
        s.vy *= 0.94;

        // 很慢的淡出（保持畫面乾淨）
        s.o *= 0.9996;
      }

      // 畫形狀
      // 先畫淡的再畫亮的（視覺更乾淨）
      shapes
        .slice()
        .sort((a,b)=> (a.life*a.o) - (b.life*b.o))
        .forEach(drawShape);

      // 清掉太淡的
      for(let i=shapes.length-1;i>=0;i--){
        if(shapes[i].o < 0.08) shapes.splice(i,1);
      }

      requestAnimationFrame(step);
    }
    step();

      ctx.arcTo(x, y+h, x, y, rr);
      ctx.arcTo(x, y, x+w, y, rr);
      ctx.closePath();

        ctx.fill();

      ctx.restore();
      ctx.globalCompositeOperation = 'source-over';

    function update(t, beat){
      const w = window.innerWidth, h = window.innerHeight;
      const rightLimit = w * 0.86; // 留杯子空間

      // shapes
      for(const s of shapes){
        // 生成瞬間的 pop 慢慢回去

        if(s.y > h-10) { s.y = h-10; s.vy *= -0.92; }

        const dy = s.y - mouse.y;
        const dist = Math.hypot(dx,dy);
        const radius = Math.max(120, s.s * 2.4);

          const nx = dx / (dist || 1);
          const ny = dy / (dist || 1);

          // 推開 + 旋轉感

          s.vy += ny * f * (0.95 + 0.65*beat);

          // 被碰到更亮（像情緒被觸發）
          s.life = Math.min(1.25, s.life + f * 0.06);

              x: s.x + rnd(-4,4),
              y: s.y + rnd(-4,4),
              vx: rnd(-1.8,1.8),
              vy: rnd(-1.8,1.8),
              r: rnd(1.2,2.8),
              a: 1,
              drag: rnd(0.90, 0.95),
              hueBias: s.hueBias + rnd(-30,30),
            });
          }
        } else {
          s.life += (1 - s.life) * 0.03;
        }

        // 阻尼
        s.vx *= 0.92;
        s.vy *= 0.92;

        // 緩慢淡出，保持乾淨
        s.o *= 0.9994;
      }

      // sparks
      for(const p of sparks){
        p.x += p.vx;
        p.y += p.vy;
        p.vx *= p.drag;
        p.vy *= p.drag;
        p.a  *= 0.94;
        // 重力一點點
        p.vy += 0.02;
      }

      // 清掉太淡的
      for(let i=shapes.length-1;i>=0;i--){
        if(shapes[i].o < 0.07) shapes.splice(i,1);
      }
      for(let i=sparks.length-1;i>=0;i--){
        if(sparks[i].a < 0.04) sparks.splice(i,1);
      }

      // 限制上限
      if(sparks.length > MAX_SPARKS) sparks.splice(0, sparks.length - MAX_SPARKS);
      if(shapes.length > MAX_SHAPES) shapes.splice(0, shapes.length - MAX_SHAPES);
    }

    function frame(){
      const t = performance.now()/1000;

      // beat：0~1（更像音樂呼吸）

      drawSparks(t, beat);

      lastT = t;
      requestAnimationFrame(frame);
    }

        shapes[i].hueBias += rnd(-20, 20);
      }
    });

      ctx.arc(p.x, p.y, p.r, 0, Math.PI*2);
      ctx.fill();
    }
    ctx.restore();
    ctx.globalCompositeOperation = 'source-over';
  }

  function drawWaves(ctx, t, beat){
    ctx.save();
    ctx.globalCompositeOperation = 'screen';

    for(let i=0;i<wvN;i++){
      const w = waves[i];
      // dir=+1 擴散：半徑從小到大；dir=-1 收縮：半徑從大到小
      const tt = w.t;
      const maxR = Math.min(W,H)*0.55;
      const r = (w.dir>0)
        ? (tt*maxR)
        : ((1-tt)*maxR);

      const col = neon(t, w.bias);
      const alpha = (1-tt) * (0.55 + 0.25*beat);

      ctx.globalAlpha = alpha;
      ctx.lineWidth = 2 + 6*(0.6+beat);
      ctx.shadowBlur = 22*(0.8+beat);
      ctx.shadowColor = col;
      ctx.strokeStyle = col.replace('hsl','hsla').replace(')', ` / ${alpha})`);

      ctx.beginPath();
      ctx.arc(w.x, w.y, Math.max(6,r), 0, Math.PI*2);
      ctx.stroke();
    }

    ctx.restore();
    ctx.globalCompositeOperation = 'source-over';
  }

  function roundRect(ctx,x,y,w,h,r){
    const rr = Math.min(r, w/2, h/2);
    ctx.beginPath();
    ctx.moveTo(x+rr,y);
    ctx.arcTo(x+w,y, x+w,y+h, rr);
    ctx.arcTo(x+w,y+h, x,y+h, rr);
    ctx.arcTo(x,y+h, x,y, rr);
    ctx.arcTo(x,y, x+w,y, rr);
    ctx.closePath();
  }

  // ====== Update physics (含收束結尾) ======
  function update(t, beat){
    const rightLimit = W*0.92; // 保留右側 HUD 空間即可
    const settlePull = settling ? (0.35 + 0.65*beat) : 0;

    // shapes
    for(let i=0;i<shN;i++){
      const s = shapes[i];

      // pop decay
      s.pop += (0 - s.pop) * 0.12;

      // base motion + rhythm
      s.x += s.vx + Math.sin(t*1.2 + s.bias)*0.08*beat;
      s.y += s.vy + Math.cos(t*1.1 + s.bias)*0.06*beat;
      s.rot += s.vr;

      // mouse push
      const dx = s.x - mouse.x;
      const dy = s.y - mouse.y;
      const dist = Math.hypot(dx,dy);
      const radius = Math.max(120, s.size*2.2);

      if(dist < radius){
        const f = (1 - dist/radius);
        const nx = dx/(dist||1), ny = dy/(dist||1);
        s.vx += nx * f * (0.85 + 0.55*beat);
        s.vy += ny * f * (0.85 + 0.55*beat);
        s.life = Math.min(1.25, s.life + f*0.05);
        if(Math.random() < 0.008*f){
          addSpark(s.x+rnd(-3,3), s.y+rnd(-3,3), rnd(-1.4,1.4), rnd(-1.4,1.4), rnd(1.0,2.2), 0.85, s.bias+rnd(-25,25));
        }
      } else {
        s.life += (1 - s.life) * 0.03;
      }

      // release "closure": pull inward + fade quicker (句點感)
      if(settling){
        const sx = settleX - s.x;
        const sy = settleY - s.y;
        const d2 = Math.hypot(sx,sy) || 1;
        const px = sx/d2, py = sy/d2;
        s.vx += px * settlePull * 0.9;
        s.vy += py * settlePull * 0.9;
        s.a  *= 0.996;        // 更快安靜
      } else {
        s.a *= 0.9992;        // 平常慢慢淡
      }

      // damping
      s.vx *= 0.92;
      s.vy *= 0.92;

      // bounds
      if(s.x < 10){ s.x=10; s.vx*=-0.92; }
      if(s.x > rightLimit){ s.x=rightLimit; s.vx*=-0.92; }
      if(s.y < 10){ s.y=10; s.vy*=-0.92; }
      if(s.y > H-10){ s.y=H-10; s.vy*=-0.92; }
    }

    // sparks
    for(let i=0;i<spN;i++){
      const p = sparks[i];
      p.x += p.vx;
      p.y += p.vy;
      p.vx *= p.drag;
      p.vy *= p.drag;
      p.vy += 0.02;      // tiny gravity
      p.a  *= settling ? 0.90 : 0.93; // 收束時更快消失（乾淨）
    }

    // waves progress
    for(let i=0;i<wvN;i++){
      const w = waves[i];
      w.t += 0.035 + 0.03*beat; // 速度跟律動
    }

    // cleanup (in-place compact)
    // shapes
    let j=0;
    for(let i=0;i<shN;i++){
      const s = shapes[i];
      if(s.a > 0.07){
        shapes[j++] = s;
      }
    }
    shN = j;

    // sparks
    j=0;
    for(let i=0;i<spN;i++){
      const p = sparks[i];
      if(p.a > 0.05){
        sparks[j++] = p;
      }
    }
    spN = j;

    // waves
    j=0;
    for(let i=0;i<wvN;i++){
      const w = waves[i];
      if(w.t < 1){
        waves[j++] = w;
      }
    }
    wvN = j;

    // settle decay
    if(settling){
      settling *= 0.94; // 很快回到 0（結束感）
      if(settling < 0.02) settling = 0;
    }

    // flash decay
    flash *= 0.88;
    if(flash < 0.01) flash = 0;

    // shake decay
    screenShake *= 0.85;
    if(screenShake < 0.2) screenShake = 0;
  }

  // ====== Render pipeline ======
  function composite(){
    // world -> main
    mctx.clearRect(0,0,W,H);

    // 左半像素化：把 world 左半縮到 pix，再放大貼到 main 左半
    const pw = pix.width, ph = pix.height;

    // 把 world 左半縮到 pix（低解析度）
    pctx.clearRect(0,0,pw,ph);
    pctx.drawImage(
      world,
      0, 0, MID*DPR, H*DPR,     // source in device pixels
      0, 0, pw, ph              // dest low-res
    );

    // 放大回 main 左半（關閉平滑 => 像素風）
    mctx.save();
    mctx.imageSmoothingEnabled = false;
    mctx.drawImage(pix, 0,0,pw,ph, 0,0, MID,H);
    mctx.restore();

    // 右半正常貼
    mctx.imageSmoothingEnabled = true;
    mctx.drawImage(
      world,
      MID*DPR, 0, (W-MID)*DPR, H*DPR,
      MID, 0, (W-MID), H
    );

    // 中線：分割提示（很淡）
    mctx.save();
    mctx.globalAlpha = 0.18;
    mctx.fillStyle = 'rgba(255,255,255,0.55)';
    mctx.fillRect(MID-1, 0, 2, H);
    mctx.restore();
  }

  function render(t, beat){
    // world clear is handled by backdrop fill (trail)
    // screen shake
    let sx=0, sy=0;
    if(screenShake){
      sx = rnd(-1,1) * screenShake;
      sy = rnd(-1,1) * screenShake;
    }

    wctx.save();
    wctx.translate(sx, sy);

    drawBackdrop(wctx, t, beat);

    // sort by brightness (cheap: no sort, but still ok; keep stable for perf)
    // draw shapes
    for(let i=0;i<shN;i++){
      drawShape(wctx, shapes[i], t, beat);
    }

    // sparks + waves
    drawSparks(wctx, t, beat);
    drawWaves(wctx, t, beat);

    // click flash overlay (impact)
    if(flash){
      wctx.save();
      wctx.globalCompositeOperation = 'screen';
      wctx.globalAlpha = flash;
      const g = wctx.createRadialGradient(mouse.x, mouse.y, 0, mouse.x, mouse.y, Math.min(W,H)*0.45);
      g.addColorStop(0, 'rgba(255,255,255,0.70)');
      g.addColorStop(0.25, 'rgba(255,0,220,0.18)');
      g.addColorStop(1, 'rgba(0,0,0,0)');
      wctx.fillStyle = g;
      wctx.fillRect(0,0,W,H);
      wctx.restore();
    }

    // slight vignette (adds focus, cheap)
    wctx.save();
    wctx.globalAlpha = 0.22;
    wctx.fillStyle = 'rgba(0,0,0,1)';
    wctx.globalCompositeOperation = 'multiply';
    const vg = wctx.createRadialGradient(W*0.5, H*0.5, Math.min(W,H)*0.20, W*0.5, H*0.5, Math.min(W,H)*0.85);
    vg.addColorStop(0,'rgba(0,0,0,0)');
    vg.addColorStop(1,'rgba(0,0,0,1)');
    wctx.fillStyle = vg;
    wctx.fillRect(0,0,W,H);
    wctx.restore();

    wctx.restore();

    composite();
  }

  // ====== Seed initial scene ======
  function seed(){
    shN=0; spN=0; wvN=0;
    const n = Math.round(clamp(W/24, 18, 54));
    const t = performance.now()/1000;
    for(let i=0;i<n;i++){
      addShape(rnd(40, W*0.92), rnd(60, H-60), 0.85, t);
    }
  }
  seed();

  // ====== Main loop ======
  function frame(){
    const t = performance.now()/1000;
    const raw = (Math.sin(t*omega) + 1)/2;
    const beat = Math.pow(raw, 1.7); // 更有「砰」的律動

    update(t, beat);
    render(t, beat);

    requestAnimationFrame(frame);
  }
  frame();

})();
