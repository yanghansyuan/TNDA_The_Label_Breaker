HTML
<!DOCTYPE html>
<html lang="zh-TW">
<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>渴望 - 1_test</title>
    <style>
        /* 讓網頁自適應，填滿視窗且沒有捲軸 */
        body, html {
            margin: 0;
            padding: 0;
            width: 100%;
            height: 100%;
            overflow: hidden;
            background-color: #f0ece1; /* 帶點溫暖的紙張底色 */
        }
        canvas {
            display: block;
        }
    </style>
</head>
<body>
    <canvas id="artCanvas"></canvas>

    <script>
        const canvas = document.getElementById('artCanvas');
        const ctx = canvas.getContext('2d');

        // 達成條件 b: 自適應視窗大小
        function resizeCanvas() {
            canvas.width = window.innerWidth;
            canvas.height = window.innerHeight;
        }
        window.addEventListener('resize', resizeCanvas);
        resizeCanvas();

        // 儲存點擊產生的形狀
        let shapes = [];

        // 達成條件 a: 滑鼠左鍵互動 (點擊產生形狀)
        window.addEventListener('mousedown', function(event) {
            // 只響應左鍵 (button === 0)
            if (event.button === 0) {
                shapes.push({
                    x: event.clientX,
                    y: event.clientY,
                    radius: Math.random() * 15 + 10,
                    tailLength: Math.random() * 100 + 50
                });
                draw();
            }
        });

        // 繪製畫面的函數
        function draw() {
            // 每次重畫前清除畫布
            ctx.clearRect(0, 0, canvas.width, canvas.height);

            // 畫出所有點擊產生的「渴望之星」
            shapes.forEach(shape => {
                // 畫紅色的尾巴 (象徵延伸、渴望)
                ctx.beginPath();
                ctx.moveTo(shape.x, shape.y);
                ctx.lineTo(shape.x - shape.tailLength, shape.y + shape.tailLength * 0.5); // 往左下角延伸
                ctx.strokeStyle = '#e63946';
                ctx.lineWidth = 2;
                ctx.stroke();

                // 畫橘色的圓形主體
                ctx.beginPath();
                ctx.arc(shape.x, shape.y, shape.radius, 0, Math.PI * 2);
                ctx.fillStyle = '#f4a261';
                ctx.fill();
                ctx.strokeStyle = '#e76f51';
                ctx.lineWidth = 3;
                ctx.stroke();
            });
            
            // 畫布中央加一點文字提示
            if(shapes.length === 0) {
                ctx.fillStyle = '#888';
                ctx.font = '20px Arial';
                ctx.textAlign = 'center';
                ctx.fillText('點擊滑鼠左鍵，釋放渴望', canvas.width/2, canvas.height/2);
            }
        }

        // 初始繪製
        draw();
    </script>
</body>

請建立一個新的文字檔，命名為 2_floating_world.html （記得這是我們保存過程的重要步驟喔！），然後將下面的程式碼貼進去：

HTML
<!DOCTYPE html>
<html lang="zh-TW">
<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>渴望 - 2_floating_world</title>
    <style>
        body, html {
            margin: 0;
            padding: 0;
            width: 100%;
            height: 100%;
            overflow: hidden;
            background-color: #e8e6df; /* 稍微深一點點的紙張色，讓發光感更明顯 */
        }
        canvas {
            display: block;
        }
    </style>
</head>
<body>
    <canvas id="artCanvas"></canvas>

    <script>
        const canvas = document.getElementById('artCanvas');
        const ctx = canvas.getContext('2d');

        function resizeCanvas() {
            canvas.width = window.innerWidth;
            canvas.height = window.innerHeight;
        }
        window.addEventListener('resize', resizeCanvas);
        resizeCanvas();

        // 從你的草稿中提取的色彩靈感 (水藍、橘紅、正紅、螢光綠)
        const colors = ['#3a86ff', '#f4a261', '#e63946', '#80ed99'];

        // --- 1. 定義「浮游物件」類別 ---
        class FloatingShape {
            constructor(x, y) {
                this.x = x || Math.random() * canvas.width;
                this.y = y || Math.random() * canvas.height;
                // 緩緩飛行的速度 (隨機方向與緩慢速度)
                this.vx = (Math.random() - 0.5) * 1.5;
                this.vy = (Math.random() - 0.5) * 1.5;
                // 旋轉角度與緩慢的旋轉速度
                this.angle = Math.random() * Math.PI * 2;
                this.vAngle = (Math.random() - 0.5) * 0.02;
                
                this.color = colors[Math.floor(Math.random() * colors.length)];
                this.size = Math.random() * 20 + 10;
                // 隨機決定形狀類型：0=圓形, 1=三角形, 2=矩形, 3=帶尾巴的線條
                this.type = Math.floor(Math.random() * 4); 
            }

            update() {
                this.x += this.vx;
                this.y += this.vy;
                this.angle += this.vAngle;

                // 讓物件像在無限空間中，從一邊消失會從另一邊出來
                if (this.x > canvas.width + 50) this.x = -50;
                if (this.x < -50) this.x = canvas.width + 50;
                if (this.y > canvas.height + 50) this.y = -50;
                if (this.y < -50) this.y = canvas.height + 50;
            }

            draw() {
                ctx.save();
                ctx.translate(this.x, this.y);
                ctx.rotate(this.angle);
                ctx.fillStyle = this.color;
                ctx.strokeStyle = this.color;
                ctx.lineWidth = 2;

                ctx.beginPath();
                if (this.type === 0) { // 圓形 (點)
                    ctx.arc(0, 0, this.size / 2, 0, Math.PI * 2);
                    ctx.fill();
                } else if (this.type === 1) { // 鏤空三角形 (面/線)
                    ctx.moveTo(0, -this.size);
                    ctx.lineTo(this.size, this.size);
                    ctx.lineTo(-this.size, this.size);
                    ctx.closePath();
                    ctx.stroke();
                } else if (this.type === 2) { // 實心矩形 (面)
                    ctx.fillRect(-this.size/2, -this.size/2, this.size, this.size * 1.5);
                } else if (this.type === 3) { // 帶尾巴的線條 (你的草稿特色)
                    ctx.arc(0, 0, 5, 0, Math.PI * 2);
                    ctx.fill();
                    ctx.moveTo(0, 0);
                    ctx.lineTo(-this.size * 3, 0); // 長長的尾巴
                    ctx.stroke();
                }
                ctx.restore();
            }
        }

        // --- 2. 定義「粒子特效」類別 ---
        class Particle {
            constructor(x, y, color) {
                this.x = x;
                this.y = y;
                // 爆發的放射狀速度
                const angle = Math.random() * Math.PI * 2;
                const speed = Math.random() * 3 + 1;
                this.vx = Math.cos(angle) * speed;
                this.vy = Math.sin(angle) * speed;
                this.color = color;
                this.size = Math.random() * 4 + 1;
                this.life = 1.0; // 生命週期 1.0 ~ 0
                this.decay = Math.random() * 0.02 + 0.01; // 消失的速度
            }
            update() {
                this.x += this.vx;
                this.y += this.vy;
                this.life -= this.decay;
            }
            draw() {
                ctx.save();
                ctx.globalAlpha = Math.max(0, this.life); // 逐漸透明
                ctx.fillStyle = this.color;
                ctx.beginPath();
                ctx.arc(this.x, this.y, this.size, 0, Math.PI * 2);
                ctx.fill();
                ctx.restore();
            }
        }

        // 儲存畫面上的物件
        const shapes = [];
        const particles = [];

        // 初始先在真空中隨機散落 15 個物件
        for(let i = 0; i < 15; i++) {
            shapes.push(new FloatingShape());
        }

        // --- 3. 滑鼠點擊互動 ---
        window.addEventListener('mousedown', function(event) {
            if (event.button === 0) { // 左鍵
                // 隨機選一個顏色作為這次點擊的爆發色
                const burstColor = colors[Math.floor(Math.random() * colors.length)];
                
                // 產生散出的粒子特效 (15顆小粒子)
                for(let i = 0; i < 15; i++) {
                    particles.push(new Particle(event.clientX, event.clientY, burstColor));
                }

                // 在該位置隨機創造一個新的浮游物件
                shapes.push(new FloatingShape(event.clientX, event.clientY));
            }
        });

        // --- 4. 動畫主迴圈 ---
        function animate() {
            // 清除上一幀的畫面，準備畫新的一幀
            ctx.clearRect(0, 0, canvas.width, canvas.height);

            // 更新並繪製所有浮游物件
            shapes.forEach(shape => {
                shape.update();
                shape.draw();
            });

            // 更新並繪製所有粒子 (並移除生命週期結束的粒子)
            for (let i = particles.length - 1; i >= 0; i--) {
                const p = particles[i];
                p.update();
                if (p.life <= 0) {
                    particles.splice(i, 1); // 刪除死掉的粒子
                } else {
                    p.draw();
                }
            }

            // 呼叫下一幀
            requestAnimationFrame(animate);
        }

        // 啟動動畫
        animate();
    </script>
</body>
</html>

    <script>
        const canvas = document.getElementById('artCanvas');
        const ctx = canvas.getContext('2d');

        function resizeCanvas() {
            canvas.width = window.innerWidth;
            canvas.height = window.innerHeight;
        }
        window.addEventListener('resize', resizeCanvas);
        resizeCanvas();

        const colors = ['#3a86ff', '#f4a261', '#e63946', '#80ed99'];

        // --- 輔助函數：計算兩點距離 ---
        function getDistance(x1, y1, x2, y2) {
            const xDist = x2 - x1;
            const yDist = y2 - y1;
            return Math.sqrt(Math.pow(xDist, 2) + Math.pow(yDist, 2));
        }

        // --- 1. 浮游物件類別 (加入碰撞偵測標籤) ---
        class FloatingShape {
            constructor(x, y) {
                this.x = x || Math.random() * canvas.width;
                this.y = y || Math.random() * canvas.height;
                this.vx = (Math.random() - 0.5) * 2;
                this.vy = (Math.random() - 0.5) * 2;
                this.angle = Math.random() * Math.PI * 2;
                this.vAngle = (Math.random() - 0.5) * 0.02;
                
                this.color = colors[Math.floor(Math.random() * colors.length)];
                this.size = Math.random() * 20 + 15; // 稍微放大一點點
                this.type = Math.floor(Math.random() * 4); 
                
                // 碰撞冷卻時間，避免黏在一起無限觸發特效
                this.collisionCooldown = 0; 
            }

            update() {
                this.x += this.vx;
                this.y += this.vy;
                this.angle += this.vAngle;

                if (this.collisionCooldown > 0) this.collisionCooldown--;

                // 邊界環繞
                if (this.x > canvas.width + 50) this.x = -50;
                if (this.x < -50) this.x = canvas.width + 50;
                if (this.y > canvas.height + 50) this.y = -50;
                if (this.y < -50) this.y = canvas.height + 50;
            }

            draw() {
                ctx.save();
                ctx.translate(this.x, this.y);
                ctx.rotate(this.angle);
                ctx.fillStyle = this.color;
                ctx.strokeStyle = this.color;
                ctx.lineWidth = 2;

                ctx.beginPath();
                if (this.type === 0) { // 圓形 (可碰撞)
                    // 如果剛碰撞過，畫一個微微的光暈
                    if(this.collisionCooldown > 0) {
                        ctx.shadowBlur = 20;
                        ctx.shadowColor = this.color;
                    }
                    ctx.arc(0, 0, this.size / 2, 0, Math.PI * 2);
                    ctx.fill();
                    ctx.shadowBlur = 0; // 重置光暈
                } else if (this.type === 1) { // 鏤空三角形
                    ctx.moveTo(0, -this.size);
                    ctx.lineTo(this.size, this.size);
                    ctx.lineTo(-this.size, this.size);
                    ctx.closePath();
                    ctx.stroke();
                } else if (this.type === 2) { // 實心矩形
                    ctx.fillRect(-this.size/2, -this.size/2, this.size, this.size * 1.5);
                } else if (this.type === 3) { // 帶尾巴的線條
                    ctx.arc(0, 0, 5, 0, Math.PI * 2);
                    ctx.fill();
                    ctx.moveTo(0, 0);
                    ctx.lineTo(-this.size * 3, 0);
                    ctx.stroke();
                }
                ctx.restore();
            }
        }

        // --- 2. 線條狀粒子特效 (碰撞時產生) ---
        class LineParticle {
            constructor(x, y, color) {
                this.x = x;
                this.y = y;
                const angle = Math.random() * Math.PI * 2;
                const speed = Math.random() * 5 + 2;
                this.vx = Math.cos(angle) * speed;
                this.vy = Math.sin(angle) * speed;
                this.color = color;
                this.length = Math.random() * 15 + 5;
                this.life = 1.0;
                this.decay = Math.random() * 0.05 + 0.02;
            }
            update() {
                this.x += this.vx;
                this.y += this.vy;
                this.life -= this.decay;
            }
            draw() {
                ctx.save();
                ctx.globalAlpha = Math.max(0, this.life);
                ctx.strokeStyle = this.color;
                ctx.lineWidth = 2;
                ctx.beginPath();
                ctx.moveTo(this.x, this.y);
                // 畫出速度方向的線條
                ctx.lineTo(this.x - this.vx * 2, this.y - this.vy * 2); 
                ctx.stroke();
                ctx.restore();
            }
        }

        // --- 3. 彗星/隕石類別 (實心三角形 + 拖曳粒子) ---
        class Comet {
            constructor() {
                // 從畫面邊緣隨機產生
                if(Math.random() < 0.5) {
                    this.x = Math.random() < 0.5 ? -50 : canvas.width + 50;
                    this.y = Math.random() * canvas.height;
                } else {
                    this.x = Math.random() * canvas.width;
                    this.y = Math.random() < 0.5 ? -50 : canvas.height + 50;
                }
                
                // 瞄準畫面中央區域飛行，速度較快
                const targetX = canvas.width / 2 + (Math.random() - 0.5) * 300;
                const targetY = canvas.height / 2 + (Math.random() - 0.5) * 300;
                const angle = Math.atan2(targetY - this.y, targetX - this.x);
                const speed = Math.random() * 4 + 6; // 速度 6~10
                
                this.vx = Math.cos(angle) * speed;
                this.vy = Math.sin(angle) * speed;
                this.color = '#fff'; // 彗星主體白色或極淺黃
                this.tailColor = colors[Math.floor(Math.random() * colors.length)];
                this.size = Math.random() * 5 + 5;
                this.angle = angle; // 飛行方向就是三角形的指向
                this.active = true;
            }

            update() {
                this.x += this.vx;
                this.y += this.vy;
                
                // 飛出畫面太遠就標記為不活躍，等待清除
                if (this.x < -100 || this.x > canvas.width + 100 || 
                    this.y < -100 || this.y > canvas.height + 100) {
                    this.active = false;
                }

                // 邊飛邊產生尾巴粒子
                for(let i=0; i<2; i++){
                    particles.push(new CometTrail(this.x, this.y, this.tailColor));
                }
            }

            draw() {
                ctx.save();
                ctx.translate(this.x, this.y);
                ctx.rotate(this.angle);
                ctx.fillStyle = this.color;
                
                // 畫實心三角形 (指向右邊，也就是飛行方向)
                ctx.beginPath();
                ctx.moveTo(this.size * 2, 0);
                ctx.lineTo(-this.size, this.size);
                ctx.lineTo(-this.size, -this.size);
                ctx.closePath();
                ctx.fill();
                ctx.restore();
            }
        }

        // 彗星尾巴的點狀粒子
        class CometTrail {
            constructor(x, y, color) {
                this.x = x + (Math.random() - 0.5) * 10;
                this.y = y + (Math.random() - 0.5) * 10;
                this.color = color;
                this.size = Math.random() * 3 + 1;
                this.life = 1.0;
                this.decay = Math.random() * 0.03 + 0.01;
            }
            update() {
                this.life -= this.decay;
            }
            draw() {
                ctx.save();
                ctx.globalAlpha = Math.max(0, this.life);
                ctx.fillStyle = this.color;
                ctx.beginPath();
                ctx.arc(this.x, this.y, this.size, 0, Math.PI * 2);
                ctx.fill();
                ctx.restore();
            }
        }

        // 儲存陣列
        const shapes = [];
        const particles = [];
        const comets = [];

        // 初始散落物件 (增加數量讓碰撞更容易發生)
        for(let i = 0; i < 25; i++) {
            shapes.push(new FloatingShape());
        }

        // 點擊互動保留 (產生新物件與點狀粒子)
        window.addEventListener('mousedown', function(event) {
            if (event.button === 0) {
                const burstColor = colors[Math.floor(Math.random() * colors.length)];
                for(let i = 0; i < 15; i++) {
                    // 為了跟線條區分，這裡偷懶用 CometTrail 當作點狀粒子
                    let p = new CometTrail(event.clientX, event.clientY, burstColor);
                    p.x = event.clientX; p.y = event.clientY; // 重置位置
                    p.vx = (Math.random() - 0.5) * 10; p.vy = (Math.random() - 0.5) * 10;
                    p.update = function() { this.x+=this.vx; this.y+=this.vy; this.life-=this.decay; }
                    particles.push(p);
                }
                shapes.push(new FloatingShape(event.clientX, event.clientY));
            }
        });

        // --- 處理圓形之間的碰撞 ---
        function handleCollisions() {
            // 只篩選出圓形 (type === 0)
            const circles = shapes.filter(s => s.type === 0);
            
            for (let i = 0; i < circles.length; i++) {
                for (let j = i + 1; j < circles.length; j++) {
                    const c1 = circles[i];
                    const c2 = circles[j];
                    
                    // 確保兩者都不在冷卻時間內
                    if(c1.collisionCooldown > 0 || c2.collisionCooldown > 0) continue;

                    const dist = getDistance(c1.x, c1.y, c2.x, c2.y);
                    // 如果距離小於兩者半徑之和，代表撞到了！
                    if (dist < (c1.size / 2 + c2.size / 2)) {
                        
                        // 1. 物理反應：交換速度 (簡單的反彈效果)
                        const tempVx = c1.vx;
                        const tempVy = c1.vy;
                        c1.vx = c2.vx;
                        c1.vy = c2.vy;
                        c2.vx = tempVx;
                        c2.vy = tempVy;

                        // 2. 視覺效果：在碰撞中心點產生線條狀光芒
                        const midX = (c1.x + c2.x) / 2;
                        const midY = (c1.y + c2.y) / 2;
                        const burstColor = Math.random() < 0.5 ? c1.color : c2.color;
                        
                        for(let k = 0; k < 8; k++) {
                            particles.push(new LineParticle(midX, midY, burstColor));
                        }

                        // 3. 設定冷卻時間 (大約 30 幀內不再重複觸發碰撞)
                        c1.collisionCooldown = 30;
                        c2.collisionCooldown = 30;
                    }
                }
            }
        }

        // 動畫主迴圈
        function animate() {
            ctx.clearRect(0, 0, canvas.width, canvas.height);

            // 1. 隨機產生彗星 (大約每 100 幀有機會產生一顆)
            if(Math.random() < 0.01) {
                comets.push(new Comet());
            }

            // 2. 執行碰撞偵測
            handleCollisions();

            // 3. 更新與繪製物件
            shapes.forEach(shape => {
                shape.update();
                shape.draw();
            });

            // 4. 更新與繪製彗星
            for (let i = comets.length - 1; i >= 0; i--) {
                const c = comets[i];
                c.update();
                if (!c.active) {
                    comets.splice(i, 1);
                } else {
                    c.draw();
                }
            }

            // 5. 更新與繪製所有粒子 (線條碰撞、點擊爆發、彗星尾巴)
            for (let i = particles.length - 1; i >= 0; i--) {
                const p = particles[i];
                p.update();
                if (p.life <= 0) {
                    particles.splice(i, 1);
                } else {
                    p.draw();
                }
            }

            requestAnimationFrame(animate);
        }

        animate();
    </script>

🛠️ 效能優化重點整理：

限制最高內部解析度 (Resolution Capping)：網頁依然會 100% 填滿你的視窗，但我們將畫布「內部」的繪圖像素限制在一個基準值內。這樣高解析度螢幕（如 Retina）就不會被迫一次運算幾百萬個像素。

移除 shadowBlur 改用「偽光暈」：把超吃效能的原生陰影/發光關掉，改用「畫一條粗的半透明線 + 一條細的實體線」來模擬發光，效能大幅提升。

物件數量上限 (Object Pooling)：設定了「浮游物件最多 40 個」的上限。現在你狂點滑鼠時，最舊的物件會自然消散，避免畫面塞爆。

粒子數量與網格線優化：稍微調降碰撞噴發的粒子數，並限制「三角形網狀連結」的判定距離，減少不必要的線條渲染 (Draw Calls)。

            // 設定一個效能基準解析度面積 (約 1280x720)
            const TARGET_RES = 1280 * 720; 
            const currentRes = window.innerWidth * window.innerHeight;
            
            // 如果視窗太大，就等比例縮小內部渲染像素

        const colors = ['#3a86ff', '#f4a261', '#e63946', '#80ed99'];

            return Math.sqrt(Math.pow(x2 - x1, 2) + Math.pow(y2 - y1, 2));

        // 巨型引力方形
        class GiantSquare {
            constructor() {
                this.x = Math.random() * canvas.width;
                this.y = Math.random() * canvas.height;
                this.size = Math.random() * 40 + 60; 
                this.color = colors[Math.floor(Math.random() * colors.length)];

                this.vAngle = (Math.random() - 0.5) * 0.005; 
                this.vx = (Math.random() - 0.5) * 0.5;
                this.vy = (Math.random() - 0.5) * 0.5;

            update() {
                this.x += this.vx; this.y += this.vy; this.angle += this.vAngle;
                if (this.x > canvas.width + 150) this.x = -150;
                if (this.x < -150) this.x = canvas.width + 150;
                if (this.y > canvas.height + 150) this.y = -150;
                if (this.y < -150) this.y = canvas.height + 150;
            }
            draw() {
                ctx.save();
                ctx.translate(this.x, this.y);
                ctx.rotate(this.angle);
                ctx.strokeStyle = this.color;
                ctx.lineWidth = 3; 
                ctx.globalAlpha = 0.6;

                ctx.fillStyle = this.color;

                ctx.fillRect(-2, -2, 4, 4);
                ctx.restore();

        class FloatingShape {
            constructor(x, y) {
                this.x = x || Math.random() * canvas.width;
                this.y = y || Math.random() * canvas.height;
                this.vx = (Math.random() - 0.5) * 2;
                this.vy = (Math.random() - 0.5) * 2;
                this.angle = Math.random() * Math.PI * 2;
                this.vAngle = (Math.random() - 0.5) * 0.02;
                this.color = colors[Math.floor(Math.random() * colors.length)];
                this.size = Math.random() * 20 + 15; 
                this.type = Math.floor(Math.random() * 4); 
                this.collisionCooldown = 0; 
            }
            update() {
                if (this.type === 2 && giantSquares.length > 0) {
                    let nearestSquare = null;
                    let minDist = Infinity;
                    giantSquares.forEach(gs => {
                        const dist = getDistance(this.x, this.y, gs.x, gs.y);
                        if(dist < minDist) { minDist = dist; nearestSquare = gs; }
                    });
                    if (nearestSquare && minDist > 20) {
                        const dx = nearestSquare.x - this.x;
                        const dy = nearestSquare.y - this.y;
                        const gravity = 0.03;
                        this.vx += (dx / minDist) * gravity;
                        this.vy += (dy / minDist) * gravity;
                        const tangentForce = 0.05;
                        this.vx += (-dy / minDist) * tangentForce;
                        this.vy += (dx / minDist) * tangentForce;
                        const speed = Math.sqrt(this.vx * this.vx + this.vy * this.vy);
                        const maxSpeed = 3.5;
                        if(speed > maxSpeed) {
                            this.vx = (this.vx / speed) * maxSpeed;
                            this.vy = (this.vy / speed) * maxSpeed;
                        }
                    }
                }
                this.x += this.vx; this.y += this.vy; this.angle += this.vAngle;
                if (this.collisionCooldown > 0) this.collisionCooldown--;
                if (this.x > canvas.width + 50) this.x = -50;
                if (this.x < -50) this.x = canvas.width + 50;
                if (this.y > canvas.height + 50) this.y = -50;
                if (this.y < -50) this.y = canvas.height + 50;
            }
            draw() {
                ctx.save();
                ctx.translate(this.x, this.y);
                ctx.rotate(this.angle);
                ctx.fillStyle = this.color;
                ctx.strokeStyle = this.color;
                ctx.lineWidth = 2;
                ctx.beginPath();
                if (this.type === 0) { 
                    // 優化 2：用透明度疊加取代 shadowBlur 來做發光
                    if(this.collisionCooldown > 0) {
                        ctx.globalAlpha = 0.4;

                        ctx.fill();
                        ctx.globalAlpha = 1.0;
                    }
                    ctx.beginPath();
                    ctx.arc(0, 0, this.size / 2, 0, Math.PI * 2);
                    ctx.fill();
                } else if (this.type === 1) { 
                    ctx.moveTo(0, -this.size); ctx.lineTo(this.size, this.size); ctx.lineTo(-this.size, this.size); ctx.closePath(); ctx.stroke();
                } else if (this.type === 2) { 
                    ctx.fillRect(-this.size/2, -this.size/2, this.size, this.size * 1.5);
                } else if (this.type === 3) { 
                    ctx.arc(0, 0, 5, 0, Math.PI * 2); ctx.fill(); ctx.moveTo(0, 0); ctx.lineTo(-this.size * 3, 0); ctx.stroke();
                }
                ctx.restore();
            }
        }

        class SuperLineParticle {
            constructor(x, y, color) {
                this.x = x; this.y = y;
                const angle = Math.random() * Math.PI * 2;
                const speed = Math.random() * 8 + 4; 
                this.vx = Math.cos(angle) * speed; this.vy = Math.sin(angle) * speed;
                this.color = color;
                this.life = 1.0;
                this.decay = Math.random() * 0.02 + 0.01;

            update() { this.x += this.vx; this.y += this.vy; this.life -= this.decay; }
            draw() {
                ctx.save();
                ctx.strokeStyle = this.color;
                ctx.beginPath();
                ctx.moveTo(this.x, this.y);
                ctx.lineTo(this.x - this.vx * 3, this.y - this.vy * 3); 
                
                // 優化 2：偽光暈效果
                ctx.globalAlpha = Math.max(0, this.life * 0.3);
                ctx.lineWidth = 6;
                ctx.stroke();
                
                ctx.globalAlpha = Math.max(0, this.life);
                ctx.lineWidth = 2;
                ctx.stroke();
                
                ctx.restore();
            }
        }

        class Comet {
            constructor() {
                if(Math.random() < 0.5) {
                    this.x = Math.random() < 0.5 ? -50 : canvas.width + 50;
                    this.y = Math.random() * canvas.height;
                } else {
                    this.x = Math.random() * canvas.width;
                    this.y = Math.random() < 0.5 ? -50 : canvas.height + 50;
                }
                const targetX = canvas.width / 2 + (Math.random() - 0.5) * 300;
                const targetY = canvas.height / 2 + (Math.random() - 0.5) * 300;
                const angle = Math.atan2(targetY - this.y, targetX - this.x);
                const speed = Math.random() * 4 + 6;
                this.vx = Math.cos(angle) * speed; this.vy = Math.sin(angle) * speed;
                this.color = '#fff'; 
                this.tailColor = colors[Math.floor(Math.random() * colors.length)];
                this.size = Math.random() * 5 + 5;
                this.angle = angle;
                this.active = true;
            }
            update() {
                this.x += this.vx; this.y += this.vy;
                if (this.x < -100 || this.x > canvas.width + 100 || this.y < -100 || this.y > canvas.height + 100) this.active = false;
                particles.push(new CometTrail(this.x, this.y, this.tailColor)); // 減少尾巴粒子產生量
            }
            draw() {

            }
        }

        }

        function handleCollisions() {
            const circles = shapes.filter(s => s.type === 0);

                for (let j = i + 1; j < circles.length; j++) {

            }
        }

            }

            requestAnimationFrame(animate);
        }

        animate();

                this.color = colors[Math.floor(Math.random() * colors.length)];
                this.size = Math.random() * 20 + 15; 
                this.type = Math.floor(Math.random() * 4); 
                this.collisionCooldown = 0; 
            }
            update() {
                if (this.type === 2 && giantSquares.length > 0) {
                    let nearestSquare = null;
                    let minDist = Infinity;
                    giantSquares.forEach(gs => {
                        const dist = getDistance(this.x, this.y, gs.x, gs.y);
                        if(dist < minDist) { minDist = dist; nearestSquare = gs; }
                    });
                    if (nearestSquare && minDist > 20) {
                        const dx = nearestSquare.x - this.x;
                        const dy = nearestSquare.y - this.y;
                        const gravity = 0.03;
                        this.vx += (dx / minDist) * gravity;
                        this.vy += (dy / minDist) * gravity;
                        const tangentForce = 0.05;
                        this.vx += (-dy / minDist) * tangentForce;
                        this.vy += (dx / minDist) * tangentForce;
                        const speed = Math.sqrt(this.vx * this.vx + this.vy * this.vy);
                        const maxSpeed = 3.5;
                        if(speed > maxSpeed) {
                            this.vx = (this.vx / speed) * maxSpeed;
                            this.vy = (this.vy / speed) * maxSpeed;
                        }
                    }
                }
                this.x += this.vx; this.y += this.vy; this.angle += this.vAngle;
                if (this.collisionCooldown > 0) this.collisionCooldown--;
                if (this.x > canvas.width + 50) this.x = -50;
                if (this.x < -50) this.x = canvas.width + 50;
                if (this.y > canvas.height + 50) this.y = -50;
                if (this.y < -50) this.y = canvas.height + 50;
            }
            draw() {
                ctx.save();
                ctx.translate(this.x, this.y);
                ctx.rotate(this.angle);
                ctx.fillStyle = this.color;
                ctx.strokeStyle = this.color;
                ctx.lineWidth = 2;
                ctx.beginPath();
                if (this.type === 0) { 
                    // 平衡點 2：把真實光暈 shadowBlur 加回來，但數值調降至 15
                    if(this.collisionCooldown > 0) {
                        ctx.shadowBlur = 15;
                        ctx.shadowColor = this.color;
                    }
                    ctx.arc(0, 0, this.size / 2, 0, Math.PI * 2);
                    ctx.fill();
                    ctx.shadowBlur = 0;
                } else if (this.type === 1) { 
                    ctx.moveTo(0, -this.size); ctx.lineTo(this.size, this.size); ctx.lineTo(-this.size, this.size); ctx.closePath(); ctx.stroke();
                } else if (this.type === 2) { 
                    ctx.fillRect(-this.size/2, -this.size/2, this.size, this.size * 1.5);
                } else if (this.type === 3) { 
                    ctx.arc(0, 0, 5, 0, Math.PI * 2); ctx.fill(); ctx.moveTo(0, 0); ctx.lineTo(-this.size * 3, 0); ctx.stroke();
                }
                ctx.restore();
            }
        }

        function drawTriangleWeb() {
            const triangles = shapes.filter(s => s.type === 1);
            ctx.lineWidth = 0.5;
            ctx.strokeStyle = 'rgba(255, 255, 255, 0.15)'; 
            ctx.beginPath();
            for (let i = 0; i < triangles.length; i++) {
                for (let j = i + 1; j < triangles.length; j++) {
                    // 平衡點 5：連線距離放寬到 350
                    const dist = getDistance(triangles[i].x, triangles[i].y, triangles[j].x, triangles[j].y);
                    if (dist < 350) { 
                        ctx.moveTo(triangles[i].x, triangles[i].y);
                        ctx.lineTo(triangles[j].x, triangles[j].y);
                    }
                }
            }
            ctx.stroke();
        }

        function handleCollisions() {
            const circles = shapes.filter(s => s.type === 0);
            for (let i = 0; i < circles.length; i++) {
                for (let j = i + 1; j < circles.length; j++) {
                    const c1 = circles[i]; const c2 = circles[j];
                    if(c1.collisionCooldown > 0 || c2.collisionCooldown > 0) continue;

                    const dist = getDistance(c1.x, c1.y, c2.x, c2.y);
                    if (dist < (c1.size / 2 + c2.size / 2)) {
                        const tempVx = c1.vx; const tempVy = c1.vy;
                        c1.vx = c2.vx; c1.vy = c2.vy;
                        c2.vx = tempVx; c2.vy = tempVy;

                        const midX = (c1.x + c2.x) / 2;
                        const midY = (c1.y + c2.y) / 2;
                        const burstColor = Math.random() < 0.5 ? c1.color : c2.color;
                        
                        // 平衡點 6：碰撞粒子數量取中間值 (25長線 + 8圓點)
                        for(let k = 0; k < 25; k++) particles.push(new SuperLineParticle(midX, midY, burstColor));
                        for(let k = 0; k < 8; k++) particles.push(new CometTrail(midX, midY, '#fff'));

                        c1.collisionCooldown = 40; 
                        c2.collisionCooldown = 40;
                    }
                }
            }
        }

        function animate() {
            ctx.clearRect(0, 0, canvas.width, canvas.height);

            drawTriangleWeb();

            giantSquares.forEach(gs => { gs.update(); gs.draw(); });

            if(Math.random() < 0.01) comets.push(new Comet());

            handleCollisions();

            shapes.forEach(shape => { shape.update(); shape.draw(); });

            for (let i = comets.length - 1; i >= 0; i--) {
                const c = comets[i]; c.update();
                if (!c.active) comets.splice(i, 1); else c.draw();
            }

            for (let i = particles.length - 1; i >= 0; i--) {
                const p = particles[i]; p.update();
                if (p.life <= 0) particles.splice(i, 1); else p.draw();
            }

            requestAnimationFrame(animate);
        }

        animate();

設定五聲音階（Pentatonic Scale）：這樣不管怎麼隨機觸發，聽起來都會和諧、空靈，不會變成噪音。

高頻震動視覺：當物件發聲時，運用 Math.sin() 搭配時間，讓它產生自然且高頻的微幅縮放（像是喇叭震動膜一樣）。

                const targetX = canvas.width / 2 + (Math.random() - 0.5) * 300; const targetY = canvas.height / 2 + (Math.random() - 0.5) * 300;
                const angle = Math.atan2(targetY - this.y, targetX - this.x); const speed = Math.random() * 4 + 6;
                this.vx = Math.cos(angle) * speed; this.vy = Math.sin(angle) * speed; this.color = '#fff'; 
                this.tailColor = colors[Math.floor(Math.random() * colors.length)]; this.size = Math.random() * 5 + 5; this.angle = angle; this.active = true;
            }
            update() {
                this.x += this.vx; this.y += this.vy;
                if (this.x < -100 || this.x > canvas.width + 100 || this.y < -100 || this.y > canvas.height + 100) this.active = false;
                for(let i=0; i<2; i++) particles.push(new CometTrail(this.x, this.y, this.tailColor));
            }
            draw() {
                ctx.save(); ctx.translate(this.x, this.y); ctx.rotate(this.angle); ctx.fillStyle = this.color;
                ctx.beginPath(); ctx.moveTo(this.size * 2, 0); ctx.lineTo(-this.size, this.size); ctx.lineTo(-this.size, -this.size); ctx.closePath(); ctx.fill(); ctx.restore();
            }
        }
        class CometTrail {
            constructor(x, y, color) {
                this.x = x + (Math.random() - 0.5) * 8; this.y = y + (Math.random() - 0.5) * 8;
                this.color = color; this.size = Math.random() * 3 + 1; this.life = 1.0; this.decay = Math.random() * 0.04 + 0.02;
            }
            update() { this.life -= this.decay; }
            draw() { ctx.save(); ctx.globalAlpha = Math.max(0, this.life); ctx.fillStyle = this.color; ctx.beginPath(); ctx.arc(this.x, this.y, this.size, 0, Math.PI * 2); ctx.fill(); ctx.restore(); }
        }

        const shapes = []; const particles = []; const comets = []; const giantSquares = [];

        for(let i = 0; i < 35; i++) shapes.push(new FloatingShape());
        for(let i = 0; i < Math.floor(Math.random() * 2) + 2; i++) giantSquares.push(new GiantSquare());

        window.addEventListener('mousedown', function(event) {
            if (event.button === 0) {
                initAudio(); // 確保點擊時啟動音訊

                const mx = event.clientX * scaleRatio; const my = event.clientY * scaleRatio;
                const burstColor = colors[Math.floor(Math.random() * colors.length)];
                
                // 點擊音效：高頻清脆的鐘聲 (Sine wave)
                playSynth(880, 'sine', 1.0, 0.3);

                for(let i = 0; i < 15; i++) particles.push(new SuperLineParticle(mx, my, burstColor));
                
                let newShape = new FloatingShape(mx, my);
                newShape.playMidi(); // 誕生時發出自己的聲音
                shapes.push(newShape);
                
                if(shapes.length > 65) shapes.shift(); 
            }
        });

        // 🎵 隨機挑選物件當作 MIDI 演奏
        function playRandomMidi() {
            if(shapes.length === 0) return;
            // 隨機選 1~2 個物件發聲
            const numToPlay = Math.random() > 0.5 ? 2 : 1;
            for(let i=0; i<numToPlay; i++) {
                const randomShape = shapes[Math.floor(Math.random() * shapes.length)];
                randomShape.playMidi();
            }
        }

        // 🎵 三角形網格共鳴發聲
        function playWebChords() {
            const triangles = shapes.filter(s => s.type === 1);
            let playedCount = 0;
            for (let i = 0; i < triangles.length; i++) {
                for (let j = i + 1; j < triangles.length; j++) {
                    const dist = getDistance(triangles[i].x, triangles[i].y, triangles[j].x, triangles[j].y);
                    if (dist < 350) { 
                        // 如果有連線，讓其中一個三角形發出較長、柔和的弦樂聲 (Triangle wave)
                        if(playedCount < 3) { // 限制最多發出三個和弦音，避免太吵
                            playSynth(triangles[i].freq * 0.5, 'triangle', 2.0, 0.08);
                            triangles[i].vibrateFrames = 30; // 震動久一點
                            playedCount++;
                        }
                    }
                }
            }
        }

        function drawTriangleWeb() {
            const triangles = shapes.filter(s => s.type === 1);
            ctx.lineWidth = 0.5; ctx.strokeStyle = 'rgba(255, 255, 255, 0.15)'; ctx.beginPath();
            for (let i = 0; i < triangles.length; i++) {
                for (let j = i + 1; j < triangles.length; j++) {
                    if (getDistance(triangles[i].x, triangles[i].y, triangles[j].x, triangles[j].y) < 350) { 
                        ctx.moveTo(triangles[i].x, triangles[i].y); ctx.lineTo(triangles[j].x, triangles[j].y);
                    }
                }
            }
            ctx.stroke();
        }

        function handleCollisions() {
            const circles = shapes.filter(s => s.type === 0);
            for (let i = 0; i < circles.length; i++) {
                for (let j = i + 1; j < circles.length; j++) {
                    const c1 = circles[i]; const c2 = circles[j];
                    if(c1.collisionCooldown > 0 || c2.collisionCooldown > 0) continue;

                    if (getDistance(c1.x, c1.y, c2.x, c2.y) < (c1.baseSize / 2 + c2.baseSize / 2)) {
                        const tempVx = c1.vx; const tempVy = c1.vy;
                        c1.vx = c2.vx; c1.vy = c2.vy; c2.vx = tempVx; c2.vy = tempVy;

                        const midX = (c1.x + c2.x) / 2; const midY = (c1.y + c2.y) / 2;
                        const burstColor = Math.random() < 0.5 ? c1.color : c2.color;
                        
                        // 🎵 撞擊音效：雙頻共鳴聲
                        playSynth(c1.freq, 'square', 0.2, 0.15);
                        playSynth(c2.freq * 1.5, 'sine', 0.4, 0.2);
                        
                        // 撞擊時雙方都強烈震動
                        c1.vibrateFrames = 20; c2.vibrateFrames = 20;

                        for(let k = 0; k < 25; k++) particles.push(new SuperLineParticle(midX, midY, burstColor));
                        for(let k = 0; k < 8; k++) particles.push(new CometTrail(midX, midY, '#fff'));

                        c1.collisionCooldown = 40; c2.collisionCooldown = 40;
                    }
                }
            }
        }

        function animate() {
            ctx.clearRect(0, 0, canvas.width, canvas.height);
            drawTriangleWeb();
            giantSquares.forEach(gs => { gs.update(); gs.draw(); });
            if(Math.random() < 0.01) comets.push(new Comet());
            handleCollisions();
            shapes.forEach(shape => { shape.update(); shape.draw(); });
            for (let i = comets.length - 1; i >= 0; i--) {
                const c = comets[i]; c.update(); if (!c.active) comets.splice(i, 1); else c.draw();
            }
            for (let i = particles.length - 1; i >= 0; i--) {
                const p = particles[i]; p.update(); if (p.life <= 0) particles.splice(i, 1); else p.draw();
            }
            requestAnimationFrame(animate);
        }

        animate();
    </script>
