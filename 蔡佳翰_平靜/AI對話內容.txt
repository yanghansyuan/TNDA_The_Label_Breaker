<script>
(() => {
  const canvas = document.getElementById("c");
  const ctx = canvas.getContext("2d", { alpha: false });

  // DPR 支援（避免高解析螢幕糊掉）
  function resize() {
    const dpr = Math.max(1, Math.min(2, window.devicePixelRatio || 1));
    canvas.width  = Math.floor(window.innerWidth * dpr);
    canvas.height = Math.floor(window.innerHeight * dpr);
    ctx.setTransform(dpr, 0, 0, dpr, 0, 0); // 之後都用 CSS 像素畫
  }
  window.addEventListener("resize", resize, { passive: true });
  resize();

  // 漣漪資料結構
  const ripples = [];
  function addRipple(x, y) {
    ripples.push({
      x, y,
      t: 0,
      life: 2.8,     // 秒
      amp: 1.0,      // 振幅
      speed: 220     // 像素/秒
    });
  }

  // 左鍵互動（只吃左鍵）
  canvas.addEventListener("mousedown", (e) => {
    if (e.button !== 0) return;
    const rect = canvas.getBoundingClientRect();
    const x = e.clientX - rect.left;
    const y = e.clientY - rect.top;
    addRipple(x, y);
  });

  // 小小的「幾乎不動」的海面：用多個低頻正弦疊加
  function calmWave(x, t) {
    const w1 = Math.sin(x * 0.010 + t * 0.55) * 1.2;
    const w2 = Math.sin(x * 0.004 - t * 0.35) * 0.9;
    const w3 = Math.sin(x * 0.017 + t * 0.22) * 0.6;
    return (w1 + w2 + w3) * 0.55; // 整體很輕微
  }

  // 漣漪對某點的影響（距離環狀擴散）
  function rippleDisplacement(x, y, now) {
    let dy = 0;
    for (const r of ripples) {
      const age = r.t;
      if (age <= 0) continue;

      const dx = x - r.x;
      const dy0 = y - r.y;
      const dist = Math.hypot(dx, dy0);

      const radius = age * r.speed;
      const band = 34; // 漣漪環的厚度（越大越柔）
      const d = Math.abs(dist - radius);

      if (d < band) {
        const fade = 1 - (age / r.life);
        const ring = (1 - d / band);
        // 震盪：越外越淡、越久越淡
        const osc = Math.sin((dist - radius) * 0.22) * 10;
        dy += osc * ring * fade * r.amp;
      }
    }
    return dy;
  }

  // 背景：深海到地平線的漸層 + 很淡的霧光
  function drawBackground(w, h) {
    const g = ctx.createLinearGradient(0, 0, 0, h);
    g.addColorStop(0.00, "#050712");
    g.addColorStop(0.45, "#070a18");
    g.addColorStop(0.70, "#050915");
    g.addColorStop(1.00, "#03040b");
    ctx.fillStyle = g;
    ctx.fillRect(0, 0, w, h);

    // 柔光（像遠方海面反光）
    const hg = ctx.createRadialGradient(w*0.55, h*0.45, 0, w*0.55, h*0.45, Math.max(w,h)*0.55);
    hg.addColorStop(0, "rgba(140,190,255,0.08)");
    hg.addColorStop(0.35, "rgba(90,150,220,0.04)");
    hg.addColorStop(1, "rgba(0,0,0,0)");
    ctx.fillStyle = hg;
    ctx.fillRect(0, 0, w, h);
  }

  // 水面：用水平線 + 下方微微紋理（很克制）
  function drawWaterSurface(w, h, t) {
    const horizonY = h * 0.52; // 地平線位置
    const lineAmpBase = 2.2;   // 基本起伏很小

    // 水面區域（地平線以下）稍微亮一點
    const sea = ctx.createLinearGradient(0, horizonY, 0, h);
    sea.addColorStop(0.00, "rgba(120,170,235,0.06)");
    sea.addColorStop(0.35, "rgba(80,130,210,0.06)");
    sea.addColorStop(1.00, "rgba(20,30,60,0.00)");
    ctx.fillStyle = sea;
    ctx.fillRect(0, horizonY, w, h - horizonY);

    // 地平線（主水平線）
    ctx.lineWidth = 1.2;
    ctx.strokeStyle = "rgba(210,230,255,0.55)";
    ctx.beginPath();

    const step = 3; // 越小越細緻
    for (let x = 0; x <= w; x += step) {
      const base = calmWave(x, t) * lineAmpBase;
      // 讓「點擊附近」也能影響地平線
      const ripple = rippleDisplacement(x, horizonY, t);
      const y = horizonY + base + ripple * 0.12;
      if (x === 0) ctx.moveTo(x, y);
      else ctx.lineTo(x, y);
    }
    ctx.stroke();

    // 很淡的「水面紋理」：幾條近水平的細線
    for (let i = 0; i < 10; i++) {
      const yy = horizonY + 22 + i * 18;
      const alpha = 0.06 - i * 0.004;
      ctx.strokeStyle = `rgba(190,220,255,${Math.max(0, alpha)})`;
      ctx.lineWidth = 1;

      ctx.beginPath();
      for (let x = 0; x <= w; x += 6) {
        const base = calmWave(x + i*40, t + i*0.12) * (1.5 + i*0.05);
        const ripple = rippleDisplacement(x, yy, t);
        const y = yy + base + ripple * (0.10 + i*0.01);
        if (x === 0) ctx.moveTo(x, y);
        else ctx.lineTo(x, y);
      }
      ctx.stroke();
    }

    // 點擊漣漪的「圈圈」視覺（很淡、很像空氣裡的波）
    for (const r of ripples) {
      const age = r.t;
      const fade = 1 - (age / r.life);
      const radius = age * r.speed;

      ctx.strokeStyle = `rgba(200,230,255,${0.16 * fade})`;
      ctx.lineWidth = 1;
      ctx.beginPath();
      ctx.arc(r.x, r.y, radius, 0, Math.PI * 2);
      ctx.stroke();

      ctx.strokeStyle = `rgba(160,210,255,${0.10 * fade})`;
      ctx.beginPath();
      ctx.arc(r.x, r.y, radius * 0.72, 0, Math.PI * 2);
      ctx.stroke();
    }
  }

  let last = performance.now();
  function tick(nowMs) {
    const now = nowMs;
    const dt = Math.min(0.033, (now - last) / 1000);
    last = now;

    // 更新漣漪
    for (let i = ripples.length - 1; i >= 0; i--) {
      const r = ripples[i];
      r.t += dt;
      if (r.t > r.life) ripples.splice(i, 1);
    }

    const w = window.innerWidth;
    const h = window.innerHeight;

    // 時間（秒）
    const t = nowMs / 1000;

    drawBackground(w, h);
    drawWaterSurface(w, h, t);

    requestAnimationFrame(tick);
  }
  requestAnimationFrame(tick);
})();
</script>

ChatGPT 說：

可以，而且用 Web Audio API 就能做到：

第一次左鍵點擊 → 解鎖音訊（瀏覽器規定）+ 產生漣漪同時「低頻 sub bass 砰一下」+「sine / synth 空靈一點」

滑鼠靠近漣漪中心 → 音量、濾波、顫動（detune）、立體聲（panning）會跟著變強

遠離 / 漣漪散去 → 聲音慢慢淡出

下面給你「直接可用」的版本：把你原本 HTML 裡的 <script> 整段替換成這個（其他 HTML/CSS 可不變）。

你可以把它當成：水面是畫面層，聲音是空間層，滑鼠是「靠近就浮現」的感應。

<script>
(() => {
  const canvas = document.getElementById("c");
  const ctx = canvas.getContext("2d", { alpha: false });

  // ======= Canvas resize (DPR aware) =======
  function resize() {
    const dpr = Math.max(1, Math.min(2, window.devicePixelRatio || 1));
    canvas.width  = Math.floor(window.innerWidth * dpr);
    canvas.height = Math.floor(window.innerHeight * dpr);
    ctx.setTransform(dpr, 0, 0, dpr, 0, 0);
  }
  window.addEventListener("resize", resize, { passive: true });
  resize();

  // ======= Mouse tracking =======
  const mouse = { x: window.innerWidth * 0.5, y: window.innerHeight * 0.5, down: false };
  canvas.addEventListener("mousemove", (e) => {
    const rect = canvas.getBoundingClientRect();
    mouse.x = e.clientX - rect.left;
    mouse.y = e.clientY - rect.top;
  }, { passive: true });

  // ======= Ripples =======
  const ripples = [];
  function addRipple(x, y) {
    const r = {
      x, y,
      t: 0,
      life: 2.8,     // seconds
      amp: 1.0,
      speed: 220
    };
    ripples.push(r);
    return r;
  }

  // ======= Audio Engine (Web Audio API) =======
  let audio = null;

  function clamp01(v){ return Math.max(0, Math.min(1, v)); }
  function lerp(a,b,t){ return a + (b-a)*t; }

  function createAudio() {
    const AC = window.AudioContext || window.webkitAudioContext;
    const ac = new AC();

    // Master
    const master = ac.createGain();
    master.gain.value = 0.85;
    master.connect(ac.destination);

    // A tiny "space" using a simple feedback delay (cheap pseudo-reverb)
    const spaceSend = ac.createGain();
    spaceSend.gain.value = 0.25;

    const delay = ac.createDelay(1.0);
    delay.delayTime.value = 0.18;

    const fb = ac.createGain();
    fb.gain.value = 0.45;

    const lp = ac.createBiquadFilter();
    lp.type = "lowpass";
    lp.frequency.value = 1800;
    lp.Q.value = 0.6;

    // send -> delay -> filter -> master, with feedback loop
    spaceSend.connect(delay);
    delay.connect(lp);
    lp.connect(master);
    delay.connect(fb);
    fb.connect(delay);

    // Sub bass bus
    const subGain = ac.createGain();
    subGain.gain.value = 0.0;
    const subLP = ac.createBiquadFilter();
    subLP.type = "lowpass";
    subLP.frequency.value = 140;
    subLP.Q.value = 0.8;
    subGain.connect(subLP);
    subLP.connect(master);
    subGain.connect(spaceSend);

    // Pad/synth bus (soft, airy)
    const padGain = ac.createGain();
    padGain.gain.value = 0.0;

    const padFilter = ac.createBiquadFilter();
    padFilter.type = "lowpass";
    padFilter.frequency.value = 900;
    padFilter.Q.value = 0.7;

    const padPan = ac.createStereoPanner();
    padPan.pan.value = 0;

    padGain.connect(padFilter);
    padFilter.connect(padPan);
    padPan.connect(master);
    padPan.connect(spaceSend);

    // Always-on gentle synth (very low volume, controlled by proximity)
    const oscA = ac.createOscillator();
    oscA.type = "sine";
    oscA.frequency.value = 110; // base

    const oscB = ac.createOscillator();
    oscB.type = "triangle";
    oscB.frequency.value = 220;

    // slight movement
    const lfo = ac.createOscillator();
    lfo.type = "sine";
    lfo.frequency.value = 0.12;

    const lfoGain = ac.createGain();
    lfoGain.gain.value = 8; // cents-ish effect via detune

    lfo.connect(lfoGain);
    lfoGain.connect(oscB.detune);

    oscA.connect(padGain);
    oscB.connect(padGain);

    oscA.start();
    oscB.start();
    lfo.start();

    function ensureRunning() {
      if (ac.state !== "running") ac.resume();
    }

    // One-shot "click ripple" sound: sub thump + sine blip
    function triggerRippleSound(x, y) {
      ensureRunning();
      const t0 = ac.currentTime;

      // stereo position based on x
      const panVal = (x / Math.max(1, window.innerWidth)) * 2 - 1;

      // --- Sub thump ---
      const subOsc = ac.createOscillator();
      subOsc.type = "sine";
      const subEnv = ac.createGain();
      subEnv.gain.value = 0;

      const pan = ac.createStereoPanner();
      pan.pan.value = panVal * 0.35;

      // pitch drop 느낌 (像低頻水波)
      subOsc.frequency.setValueAtTime(70, t0);
      subOsc.frequency.exponentialRampToValueAtTime(38, t0 + 0.18);

      subEnv.gain.setValueAtTime(0.0001, t0);
      subEnv.gain.exponentialRampToValueAtTime(0.35, t0 + 0.02);
      subEnv.gain.exponentialRampToValueAtTime(0.0001, t0 + 0.28);

      subOsc.connect(subEnv);
      subEnv.connect(pan);
      pan.connect(subGain);   // goes into sub bus

      subOsc.start(t0);
      subOsc.stop(t0 + 0.35);

      // --- Sine "spark" (gentle) ---
      const blip = ac.createOscillator();
      blip.type = "sine";
      const blipEnv = ac.createGain();
      blipEnv.gain.value = 0;

      // frequency based on y (higher when closer to horizon)
      const ny = 1 - (y / Math.max(1, window.innerHeight));
      const f = lerp(220, 640, clamp01(ny));
      blip.frequency.setValueAtTime(f, t0);
      blip.frequency.exponentialRampToValueAtTime(f * 1.12, t0 + 0.08);

      blipEnv.gain.setValueAtTime(0.0001, t0);
      blipEnv.gain.exponentialRampToValueAtTime(0.18, t0 + 0.015);
      blipEnv.gain.exponentialRampToValueAtTime(0.0001, t0 + 0.22);

      const blipPan = ac.createStereoPanner();
      blipPan.pan.value = panVal * 0.55;

      blip.connect(blipEnv);
      blipEnv.connect(blipPan);
      blipPan.connect(padGain);

      blip.start(t0);
      blip.stop(t0 + 0.25);
    }

    return {
      ac,
      master,
      padGain,
      padFilter,
      padPan,
      subGain,
      subLP,
      spaceSend,
      triggerRippleSound
    };
  }

  // ======= Input: left click interacts with both visual and audio =======
  canvas.addEventListener("mousedown", (e) => {
    if (e.button !== 0) return;
    mouse.down = true;

    const rect = canvas.getBoundingClientRect();
    const x = e.clientX - rect.left;
    const y = e.clientY - rect.top;

    if (!audio) audio = createAudio();
    audio.triggerRippleSound(x, y);

    addRipple(x, y);
  });

  window.addEventListener("mouseup", () => { mouse.down = false; }, { passive: true });

  // ======= Calm wave + ripple displacement =======
  function calmWave(x, t) {
    const w1 = Math.sin(x * 0.010 + t * 0.55) * 1.2;
    const w2 = Math.sin(x * 0.004 - t * 0.35) * 0.9;
    const w3 = Math.sin(x * 0.017 + t * 0.22) * 0.6;
    return (w1 + w2 + w3) * 0.55;
  }

  function rippleDisplacement(x, y) {
    let dy = 0;
    for (const r of ripples) {
      const age = r.t;
      if (age <= 0) continue;

      const dx = x - r.x;
      const dy0 = y - r.y;
      const dist = Math.hypot(dx, dy0);

      const radius = age * r.speed;
      const band = 34;
      const d = Math.abs(dist - radius);

      if (d < band) {
        const fade = 1 - (age / r.life);
        const ring = (1 - d / band);
        const osc = Math.sin((dist - radius) * 0.22) * 10;
        dy += osc * ring * fade * r.amp;
      }
    }
    return dy;
  }

  // ======= Background + water surface =======
  function drawBackground(w, h) {
    const g = ctx.createLinearGradient(0, 0, 0, h);
    g.addColorStop(0.00, "#050712");
    g.addColorStop(0.45, "#070a18");
    g.addColorStop(0.70, "#050915");
    g.addColorStop(1.00, "#03040b");
    ctx.fillStyle = g;
    ctx.fillRect(0, 0, w, h);

    const hg = ctx.createRadialGradient(w*0.55, h*0.45, 0, w*0.55, h*0.45, Math.max(w,h)*0.55);
    hg.addColorStop(0, "rgba(140,190,255,0.08)");
    hg.addColorStop(0.35, "rgba(90,150,220,0.04)");
    hg.addColorStop(1, "rgba(0,0,0,0)");
    ctx.fillStyle = hg;
    ctx.fillRect(0, 0, w, h);
  }

  function drawWaterSurface(w, h, t) {
    const horizonY = h * 0.52;
    const lineAmpBase = 2.2;

    const sea = ctx.createLinearGradient(0, horizonY, 0, h);
    sea.addColorStop(0.00, "rgba(120,170,235,0.06)");
    sea.addColorStop(0.35, "rgba(80,130,210,0.06)");
    sea.addColorStop(1.00, "rgba(20,30,60,0.00)");
    ctx.fillStyle = sea;
    ctx.fillRect(0, horizonY, w, h - horizonY);

    // Horizon line
    ctx.lineWidth = 1.2;
    ctx.strokeStyle = "rgba(210,230,255,0.55)";
    ctx.beginPath();
    const step = 3;
    for (let x = 0; x <= w; x += step) {
      const base = calmWave(x, t) * lineAmpBase;
      const ripple = rippleDisplacement(x, horizonY);
      const y = horizonY + base + ripple * 0.12;
      if (x === 0) ctx.moveTo(x, y);
      else ctx.lineTo(x, y);
    }
    ctx.stroke();

    // gentle texture lines
    for (let i = 0; i < 10; i++) {
      const yy = horizonY + 22 + i * 18;
      const alpha = 0.06 - i * 0.004;
      ctx.strokeStyle = `rgba(190,220,255,${Math.max(0, alpha)})`;
      ctx.lineWidth = 1;

      ctx.beginPath();
      for (let x = 0; x <= w; x += 6) {
        const base = calmWave(x + i*40, t + i*0.12) * (1.5 + i*0.05);
        const ripple = rippleDisplacement(x, yy);
        const y = yy + base + ripple * (0.10 + i*0.01);
        if (x === 0) ctx.moveTo(x, y);
        else ctx.lineTo(x, y);
      }
      ctx.stroke();
    }

    // ripple rings (visual)
    for (const r of ripples) {
      const age = r.t;
      const fade = 1 - (age / r.life);
      const radius = age * r.speed;

      ctx.strokeStyle = `rgba(200,230,255,${0.16 * fade})`;
      ctx.lineWidth = 1;
      ctx.beginPath();
      ctx.arc(r.x, r.y, radius, 0, Math.PI * 2);
      ctx.stroke();

      ctx.strokeStyle = `rgba(160,210,255,${0.10 * fade})`;
      ctx.beginPath();
      ctx.arc(r.x, r.y, radius * 0.72, 0, Math.PI * 2);
      ctx.stroke();
    }

    // A subtle "proximity glow" when mouse near a ripple
    if (ripples.length) {
      const closest = getClosestRipple(mouse.x, mouse.y);
      if (closest) {
        const { r, d } = closest;
        const strength = clamp01(1 - d / 220);
        ctx.fillStyle = `rgba(160,210,255,${0.08 * strength})`;
        ctx.beginPath();
        ctx.arc(r.x, r.y, 120 + 120 * strength, 0, Math.PI * 2);
        ctx.fill();
      }
    }
  }

  // ======= Proximity logic (mouse near ripple) =======
  function getClosestRipple(x, y) {
    let best = null;
    for (const r of ripples) {
      const dx = x - r.x;
      const dy = y - r.y;
      const d = Math.hypot(dx, dy);
      if (!best || d < best.d) best = { r, d };
    }
    return best;

      audio.subLP.frequency.setTargetAtTime(lerp(110, 180, prox), now, 0.12);

function createAudio() {
  const AC = window.AudioContext || window.webkitAudioContext;
  const ac = new AC();

  // Master
  const master = ac.createGain();
  master.gain.value = 0.85;
  master.connect(ac.destination);

  // A tiny "space" using a simple feedback delay (cheap pseudo-reverb)
  const spaceSend = ac.createGain();
  spaceSend.gain.value = 0.25;

  const delay = ac.createDelay(1.0);
  delay.delayTime.value = 0.18;

  const fb = ac.createGain();
  fb.gain.value = 0.45;

  const lp = ac.createBiquadFilter();
  lp.type = "lowpass";
  lp.frequency.value = 1800;
  lp.Q.value = 0.6;

  spaceSend.connect(delay);
  delay.connect(lp);
  lp.connect(master);
  delay.connect(fb);
  fb.connect(delay);

  // ===== Sub bass bus =====
  const subGain = ac.createGain();
  subGain.gain.value = 0.0;

  const subLP = ac.createBiquadFilter();
  subLP.type = "lowpass";
  subLP.frequency.value = 140;
  subLP.Q.value = 0.8;

  subGain.connect(subLP);
  subLP.connect(master);
  subGain.connect(spaceSend);

  // ===== Pad / synth bus =====
  const padGain = ac.createGain();
  padGain.gain.value = 0.0;

  const padFilter = ac.createBiquadFilter();
  padFilter.type = "lowpass";
  padFilter.frequency.value = 900;
  padFilter.Q.value = 0.7;

  const padPan = ac.createStereoPanner();
  padPan.pan.value = 0;

  padGain.connect(padFilter);
  padFilter.connect(padPan);
  padPan.connect(master);
  padPan.connect(spaceSend);

  // ===== Always-on gentle synth =====
  const oscA = ac.createOscillator();
  oscA.type = "sine";
  oscA.frequency.value = 110;

  const oscB = ac.createOscillator();
  oscB.type = "triangle";
  oscB.frequency.value = 220;

  // slight movement
  const lfo = ac.createOscillator();
  lfo.type = "sine";
  lfo.frequency.value = 0.12;

  const lfoGain = ac.createGain();
  lfoGain.gain.value = 8;

  lfo.connect(lfoGain);
  lfoGain.connect(oscB.detune);

  oscA.connect(padGain);
  oscB.connect(padGain);

  oscA.start();
  oscB.start();
  lfo.start();

  // ===== [NEW] SEA AIR / NOISE LAYER =====
  // White noise buffer (loop)
  const noiseBuf = ac.createBuffer(1, ac.sampleRate * 2, ac.sampleRate);
  const data = noiseBuf.getChannelData(0);
  for (let i = 0; i < data.length; i++) data[i] = Math.random() * 2 - 1;

  const noiseSrc = ac.createBufferSource();
  noiseSrc.buffer = noiseBuf;
  noiseSrc.loop = true;

  // Noise envelope (we will modulate by proximity)
  const noiseGain = ac.createGain();
  noiseGain.gain.value = 0.0;

  // Bandpass makes it feel like "air" instead of harsh hiss
  const noiseBP = ac.createBiquadFilter();
  noiseBP.type = "bandpass";
  noiseBP.frequency.value = 520; // will be modulated
  noiseBP.Q.value = 0.9;

  // A gentle highpass to remove rumble
  const noiseHP = ac.createBiquadFilter();
  noiseHP.type = "highpass";
  noiseHP.frequency.value = 160;
  noiseHP.Q.value = 0.7;

  // A slight lowpass to keep it soft
  const noiseLP = ac.createBiquadFilter();
  noiseLP.type = "lowpass";
  noiseLP.frequency.value = 2600;
  noiseLP.Q.value = 0.6;

  // Spatial position
  const noisePan = ac.createStereoPanner();
  noisePan.pan.value = 0;

  // Very slow movement in noise tone (subtle "breathing")
  const nLfo = ac.createOscillator();
  nLfo.type = "sine";
  nLfo.frequency.value = 0.08;

  const nLfoGain = ac.createGain();
  nLfoGain.gain.value = 140; // Hz range added to bandpass center

  nLfo.connect(nLfoGain);
  nLfoGain.connect(noiseBP.frequency);

  // chain: noise -> gain -> bp -> hp -> lp -> pan -> master (+ space)
  noiseSrc.connect(noiseGain);
  noiseGain.connect(noiseBP);
  noiseBP.connect(noiseHP);
  noiseHP.connect(noiseLP);
  noiseLP.connect(noisePan);
  noisePan.connect(master);
  noisePan.connect(spaceSend);

  noiseSrc.start();
  nLfo.start();

  function ensureRunning() {
    if (ac.state !== "running") ac.resume();
  }

  function triggerRippleSound(x, y) {
    ensureRunning();
    const t0 = ac.currentTime;

    const panVal = (x / Math.max(1, window.innerWidth)) * 2 - 1;

    // --- Sub thump ---
    const subOsc = ac.createOscillator();
    subOsc.type = "sine";
    const subEnv = ac.createGain();
    subEnv.gain.value = 0;

    const pan = ac.createStereoPanner();
    pan.pan.value = panVal * 0.35;

    subOsc.frequency.setValueAtTime(70, t0);
    subOsc.frequency.exponentialRampToValueAtTime(38, t0 + 0.18);

    subEnv.gain.setValueAtTime(0.0001, t0);
    subEnv.gain.exponentialRampToValueAtTime(0.35, t0 + 0.02);
    subEnv.gain.exponentialRampToValueAtTime(0.0001, t0 + 0.28);

    subOsc.connect(subEnv);
    subEnv.connect(pan);
    pan.connect(subGain);

    subOsc.start(t0);
    subOsc.stop(t0 + 0.35);

    // --- Sine "spark" ---
    const blip = ac.createOscillator();
    blip.type = "sine";
    const blipEnv = ac.createGain();
    blipEnv.gain.value = 0;

    const ny = 1 - (y / Math.max(1, window.innerHeight));
    const f = 220 + (640 - 220) * Math.max(0, Math.min(1, ny));
    blip.frequency.setValueAtTime(f, t0);
    blip.frequency.exponentialRampToValueAtTime(f * 1.12, t0 + 0.08);

    blipEnv.gain.setValueAtTime(0.0001, t0);
    blipEnv.gain.exponentialRampToValueAtTime(0.18, t0 + 0.015);
    blipEnv.gain.exponentialRampToValueAtTime(0.0001, t0 + 0.22);

    const blipPan = ac.createStereoPanner();
    blipPan.pan.value = panVal * 0.55;

    blip.connect(blipEnv);
    blipEnv.connect(blipPan);
    blipPan.connect(padGain);

    blip.start(t0);
    blip.stop(t0 + 0.25);

    // Optional: tiny noise "puff" on click (super subtle)
    // makes the water feel like it has air.
    noiseGain.gain.cancelScheduledValues(t0);
    noiseGain.gain.setValueAtTime(noiseGain.gain.value, t0);
    noiseGain.gain.setTargetAtTime(Math.min(0.06, noiseGain.gain.value + 0.02), t0, 0.02);
    noiseGain.gain.setTargetAtTime(0.0, t0 + 0.12, 0.12);
  }

  return {
    ac,
    master,
    padGain,
    padFilter,
    padPan,
    subGain,
    subLP,
    spaceSend,
    triggerRippleSound,

    // [NEW] expose noise controls for proximity modulation
    noiseGain,
    noiseBP,
    noiseLP,
    noisePan
  };
}
