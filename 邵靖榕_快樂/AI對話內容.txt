const canvas = document.getElementById("canvas");
const ctx = canvas.getContext("2d");

function resize() {
  canvas.width = window.innerWidth;
  canvas.height = window.innerHeight;
}
window.addEventListener("resize", resize);
resize();

// 快樂粒子
class HappyNode {
  constructor(x, y, r = 20) {
    this.x = x;
    this.y = y;
    this.r = r;
    this.vx = (Math.random() - 0.5) * 1.2;
    this.vy = (Math.random() - 0.5) * 1.2;
    this.smile = Math.random() > 0.3;
  }

  draw() {
    ctx.lineWidth = 2;
    ctx.strokeStyle = "#222";
    ctx.fillStyle = "#fff";

    // 身體
    ctx.beginPath();
    ctx.arc(this.x, this.y, this.r, 0, Math.PI * 2);
    ctx.fill();
    ctx.stroke();

    // 表情
    ctx.beginPath();
    ctx.arc(this.x - 6, this.y - 5, 2, 0, Math.PI * 2);
    ctx.arc(this.x + 6, this.y - 5, 2, 0, Math.PI * 2);
    ctx.stroke();

    ctx.beginPath();
    if (this.smile) {
      ctx.arc(this.x, this.y + 2, 8, 0, Math.PI);
    } else {
      ctx.arc(this.x, this.y + 8, 8, Math.PI, Math.PI * 2);
    }
    ctx.stroke();

    // 小手小腳
    for (let a = 0; a < Math.PI * 2; a += Math.PI / 2) {
      ctx.beginPath();
      ctx.moveTo(
        this.x + Math.cos(a) * this.r,
        this.y + Math.sin(a) * this.r
      );
      ctx.lineTo(
        this.x + Math.cos(a) * (this.r + 10),
        this.y + Math.sin(a) * (this.r + 10)
      );
      ctx.stroke();
    }
  }

  update() {
    this.x += this.vx;
    this.y += this.vy;

    if (this.x < this.r || this.x > canvas.width - this.r) this.vx *= -1;
    if (this.y < this.r || this.y > canvas.height - this.r) this.vy *= -1;
  }
}

let nodes = [];

// 初始生成
for (let i = 0; i < 6; i++) {
  nodes.push(
    new HappyNode(
      Math.random() * canvas.width,
      Math.random() * canvas.height
    )
  );
}

// 滑鼠互動：左鍵點擊生成快樂
canvas.addEventListener("mousedown", (e) => {
  if (e.button === 0) {
    nodes.push(new HappyNode(e.clientX, e.clientY, 18));
  }
});

// 連線（情緒關係）
function drawConnections() {
  for (let i = 0; i < nodes.length; i++) {
    for (let j = i + 1; j < nodes.length; j++) {
      const dx = nodes[i].x - nodes[j].x;
      const dy = nodes[i].y - nodes[j].y;
      const dist = Math.sqrt(dx * dx + dy * dy);

      if (dist < 120) {
        ctx.strokeStyle = "rgba(0,0,0,0.15)";
        ctx.beginPath();
        ctx.moveTo(nodes[i].x, nodes[i].y);
        ctx.lineTo(nodes[j].x, nodes[j].y);
        ctx.stroke();
      }
    }
  }
}

// 動畫循環
function animate() {
  ctx.clearRect(0, 0, canvas.width, canvas.height);

  // 地面與斜線（意象轉譯）
  ctx.strokeStyle = "#ccc";
  ctx.beginPath();
  ctx.moveTo(50, canvas.height - 100);
  ctx.lineTo(canvas.width - 50, canvas.height - 100);
  ctx.stroke();

  ctx.beginPath();
  ctx.moveTo(canvas.width * 0.2, canvas.height);
  ctx.lineTo(canvas.width * 0.6, 0);
  ctx.stroke();

  drawConnections();

  nodes.forEach(n => {
    n.update();
    n.draw();
  });

  requestAnimationFrame(animate);
}

animate();

window.addEventListener("resize", resize);
resize();

/* ===== 陽光光束 ===== */
function drawSunlight() {
  const grad = ctx.createLinearGradient(0, 0, canvas.width, canvas.height);
  grad.addColorStop(0, "rgba(255, 240, 200, 0.25)");
  grad.addColorStop(0.5, "rgba(255, 240, 200, 0.05)");
  grad.addColorStop(1, "rgba(255, 240, 200, 0)");

  ctx.fillStyle = grad;
  ctx.beginPath();
  ctx.moveTo(0, 0);
  ctx.lineTo(canvas.width * 0.7, 0);
  ctx.lineTo(canvas.width * 0.3, canvas.height);
  ctx.lineTo(0, canvas.height);
  ctx.fill();
}

/* ===== 快樂粒子 ===== */
class Happy {
  constructor(x, y) {
    this.x = x;
    this.y = y;
    this.r = 20 + Math.random() * 5;
    this.vx = (Math.random() - 0.5) * 0.6;
    this.vy = (Math.random() - 0.5) * 0.6;
    this.glow = 12;
  }

  update() {
    this.x += this.vx;
    this.y += this.vy;

    if (this.x < this.r || this.x > canvas.width - this.r) this.vx *= -1;
    if (this.y < this.r || this.y > canvas.height - this.r) this.vy *= -1;
  }

  draw() {
    /* 光暈 */
    ctx.save();
    ctx.shadowColor = "rgba(255, 220, 150, 0.6)";
    ctx.shadowBlur = this.glow;

    ctx.fillStyle = "#fff";
    ctx.strokeStyle = "#333";
    ctx.lineWidth = 2;

    ctx.beginPath();
    ctx.arc(this.x, this.y, this.r, 0, Math.PI * 2);
    ctx.fill();
    ctx.stroke();

    ctx.restore();

    /* 眼睛（彎彎的喜悅） */
    ctx.lineWidth = 2;
    ctx.beginPath();
    ctx.arc(this.x - 6, this.y - 4, 4, Math.PI, Math.PI * 2);
    ctx.arc(this.x + 6, this.y - 4, 4, Math.PI, Math.PI * 2);
    ctx.stroke();

    /* 微笑 */
    ctx.beginPath();
    ctx.arc(this.x, this.y + 2, 8, 0, Math.PI);
    ctx.stroke();

    /* 小肢體 */
    for (let a = 0; a < Math.PI * 2; a += Math.PI / 2) {
      ctx.beginPath();
      ctx.moveTo(
        this.x + Math.cos(a) * this.r,
        this.y + Math.sin(a) * this.r
      );
      ctx.lineTo(
        this.x + Math.cos(a) * (this.r + 10),
        this.y + Math.sin(a) * (this.r + 10)
      );
      ctx.stroke();
    }
  }
}

const particles = [];
for (let i = 0; i < 6; i++) {
  particles.push(
    new Happy(
      Math.random() * canvas.width,
      Math.random() * canvas.height
    )
  );
}

/* 點擊生成更多快樂 */
canvas.addEventListener("mousedown", e => {
  particles.push(new Happy(e.clientX, e.clientY));
});

/* 粒子間柔和連結 */
function connect() {
  for (let i = 0; i < particles.length; i++) {
    for (let j = i + 1; j < particles.length; j++) {
      const dx = particles[i].x - particles[j].x;
      const dy = particles[i].y - particles[j].y;
      const d = Math.hypot(dx, dy);
      if (d < 120) {
        ctx.strokeStyle = "rgba(255, 200, 120, 0.15)";
        ctx.beginPath();
        ctx.moveTo(particles[i].x, particles[i].y);
        ctx.lineTo(particles[j].x, particles[j].y);
        ctx.stroke();
      }
    }
  }
}

/* 主動畫 */
function animate() {
  ctx.clearRect(0, 0, canvas.width, canvas.height);

  drawSunlight();
  connect();

  particles.forEach(p => {
    p.update();
    p.draw();
  });

  requestAnimationFrame(animate);
}
animate();

function drawLightBeam() {
  const g = ctx.createLinearGradient(
    canvas.width * 0.75, 0,
    canvas.width * 0.2, canvas.height
  );
  g.addColorStop(0, "rgba(255, 245, 220, 0.35)");
  g.addColorStop(0.4, "rgba(255, 245, 220, 0.15)");
  g.addColorStop(1, "rgba(255, 245, 220, 0)");

  ctx.fillStyle = g;
  ctx.beginPath();
  ctx.moveTo(canvas.width * 0.55, 0);
  ctx.lineTo(canvas.width, 0);
  ctx.lineTo(canvas.width * 0.65, canvas.height);
  ctx.lineTo(canvas.width * 0.25, canvas.height);
  ctx.closePath();
  ctx.fill();

/* ===== 懸浮粒子 ===== */
class Dust {
  constructor(x, y) {
    this.x = x;
    this.y = y;
    this.r = Math.random() * 2 + 0.5;
    this.vx = (Math.random() - 0.5) * 0.15;
    this.vy = (Math.random() - 0.5) * 0.15;
    this.alpha = Math.random() * 0.5 + 0.2;
  }

  update() {
    this.x += this.vx;
    this.y += this.vy;

    if (this.x < 0) this.x = canvas.width;
    if (this.x > canvas.width) this.x = 0;
    if (this.y < 0) this.y = canvas.height;
    if (this.y > canvas.height) this.y = 0;
  }

  draw() {
    ctx.fillStyle = `rgba(255, 255, 240, ${this.alpha})`;
    ctx.beginPath();
    ctx.arc(this.x, this.y, this.r, 0, Math.PI * 2);
    ctx.fill();
  }
}

const dusts = [];
for (let i = 0; i < 180; i++) {
  dusts.push(

}

/* 點擊：加入一點存在 */
