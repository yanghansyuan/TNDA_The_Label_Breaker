我想要做一個互動式的網頁，網頁大小要遵循RWD設計，能夠自動調整視窗排版並適應不同的裝置，互動方式必須包含滑鼠左鍵點擊功能，去觸發不同的效果




＃目的

我要做一個跟「情緒」有關的互動式網頁，HTML格式，這個情緒主題是「悲傷」，我想藉由滑鼠左鍵點擊的方式去觸發不同的情緒動畫，整體動畫過場需要流暢




＃風格

我想藉由不是完美的、數學公式下的圓，這些圓應該是帶有「厚度」與「重量」的幾何體。它們的邊緣不全然平滑，甚至帶有些微的破碎感

悲傷色彩的顏色必須去飽和化，並必須符合主色：Pantone 433 C、輔色：Pantone 1817 C、背景色： Pantone Cool Gray 3 C







＃動態語彙

讓圓形緩慢地、帶著重力感從頂部墜落。它們不是輕盈地飄下，而是像鉛球一樣沉沉地落在底部。




當使用者點擊滑鼠左鍵時，圓形會發生微小的、不規律的顫動，並伴隨視覺上的「震盪波」，模擬金屬碰撞時產生的那種令人不適的音頻與共振。




隨著時間推移，這些圓形在畫面底部越堆越高，逐漸佔領視窗空間，讓使用者感到視覺上的窒息與無法逃脫。

這是一個非常具有文學性與空間感的設計構思。在建築與視覺傳達的交界處，我們不只是在寫程式，而是在建構一個關於「重量」與「心理擠壓」的敘事空間。

為了實現這種帶有「厚度」與「破碎感」的幾何體，以及模擬金屬共振的視覺效果，我建議使用 Matter.js 物理引擎配合 HTML5 Canvas。Matter.js 能完美處理物體的重量感（Gravity & Density）與堆疊，而 Canvas 則能讓我們繪製出非完美的、邊緣破碎的圓。

以下是為您構思的程式碼架構與設計說明：

互動網頁實作：沉重的餘震 (Heavy Aftershocks)

此程式碼整合了物理引擎與自定義繪圖，確保圓形具有「鉛球」般的重力，並在點擊時產生模擬金屬不適感的震盪波。

<script>
    const { Engine, Render, Runner, Bodies, Composite, Mouse, Vector, Body } = Matter;

    // 設定顏色
    const colors = {
        background: '#C8C9C7', // Pantone Cool Gray 3 C
        main: '#1D2327',       // Pantone 433 C
        accent: '#591014'      // Pantone 1817 C
    };

    // 初始化物理引擎
    const engine = Engine.create();
    engine.world.gravity.y = 1.5; // 增加重力感

    const canvas = document.createElement('canvas');
    document.body.appendChild(canvas);
    const ctx = canvas.getContext('2d');

    let width = window.innerWidth;
    let height = window.innerHeight;

    // RWD 調整
    function resize() {
        width = window.innerWidth;
        height = window.innerHeight;
        canvas.width = width;
        canvas.height = height;
        // 更新物理邊界
        if (ground) Composite.remove(engine.world, [ground, wallLeft, wallRight]);
        setupBoundaries();
    }

    let ground, wallLeft, wallRight;
    function setupBoundaries() {
        ground = Bodies.rectangle(width / 2, height + 50, width, 100, { isStatic: true });
        wallLeft = Bodies.rectangle(-50, height / 2, 100, height, { isStatic: true });
        wallRight = Bodies.rectangle(width + 50, height / 2, 100, height, { isStatic: true });
        Composite.add(engine.world, [ground, wallLeft, wallRight]);
    }

    resize();
    window.addEventListener('resize', resize);

    // 儲存所有圓形的自定義資料
    const heavyCircles = [];
    const shockwaves = [];

    // 建立破碎感的圓形
    function createBrokenCircle(x, y) {
        const radius = Math.random() * 30 + 20;
        const sides = 12; // 邊線數，用於模擬不完美圓形
        const body = Bodies.polygon(x, y, sides, radius, {
            density: 0.1,      // 增加密度使其像鉛球
            friction: 0.8,
            restitution: 0.1,  // 低彈性，增加沉重感
            render: { visible: false } 
        });

        // 加入一點旋轉速度增加墜落感
        Body.setAngularVelocity(body, (Math.random() - 0.5) * 0.2);
        
        const circleData = {
            body: body,
            radius: radius,
            offsets: Array.from({ length: sides }, () => Math.random() * 5), // 用於邊緣破碎感
            vibration: 0
        };

        heavyCircles.push(circleData);
        Composite.add(engine.world, body);
    }

    // 滑鼠點擊觸發
    window.addEventListener('mousedown', (e) => {
        // 1. 產生視覺震盪波
        shockwaves.push({ x: e.clientX, y: e.clientY, r: 0, opacity: 1 });

        // 2. 觸發物體顫動與排斥力
        heavyCircles.forEach(c => {
            const dist = Vector.magnitude(Vector.sub(c.body.position, { x: e.clientX, y: e.clientY }));
            if (dist < 300) {
                c.vibration = 15; // 設定顫動強度
                // 模擬金屬碰撞的微小位移
                const force = Vector.mult(Vector.normalise(Vector.sub(c.body.position, { x: e.clientX, y: e.clientY })), 0.05);
                Body.applyForce(c.body, c.body.position, force);
            }
        });
    });

    // 定時自動生成（模擬情緒的累積）
    setInterval(() => {
        if (heavyCircles.length < 150) { // 限制數量避免效能下降
            createBrokenCircle(Math.random() * width, -100);
        }
    }, 1500);

    // 自定義繪製循環
    (function render() {
        ctx.fillStyle = colors.background;
        ctx.fillRect(0, 0, width, height);

        Engine.update(engine);

        // 繪製震盪波 (模擬不適音頻的視覺化)
        for (let i = shockwaves.length - 1; i >= 0; i--) {
            const s = shockwaves[i];
            s.r += 10;
            s.opacity -= 0.02;
            ctx.beginPath();
            ctx.arc(s.x, s.y, s.r, 0, Math.PI * 2);
            ctx.strokeStyle = `rgba(89, 16, 20, ${s.opacity})`; // Pantone 1817 C
            ctx.lineWidth = 2;
            ctx.stroke();
            if (s.opacity <= 0) shockwaves.splice(i, 1);
        }

        // 繪製「重型」圓形
        heavyCircles.forEach(c => {
            const { position, angle } = c.body;
            ctx.save();
            ctx.translate(position.x, position.y);
            
            // 處理顫動效果
            if (c.vibration > 0) {
                ctx.translate((Math.random() - 0.5) * c.vibration, (Math.random() - 0.5) * c.vibration);
                c.vibration *= 0.9; // 逐漸衰減
            }
            
            ctx.rotate(angle);
            
            // 繪製具厚度感的邊緣 (多層描邊)
            ctx.beginPath();
            for (let i = 0; i < 12; i++) {
                const ang = (i / 12) * Math.PI * 2;
                const r = c.radius - c.offsets[i];
                const x = Math.cos(ang) * r;
                const y = Math.sin(ang) * r;
                if (i === 0) ctx.moveTo(x, y);
                else ctx.lineTo(x, y);
            }
            ctx.closePath();
            
            ctx.fillStyle = colors.main; // Pantone 433 C
            ctx.fill();
            
            // 增加「重量感」的邊緣陰影
            ctx.strokeStyle = colors.accent; // Pantone 1817 C
            ctx.lineWidth = 1.5;
            ctx.globalAlpha = 0.6;
            ctx.stroke();
            
            ctx.restore();
        });

        requestAnimationFrame(render);
    })();
</script>

設計哲學與動態語彙分析
1. 物質性（Materiality）的表現

這不只是畫圓，而是利用物理參數（Density, Friction）賦予幾何體「靈魂」。在建築敘事中，空間的窒息感來自於比例的失衡。藉由 setInterval 不斷產生的圓形，模擬了悲傷情緒在時間維度上的堆疊，最終填滿視窗，這是一種從虛擬空間轉化為心理壓力的視覺過程。

2. 不完美幾何與破碎感

代碼中使用了 Bodies.polygon 結合隨機的 offsets 數組。這讓每個圓在渲染時，邊緣都會產生微小的凹凸。這呼應了您提到的「非數學公式下的圓」，它們更像是從某種結構上剝落下來的殘骸。

3. 共振與震盪（Resonance）

點擊時產生的 shockwaves 與 vibration 邏輯，是為了將聲音的「不適感」轉化為視覺。震盪波採用了深紅色的 Pantone 1817 C，在冷灰色的背景下顯得格外刺眼且沈重。

技術變革重點說明：

視覺 - 中空生鐵環 (Hollow Cast Iron Rings)：

我們不再使用 ctx.fill() 填滿圖形。

改為使用極粗的 ctx.lineWidth (描邊寬度) 配合 ctx.strokeStyle (描邊顏色) 來繪製。

為了模擬生鐵粗糙、不完美的邊緣，我保留了原本的噪點偏移 (offsets)，這讓環的內外圈看起來都是不平整的，非常有重量感。

色彩 - 雙色混雜 (Mixed Color Palette)：

在圓環生成時，引入了隨機機制。現在它們不再是單一顏色，而是會隨機分配主色 (Pantone 433 C) 或輔色 (Pantone 1817 C)，讓畫面堆疊後更有層次感。

互動 - 多重漣漪震盪波 (Multiple Ripples)：

點擊滑鼠時，不再只產生一個圓圈。程式碼現在會一次推入 5 個初始半徑不同的震盪波物件，使它們依序擴散，形成如同水面漣漪般的視覺效果。

互動 - 裂痕機制 (Cracking System)：

資料結構層：每個圓環物件中新增了一個 crackLevel (裂痕等級) 的屬性，初始為 0。

觸發層：當震盪波掃過圓環時，除了產生顫動，還會增加其 crackLevel（最高設為 3 等級）。

渲染層：在繪圖迴圈中，會檢查每個圓環的 crackLevel。如果有裂痕，就會在圓環表面額外繪製尖銳、不規則的白色線條，模擬金屬斷裂的視覺。

            body: body,
            radius: radius,
            // 用於製造邊緣不平整的隨機偏移量
            offsets: Array.from({ length: sides }, () => Math.random() * 4),
            color: ringColor,
            vibration: 0,    // 顫動強度
            crackLevel: 0    // 裂痕等級 (0-3)
        };

        heavyRings.push(ringData);
        Composite.add(engine.world, body);
    }

    // --- 4. 滑鼠互動：多重漣漪與裂痕觸發 ---
    window.addEventListener('mousedown', (e) => {
        const clickPos = { x: e.clientX, y: e.clientY };

        // 產生 5 圈多重漣漪，透過初始半徑(r)為負值來製造時間差
        for(let i = 0; i < 5; i++) {
             shockwaves.push({ 
                 x: clickPos.x, 
                 y: clickPos.y, 
                 r: -i * 40,       // 初始半徑為負，延遲出現
                 opacity: 0.8 - (i * 0.1), // 外圈透明度較低
                 width: 5 - i*0.5  // 線條寬度
             });
        }

        // 檢查並觸發圓環的反應
        heavyRings.forEach(c => {
            const dist = Vector.magnitude(Vector.sub(c.body.position, clickPos));
            // 震盪波影響範圍
            if (dist < 350) {
                c.vibration = 20; // 設定強烈顫動
                
                // 增加裂痕等級，最高到 3 級
                if (Math.random() > 0.3) { // 不是每次必裂，增加隨機感
                     c.crackLevel = Math.min(c.crackLevel + 1, 3);
                }

                // 施加微小的斥力，模擬衝擊
                const forceDir = Vector.normalise(Vector.sub(c.body.position, clickPos));
                const forceMag = 0.08 / (dist > 50 ? dist/50 : 1); // 距離越近力量越大
                Body.applyForce(c.body, c.body.position, Vector.mult(forceDir, forceMag));
            }
        });
    });

    // 定時生成圓環 (模擬情緒積累)
    setInterval(() => {
        // 限制總數量以維持效能，堆疊到一定程度停止
        if (heavyRings.length < 200) { 
            createIronRing(Math.random() * width, -100);
        }
    }, 1200); // 生成速度稍微加快

    // --- 5. 輔助繪圖函數：繪製裂痕 ---
    function drawCracks(ctx, radius, crackLevel) {
        if (crackLevel === 0) return;

        ctx.strokeStyle = 'rgba(255, 255, 255, 0.5)'; // 裂痕顏色：半透明白
        ctx.lineWidth = 1.5;
        ctx.beginPath();
        
        // 根據裂痕等級決定畫幾條線
        const numCracks = crackLevel * 2; 
        for (let i = 0; i < numCracks; i++) {
            // 隨機生成穿越圓心的折線
            const angle = Math.random() * Math.PI * 2;
            const startR = radius * (0.4 + Math.random() * 0.2);
            const endR = radius * (0.8 + Math.random() * 0.2);
            
            const x1 = Math.cos(angle) * startR;
            const y1 = Math.sin(angle) * startR;
            // 中間加一個折點，讓裂痕更尖銳
            const midX = (Math.random()-0.5) * radius * 0.5;
            const midY = (Math.random()-0.5) * radius * 0.5;
            const x2 = Math.cos(angle + Math.PI) * endR;
            const y2 = Math.sin(angle + Math.PI) * endR;

            ctx.moveTo(x1, y1);
            ctx.lineTo(midX, midY);
            ctx.lineTo(x2, y2);
        }
        ctx.stroke();
    }


    // --- 6. 主渲染循環 ---
    (function render() {
        // 清空背景
        ctx.fillStyle = colors.background;
        ctx.fillRect(0, 0, width, height);

        // 更新物理引擎
        Engine.update(engine);

        // A. 繪製多重震盪波漣漪
        for (let i = shockwaves.length - 1; i >= 0; i--) {
            const s = shockwaves[i];
            s.r += 8; // 擴散速度
            
            // 只有當半徑大於 0 時才繪製
            if (s.r > 0) {
                s.opacity -= 0.015; // 透明度衰減
                ctx.beginPath();
                ctx.arc(s.x, s.y, s.r, 0, Math.PI * 2);
                // 漣漪顏色使用輔色 (Pantone 1817 C)
                ctx.strokeStyle = `rgba(89, 16, 20, ${s.opacity})`; 
                ctx.lineWidth = s.width;
                ctx.stroke();
            }
            // 完全透明後移除
            if (s.opacity <= 0) shockwaves.splice(i, 1);
        }

        // B. 繪製生鐵圓環
        heavyRings.forEach(c => {
            const { position, angle } = c.body;
            const sides = c.offsets.length;

            ctx.save();
            ctx.translate(position.x, position.y);
            
            // 處理受衝擊時的顫動
            if (c.vibration > 0) {
                const shakeX = (Math.random() - 0.5) * c.vibration;
                const shakeY = (Math.random() - 0.5) * c.vibration;
                ctx.translate(shakeX, shakeY);
                c.vibration *= 0.9; // 顫動衰減
            }
            
            ctx.rotate(angle);
            
            // 開始繪製不規則路徑
            ctx.beginPath();
            for (let i = 0; i < sides; i++) {
                const ang = (i / sides) * Math.PI * 2;
                // 半徑加上隨機偏移，製造粗糙邊緣
                const r = c.radius + c.offsets[i];
                const x = Math.cos(ang) * r;
                const y = Math.sin(ang) * r;
                if (i === 0) ctx.moveTo(x, y);
                else ctx.lineTo(x, y);
            }
            ctx.closePath();
            
            // --- 關鍵修改：從填充改為粗描邊，形成空心環 ---
            ctx.strokeStyle = c.color;
            // 描邊寬度設為半徑的 40%，製造厚重感
            ctx.lineWidth = c.radius * 0.4; 
            // 線段端點和轉角設為圓角，讓生鐵感更自然
            ctx.lineCap = 'round';
            ctx.lineJoin = 'round';
            ctx.stroke();

            // 繪製額外的細線增加鑄造的粗糙紋理感
            ctx.lineWidth = 1;
            ctx.strokeStyle = 'rgba(0,0,0,0.3)';
            ctx.stroke();

            // C. 繪製裂痕 (如果有的話)
            drawCracks(ctx, c.radius, c.crackLevel);
            
            ctx.restore();
        });

        requestAnimationFrame(render);
    })();

崩裂的短暫性：
當使用者點擊，原本穩定的鐵塊邊緣突然變得劇烈跳動（crumbleNoise），模擬物質在極高壓下崩解的狀態。這呼應了情緒在受刺激時瞬間的崩潰。10 秒的設定讓這個過程顯得漫長且難以忽視，直到它緩慢「癒合」。

餘震的節奏：
透過 setTimeout 產生的多重漣漪，消除了數位動畫常見的機械感。這種隨機的時間間隔更接近自然界（如地震或水體受擊）的物理表現。

        wallLeft = Bodies.rectangle(-60, height / 2, 120, height * 3, { isStatic: true });
        wallRight = Bodies.rectangle(width + 60, height / 2, 120, height * 3, { isStatic: true });
        Composite.add(engine.world, [ground, wallLeft, wallRight]);
    }
    window.addEventListener('resize', resize);
    resize();

    function createIronStone(x, y) {
        const radius = Math.random() * 20 + 25;
        const body = Bodies.polygon(x, y, 15, radius, {
            density: 0.6, friction: 0.95, restitution: 0.01, frictionAir: 0.002,
            render: { visible: false }
        });

        ironStones.push({
            body, radius, 
            offsets: Array.from({ length: 15 }, () => Math.random() * 8),
            alpha: 0.4 + Math.random() * 0.6,
            mouthType: Math.random() > 0.5 ? 'line' : 'invertedV',
            crumbleTimer: 0,
            vibration: 0
        });

    }

                    speed: 5 + Math.random() * 15,
                    opacity: 0.5,
                    weight: Math.random() * 6 + 2

        ironStones.forEach(s => { if (s.body.position.y < highestY) highestY = s.body.position.y; });

            ctx.fillStyle = colors.bgExtreme;

        // 渲染憂鬱藍色漣漪

            sw.r += sw.speed;
            sw.opacity -= 0.012;
            if (sw.r > 0) {
                ctx.beginPath();
                ctx.arc(sw.x, sw.y, sw.r, 0, Math.PI * 2);
                ctx.strokeStyle = `rgba(15, 76, 129, ${sw.opacity})`; 
                ctx.lineWidth = sw.weight;
                ctx.stroke();
            }
            if (sw.opacity <= 0) shockwaves.splice(i, 1);
        });

        // 渲染生鐵石

            const { position, angle } = s.body;
            ctx.save();
            ctx.translate(position.x, position.y);

                ctx.translate((Math.random()-0.5)*s.vibration, (Math.random()-0.5)*s.vibration);
                s.vibration *= 0.8;

            for (let i = 0; i < 15; i++) {

                const crumble = s.crumbleTimer > 0 ? (Math.random() * 20 * (s.crumbleTimer/10000)) : 0;
                const r = s.radius + s.offsets[i] + crumble;
                const x = Math.cos(ang) * r;
                const y = Math.sin(ang) * r;
                if (i === 0) ctx.moveTo(x, y); else ctx.lineTo(x, y);
            }
            ctx.closePath();
            ctx.fillStyle = `rgba(29, 35, 39, ${s.alpha})`;
            ctx.fill();

            // 繪製穿透背景的開孔
            ctx.fillStyle = isExtremeState ? colors.bgExtreme : `rgb(${200+flicker},${205+flicker},${208+flicker})`;
            const eS = s.radius * 0.16;
            ctx.beginPath(); ctx.arc(-s.radius*0.38, -s.radius*0.28, eS, 0, 7); ctx.fill();
            ctx.beginPath(); ctx.arc(s.radius*0.38, -s.radius*0.28, eS, 0, 7); ctx.fill();

            if (s.mouthType === 'line') {
                ctx.fillRect(-s.radius*0.5, s.radius*0.25, s.radius, s.radius*0.18);
            } else {
                ctx.beginPath();
                ctx.moveTo(-s.radius*0.4, s.radius*0.45);
                ctx.lineTo(0, s.radius*0.05);
                ctx.lineTo(s.radius*0.4, s.radius*0.45);
                ctx.lineWidth = s.radius * 0.2;
                ctx.strokeStyle = ctx.fillStyle;
                ctx.stroke();
            }
            ctx.restore();
        });
        ctx.restore();
        requestAnimationFrame(render);
    })(0);

動態語彙設計說明

非線性時間感：
透過 setTimeout 結合隨機的 r（半徑）初值，漣漪不再以固定的圓心向外擴散，而是呈現出破碎、斷裂的動態，模擬情緒在壓抑後的非對稱爆發。

不和諧的環境光：
背景色的「呼吸」現在與震波數量聯動。當您連續點擊時，背景會因為震波過多而產生類似電力不穩的閃爍感。這種不規則性是為了打破電腦動畫的「平滑感」，製造一種令生理不適的視覺噪點。

極限狀態的空間縮減：
當鐵塊填滿螢幕，背景轉紅且產生 translate 偏移（螢幕抖動），這在建築現象學中象徵著空間的死亡。使用者原本的觀察者角色，在此刻轉化為被堆疊物威脅的受害者。

設計細節解析

背景的「自轉與公轉」 (Spatial Oscillations)：
我使用了疊加的正弦與餘弦波來處理底層背景。Math.sin(baseOrbit) + Math.cos(baseOrbit * 2.3) 創造了一個不規則的長短波，這代表空間內部的自發性能量。而 clickImpulse 則像是外來的公轉引力，當您點擊時，背景閃爍的速度與幅度會瞬間增加，隨後才慢慢平復。

藍色憂鬱 (Melancholic Blue)：
選用了經典藍色調的 Pantone 19-4052。這種顏色在冷灰背景中擴散時，不再具有攻擊性，而更像是一種沈入水底的寂靜。

崩潰的紅點 (The Breaking Point)：
極限狀態的判定現在非常嚴格（高度佔據 88% 以上）。一旦觸發，背景會瞬間轉為死寂的深紅（Pantone 1817 C），並配合全螢幕的亂數位移。這不僅是視覺上的變化，更是一種模擬「系統超載」的空間敘事。
