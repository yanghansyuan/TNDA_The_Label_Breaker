局部扭曲：透過滑鼠 movementX/Y 計算速度並暫時對 Canvas 套用 blur 濾鏡。

有機輪廓 (outlineCanvas)：這是一個由多個 Sine/Cosine 波組成的圓形偏移路徑。它在長按 7 秒後會逐漸浮現，且因為沒有重置語句，放開滑鼠後會保持該透明度。

裂痕 (crackCanvas)：每次點擊都會在畫布上畫出隨機分叉的線段。由於畫布不自動清空（除了輪廓層），所以裂痕是 永久存在 的。

加大有機輪廓的形狀複雜度

要加大有機輪廓的複雜度，我們需要從數學上增加**諧波（Harmonics）**的數量。目前的代碼只用了兩組波，我們可以引入更多的頻率疊加，並加入隨機的「噪聲感」，讓它看起來更像是不規則的生物細胞或液態金屬。

請將原程式碼中的 drawOutline 函式替換為以下版本。這個版本使用了 疊加正弦波 與 時間偏移 來創造更詭譎、深邃的運動感：

// --- 進階版：複雜有機輪廓 ---
let outlineTime = 0;

function drawOutline() {
    outlineTime += 0.008; // 控制蠕動速度
    outlineCtx.clearRect(0, 0, width, height);
    
    // 增加光暈感
    outlineCtx.shadowBlur = 15;
    outlineCtx.shadowColor = 'rgba(255,255,255,0.3)';
    outlineCtx.strokeStyle = 'rgba(255,255,255,0.6)';
    outlineCtx.lineWidth = 1.2;
    
    outlineCtx.beginPath();

    const centerX = width / 2;
    const centerY = height / 2;
    const baseRadius = Math.min(width, height) * 0.25; // 根據螢幕大小自適應半徑

    // 增加取樣點以提高平滑度 (原本 0.1 改為 0.02)
    for (let i = 0; i <= Math.PI * 2; i += 0.02) {
        
        // 多層波形疊加 (Harmonics)
        // Wave 1: 基礎慢速大波
        let noise1 = Math.sin(i * 3 + outlineTime) * 25;
        // Wave 2: 中速中波
        let noise2 = Math.cos(i * 7 - outlineTime * 1.5) * 15;
        // Wave 3: 快速細節小波 (增加破碎感)
        let noise3 = Math.sin(i * 13 + outlineTime * 2.2) * 8;
        // Wave 4: 極細節顫動
        let noise4 = Math.cos(i * 25 + Math.sin(outlineTime)) * 4;

        const r = baseRadius + noise1 + noise2 + noise3 + noise4;
        
        const x = centerX + Math.cos(i) * r;
        const y = centerY + Math.sin(i) * r;

        if (i === 0) outlineCtx.moveTo(x, y);
        else outlineCtx.lineTo(x, y);
    }
    
    outlineCtx.closePath();
    outlineCtx.stroke();
    
    // 額外繪製一個稍微偏移的內圈，增加層次感
    drawInnerOutline(centerX, centerY, baseRadius, outlineTime);
    
    requestAnimationFrame(drawOutline);
}

// 輔助函式：繪製稍微不同的內層輪廓
function drawInnerOutline(cx, cy, r, t) {
    outlineCtx.strokeStyle = 'rgba(255,255,255,0.2)';
    outlineCtx.beginPath();
    for (let i = 0; i <= Math.PI * 2; i += 0.05) {
        let noise = Math.sin(i * 5 - t) * 10 + Math.cos(i * 11 + t) * 5;
        const radius = r * 0.8 + noise;
        const x = cx + Math.cos(i) * radius;
        const y = cy + Math.sin(i) * radius;
        if (i === 0) outlineCtx.moveTo(x, y);
        else outlineCtx.lineTo(x, y);
    }
    outlineCtx.closePath();
    outlineCtx.stroke();
}

本次更新的改動點：

多頻疊加：使用了 4 組不同頻率（3, 7, 13, 25）的波。高頻率（如 25）會產生細微的尖刺與凹凸，低頻率（如 3）則負責整體的形狀扭動。

層次感：新增了一個 drawInnerOutline，在主輪廓內部繪製一個半透明、運動方向相反的次要輪廓，模擬「透明有機物」的內部深度。

視覺強化：加入了 shadowBlur，讓輪廓在長按浮現時帶有微弱的發光效果，而非死板的白線。

高精度取樣：將循環的步進值從 0.1 縮小到 0.02，確保在高複雜度波形下線條依然圓滑不破裂。
