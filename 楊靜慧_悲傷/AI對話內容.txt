1. 內容

一個方塊在純淨的白色上滑雪，類似俯視視角，背景有零星或一小片的三角形的綠樹，隨著滑鼠滾輪滾動方塊會持續如橫向卷軸由左向右滑雪，滑過的地方要留下滑過的痕跡，按左鍵可以在白雪地種三角形的樹。

如果持續滾滑鼠滾輪持續約30秒，期間背景零星的綠樹會越來越少，30秒後所有樹都會消失，滑雪方塊也會停下，然後扁掉變成長方塊。

在持續滑滑鼠滾輪，長方塊會往下方沉，並且畫面上方開始有黑藍色的墨水以滴入水中的呈現效果滲透下來，並且越來越多，直到整個畫面被墨水布滿。

這時候滑鼠左鍵的互動效果變成按下去可以出現淺藍圓點蓋過墨染過的地方。

在墨浸透長方塊周圍後，如果有用左鍵互動讓他周圍全部變呈淺藍色，那長方塊會重新變回正方塊如橫向卷軸由左向右滑雪，經過的地方全部變呈淺藍色，墨會停止滲透並且隨著正方塊的前進開始緩緩向下沉入消失，直到隨著墨沉入而顯現的淺藍色底色占滿整個畫面。

之後正方塊隨著滑鼠滾輪滾動方塊依然會持續如橫向卷軸由左向右滑雪，背景開始出現零星的三角形的綠樹。

設計理念與操作說明：

第一階段 - 雪地滑行 (Snow Skiing)：

畫面純白，中間有一個黑色方塊。

操作：滾動滑鼠滾輪，方塊會向右滑行（背景向左移動），並留下滑雪痕跡。

互動：點擊滑鼠左鍵，可以在雪地上種植綠色的幾何樹木。

進程：持續滾動約 30 秒（或達到一定滑行距離），樹木會開始凋零消失，進入下一階段。

第二階段 - 沉沒與墨染 (Sinking & Ink)：

方塊會被壓扁成長方形，停止滑行，並隨著滾輪向下沉。

上方會滴落深藍黑色的墨水，逐漸染黑整個畫面。

互動：此時點擊滑鼠左鍵，會產生「淨化」的淺藍色光點。

目標：在墨水完全覆蓋前或覆蓋後，在長方塊周圍點擊，用淺藍色光點包圍它。

第三階段 - 重生 (Resurrection)：

如果長方塊被淺藍色光點覆蓋，它會變回正方塊。

墨水散去，背景轉為淺藍色（如同天空或冰河）。

方塊繼續滑行，綠色樹木重新開始生長。

    // --- 聲音引擎 (Web Audio API - 進階版) ---
    class SoundManager {
        constructor() {
            this.ctx = null;
            this.masterGain = null;
            this.initialized = false;
            
            this.currentMood = 'none'; 
            this.timers = [];
            
            // 聲音節點參照
            this.nodes = {
                droneOsc1: null,
                droneOsc2: null,
                droneGain: null,
                droneFilter: null,
                noise: null,
                noiseGain: null,
                melodyGain: null
            };

            // 音階設定

            this.p3Notes = [
                261.63, // C4
                329.63, // E4
                392.00, // G4
                493.88, // B4
                523.25, // C5
                392.00, // G4
                329.63  // E4
            ];
            this.p3Index = 0;
        }

        init() {
            if (this.initialized) return;
            try {
                const AudioContext = window.AudioContext || window.webkitAudioContext;
                this.ctx = new AudioContext();
                this.masterGain = this.ctx.createGain();
                this.masterGain.gain.value = 0.5;
                
                // 加個壓縮器避免爆音
                const compressor = this.ctx.createDynamicsCompressor();
                compressor.threshold.value = -24;
                compressor.knee.value = 30;
                compressor.ratio.value = 12;
                compressor.attack.value = 0.003;
                compressor.release.value = 0.25;

                this.masterGain.connect(compressor);
                compressor.connect(this.ctx.destination);
                
                this.initialized = true;
            } catch (e) {
                console.warn("Web Audio API not supported");
            }
        }

        resume() {
            if (this.ctx && this.ctx.state === 'suspended') {
                this.ctx.resume();
            }
        }

        // 播放單音 (帶有簡單 Delay 效果)
        playTone(freq, type, duration, startTime, vol = 0.1) {
            if (!this.ctx) return;
            const osc = this.ctx.createOscillator();
            const gain = this.ctx.createGain();
            
            osc.type = type;
            osc.frequency.value = freq;
            
            // ADSR Envelope
            const now = startTime || this.ctx.currentTime;
            gain.gain.setValueAtTime(0, now);
            gain.gain.linearRampToValueAtTime(vol, now + 0.1); // Attack
            gain.gain.exponentialRampToValueAtTime(0.01, now + duration); // Release
            
            // Delay Effect
            const delay = this.ctx.createDelay();
            delay.delayTime.value = 0.3; // 300ms delay
            const delayGain = this.ctx.createGain();
            delayGain.gain.value = 0.3; // feedback volume

            osc.connect(gain);
            gain.connect(this.masterGain);
            
            // Connect to Delay
            gain.connect(delay);
            delay.connect(delayGain);
            delayGain.connect(delay); // feedback loop
            delay.connect(this.masterGain);

            osc.start(now);
            osc.stop(now + duration + 1.0); // 延長一點給 delay 尾音
        }

        // 第一階段：療癒、溫暖
        startPhase1() {
            if (this.currentMood === 'healing') return;
            this.currentMood = 'healing';
            
            const now = this.ctx.currentTime;

            // 建立持續的 Drone (雙層 Triangle，微調 Detune 增加溫暖感)
            this.nodes.droneOsc1 = this.ctx.createOscillator();
            this.nodes.droneOsc2 = this.ctx.createOscillator();
            this.nodes.droneGain = this.ctx.createGain();
            this.nodes.droneFilter = this.ctx.createBiquadFilter();

            this.nodes.droneOsc1.type = 'triangle';
            this.nodes.droneOsc1.frequency.value = 130.81; // C3
            
            this.nodes.droneOsc2.type = 'triangle';
            this.nodes.droneOsc2.frequency.value = 196.00; // G3 (五度音)
            this.nodes.droneOsc2.detune.value = 10; // 稍微偏音產生 Chorus 效果

            this.nodes.droneFilter.type = 'lowpass';
            this.nodes.droneFilter.frequency.value = 800; // 溫暖濾波

            // 連接
            this.nodes.droneOsc1.connect(this.nodes.droneFilter);
            this.nodes.droneOsc2.connect(this.nodes.droneFilter);
            this.nodes.droneFilter.connect(this.nodes.droneGain);
            this.nodes.droneGain.connect(this.masterGain);

            // 淡入
            this.nodes.droneGain.gain.setValueAtTime(0, now);
            this.nodes.droneGain.gain.linearRampToValueAtTime(0.15, now + 3);

            this.nodes.droneOsc1.start(now);
            this.nodes.droneOsc2.start(now);

            // 隨機旋律迴圈
            const loop = () => {
                if (this.currentMood !== 'healing') return;
                const delay = 800 + Math.random() * 2000;
                const note = this.p1Scale[Math.floor(Math.random() * this.p1Scale.length)];
                // 使用 Sine 波讓聲音圓潤
                this.playTone(note, 'sine', 2.5, this.ctx.currentTime, 0.08);
                this.timers.push(setTimeout(loop, delay));
            };
            loop();
        }

        // 轉場：滑入悲傷 (變扁階段)
        morphToSad() {
            if (this.currentMood === 'morphing') return;
            this.currentMood = 'morphing';
            
            // 清除旋律 timer，讓旋律自然停止生成，但不切斷正在響的
            this.timers.forEach(t => clearTimeout(t));
            this.timers = [];

            const now = this.ctx.currentTime;
            const duration = 2.0; // 變形時間

            if (this.nodes.droneOsc1 && this.nodes.droneOsc2) {
                // 音高滑落：從 C3/G3 滑落到低頻不和諧音程 (例如 A1 和 Bb1)
                // 這模擬「下沉」和「壓力」
                this.nodes.droneOsc1.frequency.exponentialRampToValueAtTime(55.0, now + duration); // A1
                this.nodes.droneOsc2.frequency.exponentialRampToValueAtTime(58.27, now + duration); // Bb1 (大二度衝突)

                // 濾波器打開，讓聲音變亮/變尖
                this.nodes.droneFilter.frequency.linearRampToValueAtTime(2000, now + duration);
                
                // 波形切換：雖然 Oscillator 不能直接換 type 的過渡，但我們可以利用 detune 和 high filter 模擬粗糙感
                // 或者我們可以淡入一個 Sawtooth
            }
        }

        // 第二階段：完全進入悲傷
        startPhase2() {
            if (this.currentMood === 'sad') return;
            this.currentMood = 'sad';
            
            // 確保之前的 Drone 已經變形完成，現在它是我們的主要噪音源
            // 我們可以加入一些低頻 Rumble

            // 如果還沒有 Drone (罕見情況)，建立一個

                // fallback
                this.nodes.droneOsc1 = this.ctx.createOscillator();
                this.nodes.droneGain = this.ctx.createGain();
                this.nodes.droneOsc1.frequency.value = 55.0;
                this.nodes.droneOsc1.type = 'sawtooth';
                this.nodes.droneOsc1.connect(this.nodes.droneGain);
                this.nodes.droneGain.connect(this.masterGain);
                this.nodes.droneOsc1.start();
                this.nodes.droneGain.gain.value = 0.15;
            } else {
                // 改變現有 Drone 的波形 (這會有點硬切，但在低頻下還可以接受，或者我們就保持 Triangle 但因為 Filter 開了所以很亮)
                // 為了更平滑，我們保持原樣，因為它已經是不和諧的 A1+Bb1
            }

            // 偶爾加入高頻不和諧音
            const sadLoop = () => {
                if (this.currentMood !== 'sad') return;
                if (Math.random() > 0.7) {
                    this.playTone(466.16, 'sawtooth', 4.0, this.ctx.currentTime, 0.02); // Bb4
                }
                this.timers.push(setTimeout(sadLoop, 4000));
            };
            sadLoop();
        }

        // 第三階段：沉靜循環 (重生)
        startPhase3() {
            if (this.currentMood === 'calm') return;
            this.currentMood = 'calm';
            
            const now = this.ctx.currentTime;

            // 1. 處理舊的悲傷 Drone：慢慢過濾掉高頻，變回溫暖的低頻，然後淡出或融合
            if (this.nodes.droneFilter) {
                // 關閉 Filter，把刺耳的高頻濾掉
                this.nodes.droneFilter.frequency.exponentialRampToValueAtTime(100, now + 4);
                // 音量慢慢變小
                this.nodes.droneGain.gain.linearRampToValueAtTime(0.05, now + 5);
                
                // 讓不和諧音程解決到和諧音程 (A1/Bb1 -> C2/C2)
                if (this.nodes.droneOsc1) this.nodes.droneOsc1.frequency.linearRampToValueAtTime(65.41, now + 5); // C2
                if (this.nodes.droneOsc2) this.nodes.droneOsc2.frequency.linearRampToValueAtTime(130.81, now + 5); // C3
            }

            // 2. 啟動新的循環旋律
            this.p3Index = 0;
            const melodyLoop = () => {
                if (this.currentMood !== 'calm') return;
                
                const freq = this.p3Notes[this.p3Index];
                // 使用 Sine 波，加上長 Release，像管鐘或電鋼琴
                this.playTone(freq, 'sine', 4.0, this.ctx.currentTime, 0.12);
                
                this.p3Index = (this.p3Index + 1) % this.p3Notes.length;
                this.timers.push(setTimeout(melodyLoop, 1500));
            };
            
            // 延遲一點點開始，讓轉場呼吸
            this.timers.push(setTimeout(melodyLoop, 1000));
        }
    }

    function checkRebirthCondition() {
        if (currentState !== STATE.SINKING && currentState !== STATE.DARKNESS) return;
        
        const px = player.x;
        const py = player.y;
        const pw = player.w;
        const ph = player.h;

        const checkPoints = [
            {x: px, y: py},             
            {x: px + pw, y: py},        
            {x: px, y: py + ph},        
            {x: px + pw, y: py + ph},   
            {x: px + pw/2, y: py + ph/2}

        const activePurifiers = renderLayers.filter(l => l.type === 'purifier' && l.opacity > 0.5);

        for (let pt of checkPoints) {
            let isPointCovered = false;
            for (let p of activePurifiers) {
                const dx = pt.x - p.x;
                const dy = pt.y - p.y;
                if (Math.sqrt(dx*dx + dy*dy) < p.r) {
                    isPointCovered = true;
                    break;
                }
            }
            if (isPointCovered) coveredPoints++;
        }

        if (coveredPoints === 5 && darknessLevel > 0.95) {
            triggerRebirth();
        }

    function drawRoundedCube(ctx, x, y, w, h, baseColor) {
        const r = 4; 
        const depth = player.baseSize * 0.4; 

        const isoX = depth * 0.866; 
        const isoY = depth * 0.5;   

        const f_tr = {x: x + w, y: y};
        const f_br = {x: x + w, y: y + h};
        const b_tr = {x: x + w + isoX, y: y - isoY};
        const b_br = {x: x + w + isoX, y: y + h - isoY};
        
        ctx.lineJoin = 'round';
        ctx.lineWidth = 1;

        ctx.fillStyle = player.sideColor;
        ctx.beginPath();
        ctx.moveTo(f_tr.x, f_tr.y + r);
        ctx.lineTo(f_tr.x, f_br.y - r);
        ctx.arcTo(f_tr.x, f_br.y, f_br.x + isoX, f_br.y - isoY, r); 
        ctx.lineTo(b_br.x, b_br.y + r);
        ctx.arcTo(b_br.x, b_br.y, b_tr.x, b_tr.y, r); 
        ctx.lineTo(b_tr.x, b_tr.y);
        ctx.lineTo(f_tr.x, f_tr.y);
        ctx.closePath();
        ctx.fill();

        ctx.fillStyle = player.topColor;
        ctx.beginPath();
        ctx.moveTo(x, y); 
        ctx.lineTo(x + isoX, y - isoY); 
        ctx.lineTo(b_tr.x, b_tr.y); 
        ctx.lineTo(f_tr.x, f_tr.y); 
        ctx.closePath();
        ctx.fill();

        ctx.fillStyle = baseColor;
        ctx.beginPath();
        ctx.roundRect(x, y, w, h, r);
        ctx.fill();

    function drawBrushInk(ctx, item) {
        ctx.save();
        ctx.translate(item.x, item.y);
        
        ctx.globalAlpha = item.opacity;
        ctx.fillStyle = COLORS.ink;
        
        ctx.beginPath();
        ctx.arc(0, 0, item.r * 0.8, 0, Math.PI * 2);
        ctx.fill();

        ctx.globalAlpha = item.opacity * 0.3;
        ctx.beginPath();
        const segments = 16;
        for (let i = 0; i <= segments; i++) {
            const angle = (i / segments) * Math.PI * 2;
            const noise = Math.sin(angle * 5 + item.seed) * (item.r * 0.1);
            const r = item.r + noise;
            ctx.lineTo(Math.cos(angle) * r, Math.sin(angle) * r);
        }
        ctx.closePath();
        ctx.fill();

        ctx.restore();
    }

    function drawSmoothTracks(ctx) {

            const sx = offsetX - (scrollDistance - p.worldX);

        ctx.stroke();
