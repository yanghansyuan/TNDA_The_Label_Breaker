        class AudioEngine {
            constructor() {
                this.ctx = null;
                this.gainNode = null;
                this.sourceNode = null;
                this.isPlaying = false;
                this.currentType = 'white';
                this.volume = 0.5;
            }

            init() {
                if (!this.ctx) {
                    this.ctx = new (window.AudioContext || window.webkitAudioContext)();
                    this.gainNode = this.ctx.createGain();
                    this.gainNode.connect(this.ctx.destination);
                    this.gainNode.gain.value = this.volume;
                }
            }

            // 生成噪音緩衝區
            createNoiseBuffer() {
                const bufferSize = 2 * this.ctx.sampleRate; // 2秒的緩衝
                const buffer = this.ctx.createBuffer(1, bufferSize, this.ctx.sampleRate);
                const output = buffer.getChannelData(0);

                for (let i = 0; i < bufferSize; i++) {
                    let white = Math.random() * 2 - 1;
                    
                    if (this.currentType === 'white') {
                        // 純白噪音
                        output[i] = white * 0.5; 
                    } else if (this.currentType === 'pink') {
                        // 簡易粉紅噪音演算法 (近似)
                        output[i] = (Math.random() * 2 - 1) * 0.5; 
                        // Note: 真實粉紅噪音需要複雜濾波，這裡用後製濾波器模擬
                    } else if (this.currentType === 'brown') {
                         // 簡易布朗噪音 (近似)
                         output[i] = (Math.random() * 2 - 1) * 0.5;
                    }
                }
                return buffer;
            }

            play() {
                this.init();
                if (this.ctx.state === 'suspended') {
                    this.ctx.resume();
                }

                if (this.isPlaying) this.stop();

                // 建立噪音源
                const buffer = this.createNoiseBuffer();
                this.sourceNode = this.ctx.createBufferSource();
                this.sourceNode.buffer = buffer;
                this.sourceNode.loop = true;

                // 根據類型添加濾波器以塑造音色
                let lastNode = this.sourceNode;

                if (this.currentType === 'pink') {
                    // 粉紅噪音濾波 (模擬雨聲/自然聲)
                    const b0 = this.ctx.createBiquadFilter();
                    b0.type = 'lowshelf';
                    b0.frequency.value = 400;
                    b0.gain.value = -5;
                    lastNode.connect(b0);
                    lastNode = b0;
                } else if (this.currentType === 'brown') {
                    // 布朗噪音濾波 (模擬低頻/深海)
                    const b0 = this.ctx.createBiquadFilter();
                    b0.type = 'lowpass';
                    b0.frequency.value = 150; 
                    lastNode.connect(b0);
                    lastNode = b0;
                } else {
                    // 白噪音稍微削減高頻以免刺耳
                    const b0 = this.ctx.createBiquadFilter();
                    b0.type = 'lowpass';
                    b0.frequency.value = 10000;
                    lastNode.connect(b0);
                    lastNode = b0;
                }

                // 淡入效果
                this.gainNode.gain.setValueAtTime(0, this.ctx.currentTime);
                this.gainNode.gain.linearRampToValueAtTime(this.volume, this.ctx.currentTime + 1);

                lastNode.connect(this.gainNode);
                this.sourceNode.start();
                this.isPlaying = true;
            }

            stop() {
                if (this.sourceNode) {
                    // 淡出效果
                    this.gainNode.gain.cancelScheduledValues(this.ctx.currentTime);
                    this.gainNode.gain.setValueAtTime(this.gainNode.gain.value, this.ctx.currentTime);
                    this.gainNode.gain.linearRampToValueAtTime(0, this.ctx.currentTime + 0.5);
                    
                    const oldSource = this.sourceNode;
                    setTimeout(() => {
                        try { oldSource.stop(); } catch(e){}
                    }, 500);
                    
                    this.sourceNode = null;
                }
                this.isPlaying = false;
            }

            setVolume(val) {
                this.volume = val;
                if (this.gainNode) {
                    // 平滑過渡音量
                    this.gainNode.gain.setTargetAtTime(val, this.ctx.currentTime, 0.1);
                }
            }

            setType(type) {
                if (this.currentType === type) return;
                this.currentType = type;
                if (this.isPlaying) {
                    this.play(); // 重新播放以切換音色
                }
            }
        }

### 設計重點解析：

1.  **3D 空間 (WebGL / Three.js)**：
    *   **場景**：我沒有使用複雜的貼圖，而是使用了 `IcosahedronGeometry` 配合 `MeshStandardMaterial`，賦予其粗糙度 (`roughness`) 和一點點金屬感，這能讓光影在表面上的擴散更柔和，符合「素色光影感」。
    *   **光影**：使用了兩個不同顏色的 `PointLight`（暖橙色與冷藍色）在不同軌道上運行。這在物體表面創造了不斷變化的色彩漸層與影子，模擬出一種緩慢的呼吸感。
    *   **互動**：滑鼠移動時，相機會有輕微的視差位移（Parallax），增加空間的立體感與沉浸感。

2.  **聲音引擎 (Web Audio API)**：
    *   **無外部素材**：我編寫了一個 `AudioEngine` 類別，直接利用數學公式在瀏覽器中生成聲音波形。這意味著網頁載入極快，且沒有版權或連結失效問題。
    *   **三種模式**：
        *   **Focus (白噪音)**：全頻段聲音，適合遮蔽背景雜音。
        *   **Rain (粉紅噪音模擬)**：透過低頻濾波器 (LowShelf)，聽起來像下雨或風吹過樹葉。
        *   **Deep (布朗噪音模擬)**：透過更強的低通濾波器 (LowPass)，聽起來像遠處的雷聲或深海，非常適合放鬆。

            constructor() {

                this.bufferSize = 0;

            // 初始化音訊環境 (必須在使用者點擊後呼叫)
            init() {
                const AudioContext = window.AudioContext || window.webkitAudioContext;
                this.ctx = new AudioContext();
                
                // 總音量控制
                this.gainNode = this.ctx.createGain();
                this.gainNode.gain.value = 0; // 初始靜音，待會淡入
                this.gainNode.connect(this.ctx.destination);

                // 分析器 (讓 3D 物體隨聲音跳動)
                this.analyser = this.ctx.createAnalyser();
                this.analyser.fftSize = 512; // 採樣大小
                this.analyser.smoothingTimeConstant = 0.8; // 平滑度
                
                this.bufferSize = 2 * this.ctx.sampleRate; // 2秒緩衝
            }

            // 生成噪音 Buffer
            createNoiseBuffer(type) {
                const buffer = this.ctx.createBuffer(1, this.bufferSize, this.ctx.sampleRate);
                const data = buffer.getChannelData(0);
                
                let lastOut = 0;
                
                for (let i = 0; i < this.bufferSize; i++) {
                    const white = Math.random() * 2 - 1;
                    
                    if (type === 'white') {
                        data[i] = white;
                    } 
                    else if (type === 'pink') {
                        // Pink Noise 近似演算法 (Paul Kellet's method)
                        let b0, b1, b2, b3, b4, b5, b6;
                        b0 = b1 = b2 = b3 = b4 = b5 = b6 = 0.0;
                        // 簡化版 1/f 濾波
                        // 為了代碼簡潔，這裡使用簡單的積分濾波來模擬 \"較柔和\" 的聲音
                        data[i] = (lastOut + (0.02 * white)) / 1.02;
                        lastOut = data[i];
                        data[i] *= 3.5; // 補償增益
                    } 
                    else if (type === 'brown') {
                        // Brown Noise (布朗噪音) - 隨機漫步
                        const brown = (lastOut + (0.02 * white)) / 1.02;
                        lastOut = brown;
                        data[i] = brown * 3.5; 
                    }
                }
                return buffer;
            }

            play(type = 'white') {
                if (!this.ctx) this.init();
                
                // 確保 Context 運行中
                if (this.ctx.state === 'suspended') {
                    this.ctx.resume();
                }

                // 停止舊聲音
                if (this.sourceNode) {
                    try { this.sourceNode.stop(); } catch(e){}
                }

                this.currentType = type;
                const buffer = this.createNoiseBuffer(type);
                
                this.sourceNode = this.ctx.createBufferSource();
                this.sourceNode.buffer = buffer;
                this.sourceNode.loop = true;

                // 串接：Source -> Analyser -> Gain -> Destination
                this.sourceNode.connect(this.analyser);
                this.analyser.connect(this.gainNode);
                
                this.sourceNode.start();
                this.isPlaying = true;

                // 淡入效果
                this.fadeIn();
            }

            fadeIn() {
                this.gainNode.gain.cancelScheduledValues(this.ctx.currentTime);
                this.gainNode.gain.setValueAtTime(0, this.ctx.currentTime);
                this.gainNode.gain.linearRampToValueAtTime(this.volume, this.ctx.currentTime + 2); // 2秒淡入
            }

            setVolume(val) {
                this.volume = val;
                if (this.gainNode) {
                    this.gainNode.gain.setTargetAtTime(val, this.ctx.currentTime, 0.1);
                }
            }

            // 取得音頻數據供 3D 使用 (0 ~ 255)
            getAudioData() {
                if (!this.analyser) return 0;
                const array = new Uint8Array(this.analyser.frequencyBinCount);
                this.analyser.getByteFrequencyData(array);
                // 取平均值作為整體音量強度
                let sum = 0;
                for(let i = 0; i < array.length; i++) {
                    sum += array[i];
                }
                return sum / array.length;
            }
        }

        /**
         * 聲音引擎 (Audio Engine)
         * 特點：即時生成、自動淡入、低通濾波讓聲音更圓潤
         */
        class SoundSystem {
            constructor() {
                this.ctx = null;
                this.mainGain = null;
                this.noiseSource = null;
                this.analyser = null;
                this.currentType = 'white';
                this.volume = 0.5;
                this.isPlaying = false;
            }

            init() {
                if (this.ctx) return;
                const AudioContext = window.AudioContext || window.webkitAudioContext;
                this.ctx = new AudioContext();

                // 主音量
                this.mainGain = this.ctx.createGain();
                this.mainGain.gain.value = 0; // 初始靜音
                this.mainGain.connect(this.ctx.destination);

                // 分析器 (用於視覺連動)
                this.analyser = this.ctx.createAnalyser();
                this.analyser.fftSize = 256; 
                this.analyser.smoothingTimeConstant = 0.9; // 讓數值變化非常平滑，不跳動
                this.analyser.connect(this.mainGain);
            }

            createBuffer(type) {
                const bufferSize = 2 * this.ctx.sampleRate;
                const buffer = this.ctx.createBuffer(1, bufferSize, this.ctx.sampleRate);
                const data = buffer.getChannelData(0);
                
                let lastOut = 0;
                for (let i = 0; i < bufferSize; i++) {
                    const white = Math.random() * 2 - 1;
                    
                    if (type === 'white') {
                        data[i] = white;
                    } else if (type === 'pink') {
                        // 簡易粉紅噪音濾波
                        const b0 = 0.99886 * lastOut + white * 0.0555179;
                        data[i] = b0 * 3.5; // 補償增益
                        lastOut = b0;
                    } else if (type === 'brown') {
                        // 布朗噪音 (積分)
                        const b0 = (lastOut + (0.02 * white)) / 1.02;
                        data[i] = b0 * 3.5;
                        lastOut = b0;
                    }
                }
                return buffer;
            }

            play(type) {
                this.init();
                if (this.ctx.state === 'suspended') this.ctx.resume();
                
                // 如果正在播放，先停止舊的並淡出
                if (this.noiseSource) {
                    try { this.noiseSource.stop(); } catch(e){}
                }

                this.currentType = type;
                const buffer = this.createBuffer(type);
                this.noiseSource = this.ctx.createBufferSource();
                this.noiseSource.buffer = buffer;
                this.noiseSource.loop = true;

                // 聲音修飾：低通濾波器 (讓聲音不刺耳，產生「平靜感」)
                const filter = this.ctx.createBiquadFilter();
                filter.type = 'lowpass';
                // 根據不同噪音類型調整濾波頻率
                if (type === 'white') filter.frequency.value = 1500; // 削減高頻
                else if (type === 'pink') filter.frequency.value = 2500;
                else filter.frequency.value = 500; // 布朗噪音本來就很低

                // 連接路徑：Source -> Filter -> Analyser -> Gain -> Out
                this.noiseSource.connect(filter);
                filter.connect(this.analyser);

                this.noiseSource.start();
                this.isPlaying = true;

                // 淡入效果
                this.mainGain.gain.cancelScheduledValues(this.ctx.currentTime);
                this.mainGain.gain.setValueAtTime(this.mainGain.gain.value, this.ctx.currentTime);
                this.mainGain.gain.linearRampToValueAtTime(this.volume, this.ctx.currentTime + 2);
            }

            setVolume(val) {
                this.volume = val;
                if (this.mainGain) {
                    this.mainGain.gain.setTargetAtTime(val, this.ctx.currentTime, 0.1);
                }
            }

            // 獲取平滑後的音量值 (0.0 ~ 1.0) 供視覺使用
            getSmoothedVolume() {
                if (!this.analyser) return 0;
                const dataArray = new Uint8Array(this.analyser.frequencyBinCount);
                this.analyser.getByteFrequencyData(dataArray);
                
                // 計算平均值
                let values = 0;
                const length = dataArray.length;
                for (let i = 0; i < length; i++) {
                    values += dataArray[i];
                }
                return (values / length) / 255;
            }

         * PART 1: AUDIO ENGINE
         * 包含靜音與淡入淡出邏輯

        class AudioEngine {
            constructor() {
                this.ctx = null;
                this.mainGain = null;
                this.source = null;
                this.analyser = null;
                this.volume = 0.6;
                this.isPlaying = false;
                this.isMuted = false;
            }

            init() {
                if (this.ctx) return;
                const AudioContext = window.AudioContext || window.webkitAudioContext;
                this.ctx = new AudioContext();
                
                this.mainGain = this.ctx.createGain();
                this.mainGain.gain.value = 0;
                this.mainGain.connect(this.ctx.destination);

                this.analyser = this.ctx.createAnalyser();
                this.analyser.fftSize = 256;
                this.analyser.smoothingTimeConstant = 0.85; // 平滑化數值
                this.analyser.connect(this.mainGain);
            }

            // 生成噪音波形
            createBuffer(type) {
                const bufferSize = 2 * this.ctx.sampleRate;
                const buffer = this.ctx.createBuffer(1, bufferSize, this.ctx.sampleRate);
                const data = buffer.getChannelData(0);
                let lastOut = 0;

                for (let i = 0; i < bufferSize; i++) {
                    const white = Math.random() * 2 - 1;
                    if (type === 'white') {
                        data[i] = white;
                    } else if (type === 'pink') {
                        const b0 = 0.99886 * lastOut + white * 0.0555179;
                        data[i] = b0 * 3.5;
                        lastOut = b0;
                    } else if (type === 'brown') {
                        const b0 = (lastOut + (0.02 * white)) / 1.02;
                        data[i] = b0 * 3.5;
                        lastOut = b0;
                    }
                }
                return buffer;
            }

            play(type) {
                this.init();
                if (this.ctx.state === 'suspended') this.ctx.resume();

                // 處理靜音邏輯
                if (type === 'off') {
                    this.mute();
                    return;
                }

                this.isMuted = false;
                
                // 停止當前聲音（淡出）
                if (this.source) {
                    try { this.source.stop(); } catch(e){}
                }

                // 建立新聲音源
                const buffer = this.createBuffer(type);
                this.source = this.ctx.createBufferSource();
                this.source.buffer = buffer;
                this.source.loop = true;

                // 濾波器：修飾聲音質感
                const filter = this.ctx.createBiquadFilter();
                filter.type = 'lowpass';
                if (type === 'white') filter.frequency.value = 800; // 降低白噪音刺耳感
                else if (type === 'pink') filter.frequency.value = 2000;
                else filter.frequency.value = 400;

                this.source.connect(filter);
                filter.connect(this.analyser);
                this.source.start();
                this.isPlaying = true;

                // 淡入 (Fade In)
                const now = this.ctx.currentTime;
                this.mainGain.gain.cancelScheduledValues(now);
                this.mainGain.gain.setValueAtTime(0, now);
                this.mainGain.gain.linearRampToValueAtTime(this.volume, now + 1.5);
            }

            mute() {
                if (!this.mainGain) return;
                this.isMuted = true;
                const now = this.ctx.currentTime;
                // 淡出 (Fade Out)
                this.mainGain.gain.cancelScheduledValues(now);
                this.mainGain.gain.setValueAtTime(this.mainGain.gain.value, now);
                this.mainGain.gain.exponentialRampToValueAtTime(0.001, now + 1);
                
                setTimeout(() => {
                    if (this.isMuted && this.source) {
                        this.source.stop();
                        this.isPlaying = false;
                    }
                }, 1000);
            }

            setVolume(val) {
                this.volume = val;
                if (this.mainGain && !this.isMuted) {
                    this.mainGain.gain.setTargetAtTime(val, this.ctx.currentTime, 0.1);
                }
            }

            getVolumeData() {
                if (!this.analyser || this.isMuted) return 0;
                const array = new Uint8Array(this.analyser.frequencyBinCount);
                this.analyser.getByteFrequencyData(array);
                let sum = 0;
                for (let i = 0; i < array.length; i++) sum += array[i];
                return (sum / array.length) / 255;
            }
        }

        /**
         * PART 1: AUDIO ENGINE (Soundscapes)
         * 包含間歇式雨聲邏輯與 LFO 調變
         */
        class AudioEngine {
            constructor() {
                this.ctx = null;
                this.masterGain = null;
                this.activeSource = null;
                this.lfoNodes = []; // 儲存 LFO 相關節點以便清除
                this.analyser = null;
                this.volume = 0.5;
                this.isMuted = false;
            }

            init() {
                if (this.ctx) return;
                const AudioContext = window.AudioContext || window.webkitAudioContext;
                this.ctx = new AudioContext();
                
                this.masterGain = this.ctx.createGain();
                this.masterGain.gain.value = 0;
                this.masterGain.connect(this.ctx.destination);

                this.analyser = this.ctx.createAnalyser();
                this.analyser.fftSize = 256;
                this.analyser.smoothingTimeConstant = 0.8;
                this.analyser.connect(this.masterGain);
            }

            // 基礎噪音生成 (White/Pink/Brown)
            createNoiseBuffer() {
                const bufferSize = 2 * this.ctx.sampleRate;
                const buffer = this.ctx.createBuffer(1, bufferSize, this.ctx.sampleRate);
                const data = buffer.getChannelData(0);
                
                let lastOut = 0;
                // 使用混合算法，先生成基礎 Buffer，播放時再濾波
                for (let i = 0; i < bufferSize; i++) {
                    const white = Math.random() * 2 - 1;
                    // 這裡只生成白噪音，特殊的顏色變化靠 Filter 處理
                    data[i] = white;
                }
                return buffer;
            }

            stopCurrent() {
                if (this.activeSource) {
                    try { this.activeSource.stop(); } catch(e){}
                    this.activeSource = null;
                }
                // 清除舊的 LFO 連接
                this.lfoNodes.forEach(node => {
                    try { node.stop(); node.disconnect(); } catch(e){}
                });
                this.lfoNodes = [];
            }

            play(type) {
                this.init();
                if (this.ctx.state === 'suspended') this.ctx.resume();

                if (type === 'off') {
                    this.mute();
                    return;
                }
                
                this.isMuted = false;
                this.stopCurrent();

                const buffer = this.createNoiseBuffer();
                this.activeSource = this.ctx.createBufferSource();
                this.activeSource.buffer = buffer;
                this.activeSource.loop = true;

                // --- 聲音塑形核心 ---
                const filter = this.ctx.createBiquadFilter();
                
                // 1. FOCUS (White Noise 變體)
                if (type === 'focus') {
                    filter.type = 'lowpass';
                    filter.frequency.value = 1200; // 稍微削減刺耳高頻
                    this.activeSource.connect(filter);
                } 
                
                // 2. RAIN (間歇式雨聲 - 核心修改)
                else if (type === 'rain') {
                    // 使用 Pink Noise 的濾波係數模擬
                    // 但因為 Web Audio 原生沒有 Pink Noise，我們用白噪音 + 強力 LowShelf/LowPass 模擬
                    const pinkFilter = this.ctx.createBiquadFilter();
                    pinkFilter.type = 'lowshelf';
                    pinkFilter.frequency.value = 400;
                    pinkFilter.gain.value = -10;

                    // 動態濾波器 (模擬風雨強弱)
                    filter.type = 'lowpass';
                    filter.frequency.value = 600; // 基礎頻率

                    // LFO (低頻振盪器) - 這是製造「間歇感」的關鍵
                    const lfo = this.ctx.createOscillator();
                    lfo.type = 'sine';
                    lfo.frequency.value = 0.15; // 非常慢的頻率 (約 6.6 秒一個循環)
                    
                    const lfoGain = this.ctx.createGain();
                    lfoGain.gain.value = 400; // 頻率變化幅度 (+/- 400Hz)

                    // 連接：LFO -> LFO Gain -> Filter Frequency
                    lfo.connect(lfoGain);
                    lfoGain.connect(filter.frequency);
                    lfo.start();
                    
                    this.lfoNodes.push(lfo, lfoGain);

                    // 串接路徑
                    this.activeSource.connect(pinkFilter);
                    pinkFilter.connect(filter);
                } 
                
                // 3. DEEP (Brown Noise 變體)
                else if (type === 'deep') {
                    filter.type = 'lowpass';
                    filter.frequency.value = 250; // 極低頻，像深海或雷鳴
                    this.activeSource.connect(filter);
                }

                // 統一輸出
                filter.connect(this.analyser);
                this.activeSource.start();

                // 淡入
                const now = this.ctx.currentTime;
                this.masterGain.gain.cancelScheduledValues(now);
                this.masterGain.gain.setValueAtTime(0, now);
                this.masterGain.gain.linearRampToValueAtTime(this.volume, now + 2);
            }

            mute() {
                if (!this.masterGain) return;
                this.isMuted = true;
                const now = this.ctx.currentTime;
                // 優雅淡出
                this.masterGain.gain.cancelScheduledValues(now);
                this.masterGain.gain.setValueAtTime(this.masterGain.gain.value, now);
                this.masterGain.gain.exponentialRampToValueAtTime(0.001, now + 1.5);
                
                setTimeout(() => {
                    if (this.isMuted) this.stopCurrent();
                }, 1600);
            }

            setVolume(val) {
                this.volume = val;
                if (!this.isMuted && this.masterGain) {
                    this.masterGain.gain.setTargetAtTime(val, this.ctx.currentTime, 0.1);
                }
            }

            getEnergy() {
                if (!this.analyser || this.isMuted) return 0;
                const arr = new Uint8Array(this.analyser.frequencyBinCount);
                this.analyser.getByteFrequencyData(arr);
                let sum = 0;
                for(let i=0; i<arr.length; i++) sum += arr[i];
                return (sum / arr.length) / 255;
            }
        }

        /**
         * PART 1: TEXTURE GENERATOR
         * 程序化生成 Bump Maps (質感貼圖)
         */
        class TextureGenerator {
            constructor() {}

            // 建立基礎 Canvas
            createCanvas() {
                const canvas = document.createElement('canvas');
                canvas.width = 512;
                canvas.height = 512;
                return canvas;
            }

            // 1. Focus: 拉絲金屬 (Brushed Metal)
            createBrushedTexture() {
                const canvas = this.createCanvas();
                const ctx = canvas.getContext('2d');
                
                // 填充底色
                ctx.fillStyle = '#808080';
                ctx.fillRect(0, 0, 512, 512);

                // 繪製大量隨機水平線
                for(let i=0; i<3000; i++) {
                    const shade = Math.floor(Math.random() * 255);
                    ctx.fillStyle = `rgb(${shade},${shade},${shade})`;
                    ctx.globalAlpha = 0.1;
                    const h = Math.random() * 512;
                    const w = Math.random() * 200 + 50;
                    ctx.fillRect(Math.random() * 512 - 50, h, w, Math.random() * 2);
                }
                
                const tex = new THREE.CanvasTexture(canvas);
                tex.wrapS = THREE.RepeatWrapping;
                tex.wrapT = THREE.RepeatWrapping;
                return tex;
            }

            // 2. Rain: 水痕噪點 (Liquid Noise)
            createLiquidTexture() {
                const canvas = this.createCanvas();
                const ctx = canvas.getContext('2d');
                
                ctx.fillStyle = '#808080';
                ctx.fillRect(0, 0, 512, 512);

                // 繪製柔和的圓斑
                for(let i=0; i<100; i++) {
                    const x = Math.random() * 512;
                    const y = Math.random() * 512;
                    const r = Math.random() * 60 + 20;
                    const grad = ctx.createRadialGradient(x, y, 0, x, y, r);
                    grad.addColorStop(0, 'rgba(255,255,255,0.2)');
                    grad.addColorStop(1, 'rgba(0,0,0,0)');
                    ctx.fillStyle = grad;
                    ctx.beginPath();
                    ctx.arc(x, y, r, 0, Math.PI*2);
                    ctx.fill();
                }
                return new THREE.CanvasTexture(canvas);
            }

            // 3. Deep: 粗糙岩石 (Granite Noise)
            createRockTexture() {
                const canvas = this.createCanvas();
                const ctx = canvas.getContext('2d');
                
                const imgData = ctx.createImageData(512, 512);
                const data = imgData.data;
                
                for (let i = 0; i < data.length; i += 4) {
                    const val = Math.random() * 255;
                    data[i] = val;     // R
                    data[i+1] = val;   // G
                    data[i+2] = val;   // B
                    data[i+3] = 255;   // A
                }
                ctx.putImageData(imgData, 0, 0);
                return new THREE.CanvasTexture(canvas);
            }
        }

        /**
         * PART 2: AUDIO ENGINE (With Tuner)
         * 新增 Delay (Space) 與 Lowpass/Highpass (Tone)
         */
        class AudioEngine {
            constructor() {
                this.ctx = null;
                this.masterGain = null;
                this.source = null;
                
                // 效果節點
                this.toneFilter = null; // 控制音色明暗
                this.delayNode = null;  // 空間感
                this.feedbackNode = null;
                this.wetGain = null;    // 空間感強度

                this.analyser = null;
                this.lfoNodes = [];
                
                this.volume = 0.6;
                this.toneVal = 0.7; // 0~1
                this.spaceVal = 0.3; // 0~1
                this.isMuted = false;
            }

            init() {
                if (this.ctx) return;
                const AudioContext = window.AudioContext || window.webkitAudioContext;
                this.ctx = new AudioContext();

                // 1. Master Chain
                this.masterGain = this.ctx.createGain();
                this.masterGain.gain.value = 0;
                this.masterGain.connect(this.ctx.destination);

                this.analyser = this.ctx.createAnalyser();
                this.analyser.fftSize = 256;
                this.analyser.smoothingTimeConstant = 0.85;
                this.analyser.connect(this.masterGain);

                // 2. Effects Chain (Insert Effects)
                // Tone Filter (Lowpass)
                this.toneFilter = this.ctx.createBiquadFilter();
                this.toneFilter.type = 'lowpass';
                this.updateTone(); // Apply initial value

                // Delay / Space
                this.delayNode = this.ctx.createDelay();
                this.delayNode.delayTime.value = 0.4; // 0.4s echo
                
                this.feedbackNode = this.ctx.createGain();
                this.feedbackNode.gain.value = 0.4; // Decay

                this.wetGain = this.ctx.createGain();
                this.updateSpace(); // Apply initial value

                // Delay Loop: Delay -> Feedback -> Delay
                this.delayNode.connect(this.feedbackNode);
                this.feedbackNode.connect(this.delayNode);
                
                // Delay Output: Delay -> WetGain -> Analyser (Mix point)
                this.delayNode.connect(this.wetGain);
                this.wetGain.connect(this.analyser);

                // Filter Output: Tone -> Analyser (Dry Signal)
                this.toneFilter.connect(this.analyser);
                // Filter also feeds into Delay
                this.toneFilter.connect(this.delayNode);
            }

            createBuffer(type) {
                const size = 2 * this.ctx.sampleRate;
                const buffer = this.ctx.createBuffer(1, size, this.ctx.sampleRate);
                const data = buffer.getChannelData(0);
                for(let i=0; i<size; i++) data[i] = Math.random() * 2 - 1;
                return buffer;
            }

            stop() {
                if (this.source) { try { this.source.stop(); } catch(e){} }
                this.lfoNodes.forEach(n => { try{ n.stop(); n.disconnect(); } catch(e){} });
                this.lfoNodes = [];
            }

            play(type) {
                this.init();
                if (this.ctx.state === 'suspended') this.ctx.resume();
                if (type === 'off') { this.mute(); return; }
                
                this.isMuted = false;
                this.stop();

                const buffer = this.createBuffer(type);
                this.source = this.ctx.createBufferSource();
                this.source.buffer = buffer;
                this.source.loop = true;

                // Pre-processing Filters (Specific to sound type)
                let outputNode = this.source;

                if (type === 'focus') {
                    // White noise adjustment
                    const f = this.ctx.createBiquadFilter();
                    f.type = 'highpass';
                    f.frequency.value = 100; 
                    this.source.connect(f);
                    outputNode = f;
                } else if (type === 'rain') {
                    // Pink-ish + LFO Logic
                    const pink = this.ctx.createBiquadFilter();
                    pink.type = 'lowshelf'; 
                    pink.frequency.value = 500; pink.gain.value = -10;

                    const lfoFilter = this.ctx.createBiquadFilter();
                    lfoFilter.type = 'lowpass';
                    lfoFilter.frequency.value = 800;

                    // LFO
                    const lfo = this.ctx.createOscillator();
                    lfo.frequency.value = 0.12;
                    const lfoGain = this.ctx.createGain();
                    lfoGain.gain.value = 600;
                    lfo.connect(lfoGain);
                    lfoGain.connect(lfoFilter.frequency);
                    lfo.start();
                    this.lfoNodes.push(lfo, lfoGain);

                    this.source.connect(pink);
                    pink.connect(lfoFilter);
                    outputNode = lfoFilter;
                } else if (type === 'deep') {
                    // Brown-ish
                    const brown = this.ctx.createBiquadFilter();
                    brown.type = 'lowpass';
                    brown.frequency.value = 300;
                    this.source.connect(brown);
                    outputNode = brown;
                }

                // Connect sound source to our Tone Filter (Entry of Effects Chain)
                outputNode.connect(this.toneFilter);
                this.source.start();

                // Fade In
                const now = this.ctx.currentTime;
                this.masterGain.gain.cancelScheduledValues(now);
                this.masterGain.gain.setValueAtTime(0, now);
                this.masterGain.gain.linearRampToValueAtTime(this.volume, now + 1.5);
            }

            mute() {
                if(!this.masterGain) return;
                this.isMuted = true;
                const now = this.ctx.currentTime;
                this.masterGain.gain.cancelScheduledValues(now);
                this.masterGain.gain.setValueAtTime(this.masterGain.gain.value, now);
                this.masterGain.gain.exponentialRampToValueAtTime(0.001, now + 1);
                setTimeout(() => { if(this.isMuted) this.stop(); }, 1200);
            }

            // --- Tuner Functions ---
            
            // 音色：調整 Lowpass Cutoff (300Hz ~ 15000Hz)
            updateTone() {
                if(!this.toneFilter) return;
                // Logarithmic mapping for better feel
                const freq = 300 * Math.pow(15000/300, this.toneVal);
                this.toneFilter.frequency.setTargetAtTime(freq, this.ctx.currentTime, 0.1);
            }

            // 空間：調整 Wet Mix Level (0 ~ 1.0)
            updateSpace() {
                if(!this.wetGain) return;
                this.wetGain.gain.setTargetAtTime(this.spaceVal * 0.8, this.ctx.currentTime, 0.1);
            }

            setParam(key, val) {
                if(key === 'volume') {
                    this.volume = val;
                    if(!this.isMuted && this.masterGain) this.masterGain.gain.setTargetAtTime(val, this.ctx.currentTime, 0.1);
                } else if(key === 'tone') {
                    this.toneVal = val;
                    this.updateTone();
                } else if(key === 'space') {
                    this.spaceVal = val;
                    this.updateSpace();
                }
            }

            getEnergy() {
                if (!this.analyser || this.isMuted) return 0;
                const arr = new Uint8Array(this.analyser.frequencyBinCount);
                this.analyser.getByteFrequencyData(arr);
                let sum = 0;
                for(let i=0; i<arr.length; i++) sum += arr[i];
                return (sum / arr.length) / 255;
            }
        }

        /**
         * PART 3: VISUAL SYSTEM (Procedural Materials)
         */

                this.wet.connect(this.analyser); // Delay path
                this.toneFilter.connect(this.analyser); // Dry path
                this.toneFilter.connect(this.delay); // Send to delay
            }

            stop() {
                if(this.source) { try{this.source.stop();}catch(e){} }
                this.lfoNodes.forEach(n => { try{n.stop(); n.disconnect();}catch(e){} });
                this.lfoNodes = [];
            }

            createBuffer(type) {
                const b = this.ctx.createBuffer(1, 2*this.ctx.sampleRate, this.ctx.sampleRate);
                const d = b.getChannelData(0);
                for(let i=0; i<d.length; i++) d[i] = Math.random()*2 - 1;
                return b;
            }

            play(type) {
                this.init();
                if(this.ctx.state === 'suspended') this.ctx.resume();
                if(type === 'off') { this.mute(); return; }

                this.isMuted = false;
                this.stop();

                const buf = this.createBuffer(type);
                this.source = this.ctx.createBufferSource();
                this.source.buffer = buf;
                this.source.loop = true;

                let out = this.source;

                // Sound Sculpting
                if (type === 'focus') {
                    const hp = this.ctx.createBiquadFilter();
                    hp.type = 'highpass'; hp.frequency.value = 150;
                    this.source.connect(hp); out = hp;
                } else if (type === 'rain') {
                    // Rain Simulation with LFO
                    const pink = this.ctx.createBiquadFilter();
                    pink.type = 'lowshelf'; pink.frequency.value = 500; pink.gain.value = -12;
                    
                    const dynamic = this.ctx.createBiquadFilter();
                    dynamic.type = 'lowpass'; dynamic.frequency.value = 700;

                    const lfo = this.ctx.createOscillator();
                    lfo.frequency.value = 0.15; // Slow wave
                    const lfoG = this.ctx.createGain();
                    lfoG.gain.value = 500;
                    lfo.connect(lfoG);
                    lfoG.connect(dynamic.frequency);
                    lfo.start();
                    this.lfoNodes.push(lfo, lfoG);

                    this.source.connect(pink); pink.connect(dynamic); out = dynamic;
                } else if (type === 'deep') {
                    const lp = this.ctx.createBiquadFilter();
                    lp.type = 'lowpass'; lp.frequency.value = 250;
                    this.source.connect(lp); out = lp;
                }

                out.connect(this.toneFilter);
                this.source.start();

                // Fade In
                const t = this.ctx.currentTime;
                this.master.gain.cancelScheduledValues(t);
                this.master.gain.setValueAtTime(0, t);
                this.master.gain.linearRampToValueAtTime(this.vol, t+2);
            }

            mute() {
                if(!this.master) return;
                this.isMuted = true;
                const t = this.ctx.currentTime;
                this.master.gain.cancelScheduledValues(t);
                this.master.gain.setValueAtTime(this.master.gain.value, t);
                this.master.gain.exponentialRampToValueAtTime(0.001, t+1);
                setTimeout(() => { if(this.isMuted) this.stop(); }, 1200);
            }

            updateTone() {
                if(!this.toneFilter) return;
                // Log mapping: 200Hz -> 12000Hz
                const val = 200 * Math.pow(60, this.tone);
                this.toneFilter.frequency.setTargetAtTime(val, this.ctx.currentTime, 0.1);
            }

            updateSpace() {
                if(!this.wet) return;
                this.wet.gain.setTargetAtTime(this.space, this.ctx.currentTime, 0.1);
            }

            set(p, v) {
                if(p==='vol') { 
                    this.vol = v; 
                    if(!this.isMuted && this.master) this.master.gain.setTargetAtTime(v, this.ctx.currentTime, 0.1); 
                }
                else if(p==='tone') { this.tone = v; this.updateTone(); }
                else if(p==='space') { this.space = v; this.updateSpace(); }
            }

            getEnergy() {
                if(!this.analyser || this.isMuted) return 0;
                const d = new Uint8Array(this.analyser.frequencyBinCount);
                this.analyser.getByteFrequencyData(d);
                let s = 0; for(let i=0; i<d.length; i++) s+=d[i];
                return (s/d.length)/255;
            }
        }

            createRipple() {

                    y: Math.random() * this.canvas.height,

                };
            }

            createBar(randomY = false) {
                return {
                    y: randomY ? Math.random() * this.canvas.height : -50,
                    h: Math.random() * 40 + 10,
                    speed: Math.random() * 0.5 + 0.2,
                    alpha: 0
                };
            }

            render(energy) {
                const w = this.canvas.width;
                const h = this.canvas.height;
                const time = performance.now() * 0.001;

                // 清除畫面 (帶有拖尾效果)
                this.ctx.fillStyle = this.mode === 'off' ? 'rgba(5,5,5,0.1)' : 'rgba(5,5,5,0.2)';
                this.ctx.fillRect(0, 0, w, h);

                    // --- FOCUS: 和緩的線條波形 ---
                    this.ctx.strokeStyle = `rgba(200, 200, 200, ${0.1 + energy * 0.2})`;
                    this.ctx.lineWidth = 1;
                    
                    // 繪製 3 條正弦波
                    for (let i = 0; i < 3; i++) {
                        this.ctx.beginPath();
                        const offset = i * 100;
                        const speed = time * (0.5 + i * 0.1);
                        for (let x = 0; x < w; x += 10) {
                            const y = h/2 + Math.sin(x * 0.005 + speed + offset) * (50 + energy * 100);
                            if (x === 0) this.ctx.moveTo(x, y);
                            else this.ctx.lineTo(x, y);
                        }
                        this.ctx.stroke();
                    }
                } 
                else if (this.mode === 'rain') {
                    // --- RAIN: 不規則漣漪圓圈 ---
                    // 隨機生成新漣漪
                    if (Math.random() < 0.05 + energy * 0.1) {
                        this.elements.push(this.createRipple());
                    }

                    // 更新與繪製
                    for (let i = this.elements.length - 1; i >= 0; i--) {
                        const p = this.elements[i];
                        p.r += p.speed;
                        p.alpha -= 0.005;
                        
                        this.ctx.beginPath();
                        this.ctx.arc(p.x, p.y, p.r, 0, Math.PI * 2);
                        this.ctx.strokeStyle = `rgba(100, 200, 255, ${p.alpha})`;
                        this.ctx.lineWidth = 1.5;
                        this.ctx.stroke();

                        if (p.alpha <= 0) this.elements.splice(i, 1);
                    }
                } 
                else if (this.mode === 'deep') {
                    // --- DEEP: 向下漸變的漸層條紋 ---
                    if (Math.random() < 0.02) this.elements.push(this.createBar());

                    for (let i = this.elements.length - 1; i >= 0; i--) {
                        const b = this.elements[i];
                        b.y += b.speed;
                        // 淡入淡出邏輯
                        if (b.y < h * 0.2) b.alpha = Math.min(b.alpha + 0.01, 0.3);
                        else if (b.y > h * 0.8) b.alpha -= 0.005;

                        // 繪製
                        const grad = this.ctx.createLinearGradient(0, b.y, 0, b.y + b.h);
                        grad.addColorStop(0, `rgba(255, 150, 50, 0)`);
                        grad.addColorStop(0.5, `rgba(255, 150, 50, ${b.alpha * (0.5 + energy)})`);
                        grad.addColorStop(1, `rgba(255, 150, 50, 0)`);
                        
                        this.ctx.fillStyle = grad;
                        this.ctx.fillRect(0, b.y, w, b.h);

                        if (b.y > h || b.alpha <= 0) this.elements.splice(i, 1);
                    }
                }
            }
        }

                for(let i=0;i<80;i++) {

                    g.addColorStop(0,'rgba(255,255,255,0.3)'); g.addColorStop(1,'transparent');
                    ctx.fillStyle=g; ctx.beginPath(); ctx.arc(x,y,r,0,6.28); ctx.fill();
                }
                return new THREE.CanvasTexture(c);

                    const v=Math.random()*255;

                ctx.putImageData(id,0,0); return new THREE.CanvasTexture(c);
            }

        /**
         * PART 1: AUDIO ENGINE (Refined & Lower Volume)
         * - 純淨 (Pure) / 夢幻 (Reverie) / 深淵 (Deep)
         * - 全體音量降低 50%
         */
        class AudioEngine {
            constructor() {
                this.ctx = null;
                this.master = null;
                this.source = null;
                this.toneFilter = null;
                this.delay = null;
                this.wet = null;
                this.analyser = null;
                this.lfoNodes = [];
                
                // 基礎音量預設降低
                this.baseGain = 0.5; // 全局衰減係數 (降低 50%)
                this.vol = 0.5;      // 使用者設定值
                
                this.tone = 0.7;
                this.space = 0.3;
                this.isMuted = false;
            }

            init() {
                if (this.ctx) return;
                const AC = window.AudioContext || window.webkitAudioContext;
                this.ctx = new AC();
                
                this.master = this.ctx.createGain();
                this.master.gain.value = 0;
                this.master.connect(this.ctx.destination);

                this.analyser = this.ctx.createAnalyser();
                this.analyser.fftSize = 256;
                this.analyser.smoothingTimeConstant = 0.85; // 更平滑的視覺反應
                this.analyser.connect(this.master);

                // Insert Effects
                this.toneFilter = this.ctx.createBiquadFilter();
                this.toneFilter.type = 'lowpass';
                this.updateTone();

                this.delay = this.ctx.createDelay();
                this.delay.delayTime.value = 0.4; // 400ms delay
                const feedback = this.ctx.createGain();
                feedback.gain.value = 0.3; // 稍微減少回授
                this.delay.connect(feedback);
                feedback.connect(this.delay);

                this.wet = this.ctx.createGain();
                this.updateSpace();
                
                // Routing
                this.delay.connect(this.wet);
                this.wet.connect(this.analyser);
                this.toneFilter.connect(this.analyser); // Dry
                this.toneFilter.connect(this.delay);    // Send
            }

            stop() {
                if(this.source) { try{this.source.stop();}catch(e){} }
                this.lfoNodes.forEach(n => { try{n.stop(); n.disconnect();}catch(e){} });
                this.lfoNodes = [];
            }

            createBuffer() {
                // Generate White Noise Buffer
                const b = this.ctx.createBuffer(1, 2*this.ctx.sampleRate, this.ctx.sampleRate);
                const d = b.getChannelData(0);
                for(let i=0; i<d.length; i++) d[i] = Math.random()*2 - 1;
                return b;
            }

            play(type) {
                this.init();
                if(this.ctx.state === 'suspended') this.ctx.resume();
                if(type === 'off') { this.mute(); return; }

                this.isMuted = false;
                this.stop();

                const buf = this.createBuffer();
                this.source = this.ctx.createBufferSource();
                this.source.buffer = buf;
                this.source.loop = true;

                let out = this.source;

                // --- Sound Sculpting ---
                if (type === 'pure') { // 原 Focus
                    // 乾淨的高通濾波，去除低頻轟鳴
                    const hp = this.ctx.createBiquadFilter();
                    hp.type = 'highpass'; hp.frequency.value = 200;
                    this.source.connect(hp); out = hp;
                } 
                else if (type === 'reverie') { // 原 Rain
                    // 間歇性 LFO 雨聲
                    const pink = this.ctx.createBiquadFilter();
                    pink.type = 'lowshelf'; pink.frequency.value = 500; pink.gain.value = -10;
                    
                    const dynamic = this.ctx.createBiquadFilter();
                    dynamic.type = 'lowpass'; dynamic.frequency.value = 800;

                    const lfo = this.ctx.createOscillator();
                    lfo.frequency.value = 0.12; // 緩慢的波浪
                    const lfoG = this.ctx.createGain();
                    lfoG.gain.value = 400; 
                    lfo.connect(lfoG);
                    lfoG.connect(dynamic.frequency);
                    lfo.start();
                    this.lfoNodes.push(lfo, lfoG);

                    this.source.connect(pink); pink.connect(dynamic); out = dynamic;
                } 
                else if (type === 'deep') {
                    // 極低頻
                    const lp = this.ctx.createBiquadFilter();
                    lp.type = 'lowpass'; lp.frequency.value = 180; // 更低沉
                    this.source.connect(lp); out = lp;
                }

                out.connect(this.toneFilter);
                this.source.start();

                // Fade In to Target Volume (Base * UserVol)
                const t = this.ctx.currentTime;
                const targetVol = this.vol * this.baseGain;
                this.master.gain.cancelScheduledValues(t);
                this.master.gain.setValueAtTime(0, t);
                this.master.gain.linearRampToValueAtTime(targetVol, t+2.5);
            }

            mute() {
                if(!this.master) return;
                this.isMuted = true;
                const t = this.ctx.currentTime;
                this.master.gain.cancelScheduledValues(t);
                this.master.gain.setValueAtTime(this.master.gain.value, t);
                this.master.gain.exponentialRampToValueAtTime(0.001, t+1.5);
                setTimeout(() => { if(this.isMuted) this.stop(); }, 1600);
            }

            updateTone() {
                if(!this.toneFilter) return;
                // Log mapping
                const val = 150 * Math.pow(80, this.tone);
                this.toneFilter.frequency.setTargetAtTime(val, this.ctx.currentTime, 0.2);
            }

            updateSpace() {
                if(!this.wet) return;
                this.wet.gain.setTargetAtTime(this.space * 0.8, this.ctx.currentTime, 0.2);
            }

            set(p, v) {
                if(p==='vol') { 
                    this.vol = v; 
                    if(!this.isMuted && this.master) {
                        this.master.gain.setTargetAtTime(v * this.baseGain, this.ctx.currentTime, 0.1); 
                    }
                }
                else if(p==='tone') { this.tone = v; this.updateTone(); }
                else if(p==='space') { this.space = v; this.updateSpace(); }
            }

            getEnergy() {
                if(!this.analyser || this.isMuted) return 0;
                const d = new Uint8Array(this.analyser.frequencyBinCount);
                this.analyser.getByteFrequencyData(d);
                let s = 0; for(let i=0; i<d.length; i++) s+=d[i];
                return (s/d.length)/255;
            }
        }

        /**
         * PART 2: BACKGROUND ART (Refined DEEP Bloom)

        class BackgroundArt {
            constructor() {
                this.canvas = document.getElementById('bg-canvas');
                this.ctx = this.canvas.getContext('2d');
                this.mode = 'off';
                this.elements = [];
                this.resize();
                window.addEventListener('resize', () => this.resize());
            }

            resize() {
                this.canvas.width = window.innerWidth;
                this.canvas.height = window.innerHeight;
            }

            setMode(m) {
                this.mode = m;
                this.elements = [];
                if (m === 'deep') {
                    // 預先生成一些 Deep 粒子
                    for(let i=0; i<8; i++) this.elements.push(this.createBar(true));
                }
            }

            // Reverie Ripple
            createRipple() {
                return {
                    x: Math.random() * this.canvas.width,
                    y: Math.random() * this.canvas.height,
                    r: 0,
                    speed: Math.random() * 0.4 + 0.1,
                    alpha: Math.random() * 0.3 + 0.1
                };
            }

            // Deep Bar (Bloom Style)
            createBar(randomY = false) {
                const w = this.canvas.width;
                const h = this.canvas.height;
                return {
                    x: Math.random() * w, // 隨機水平位置 (如果需要垂直降落感，其實X不重要，重要的是漸層)
                    y: randomY ? Math.random() * h : -100,
                    width: w, // 全寬漸層
                    height: Math.random() * 150 + 50, // 條紋高度
                    speed: Math.random() * 0.2 + 0.05, // 極慢
                    alpha: 0,
                    life: Math.random() * 0.5 + 0.5
                };
            }

            render(energy) {
                const w = this.canvas.width;
                const h = this.canvas.height;
                const t = performance.now() * 0.001;
                
                // 1. Clear with trailing effect (更暗的拖尾)
                this.ctx.globalCompositeOperation = 'source-over';
                this.ctx.fillStyle = 'rgba(2, 2, 2, 0.15)'; 
                this.ctx.fillRect(0, 0, w, h);

                // 2. Draw Modes
                if (this.mode === 'pure') {
                    // --- PURE: 絲滑的線條波 ---
                    this.ctx.globalCompositeOperation = 'lighter';
                    this.ctx.strokeStyle = `rgba(200, 220, 255, ${0.05 + energy * 0.15})`;
                    this.ctx.lineWidth = 1;
                    
                    for (let i = 0; i < 4; i++) {
                        this.ctx.beginPath();
                        const speed = t * (0.3 + i * 0.05);
                        for (let x = 0; x < w; x += 15) {
                            const y = h/2 + Math.sin(x * 0.003 + speed + i) * (60 + energy * 80);
                            if(x===0) this.ctx.moveTo(x,y); else this.ctx.lineTo(x,y);
                        }
                        this.ctx.stroke();
                    }
                }
                else if (this.mode === 'reverie') {
                    // --- REVERIE: 漣漪 ---
                    this.ctx.globalCompositeOperation = 'screen';
                    if (Math.random() < 0.03 + energy * 0.05) this.elements.push(this.createRipple());

                    for (let i = this.elements.length - 1; i >= 0; i--) {
                        const p = this.elements[i];
                        p.r += p.speed;
                        p.alpha -= 0.002;
                        
                        this.ctx.beginPath();
                        this.ctx.arc(p.x, p.y, p.r, 0, Math.PI * 2);
                        this.ctx.strokeStyle = `rgba(100, 200, 255, ${p.alpha})`;
                        this.ctx.lineWidth = 1;
                        this.ctx.stroke();
                        if (p.alpha <= 0) this.elements.splice(i, 1);
                    }
                }
                else if (this.mode === 'deep') {
                    // --- DEEP: Bloom Gradient Bars ---
                    // 使用 screen 混合模式讓重疊處發光
                    this.ctx.globalCompositeOperation = 'screen'; 
                    
                    // 偶爾生成新條紋
                    if (Math.random() < 0.01) this.elements.push(this.createBar());

                    for (let i = this.elements.length - 1; i >= 0; i--) {
                        const b = this.elements[i];
                        b.y += b.speed;
                        
                        // 淡入淡出生命週期
                        if (b.y < h * 0.3) b.alpha = Math.min(b.alpha + 0.002, 0.15); // 最高透明度很低，保持幽暗
                        else if (b.y > h * 0.7) b.alpha -= 0.001;

                        // 核心：Bloom Effect
                        this.ctx.shadowBlur = 40; // 強烈的模糊
                        this.ctx.shadowColor = `rgba(255, 100, 40, ${b.alpha * 0.8})`; // 琥珀色光暈
                        
                        // 繪製水平漸層條紋
                        const grad = this.ctx.createLinearGradient(0, b.y, 0, b.y + b.height);
                        // 頭部稍微亮一點
                        grad.addColorStop(0, `rgba(255, 120, 60, 0)`);
                        grad.addColorStop(0.3, `rgba(255, 120, 60, ${b.alpha * (0.5 + energy)})`);
                        grad.addColorStop(1, `rgba(100, 40, 20, 0)`);
                        
                        this.ctx.fillStyle = grad;
                        // 畫一個橫跨畫面的矩形，但我們只畫在隨機高度
                        this.ctx.fillRect(0, b.y, w, b.height);
                        
                        // Reset Shadow to avoid affecting next frame too much (though we clear rect)
                        this.ctx.shadowBlur = 0;

                        if (b.y > h + 100 || b.alpha <= 0) this.elements.splice(i, 1);
                    }
                }
            }
        }

        class TextureGen {
            createCanvas() { const c = document.createElement('canvas'); c.width=512; c.height=512; return c; }
            pureTex() {
                const c = this.createCanvas(); const ctx = c.getContext('2d');
                ctx.fillStyle = '#808080'; ctx.fillRect(0,0,512,512);
                for(let i=0;i<4000;i++) {
                    const v = Math.random()*255; ctx.fillStyle=`rgba(${v},${v},${v},0.08)`;
                    ctx.fillRect(Math.random()*512-50, Math.random()*512, Math.random()*200+50, 1);
                }
                return new THREE.CanvasTexture(c);
            }
            reverieTex() {
                const c = this.createCanvas(); const ctx = c.getContext('2d');
                ctx.fillStyle = '#808080'; ctx.fillRect(0,0,512,512);
                for(let i=0;i<100;i++) {
                    const x=Math.random()*512, y=Math.random()*512, r=Math.random()*60+10;
                    const g = ctx.createRadialGradient(x,y,0,x,y,r);
                    g.addColorStop(0,'rgba(255,255,255,0.2)'); g.addColorStop(1,'transparent');
                    ctx.fillStyle=g; ctx.beginPath(); ctx.arc(x,y,r,0,6.28); ctx.fill();
                }
                return new THREE.CanvasTexture(c);
            }
            deepTex() {
                const c = this.createCanvas(); const ctx = c.getContext('2d');
                const id = ctx.createImageData(512,512);
                for(let i=0;i<id.data.length;i+=4){
                    const v=Math.random()*200; // Darker noise
                    id.data[i]=v; id.data[i+1]=v; id.data[i+2]=v; id.data[i+3]=255;
                }
                ctx.putImageData(id,0,0); return new THREE.CanvasTexture(c);
            }
        }

        class Scene3D {
            constructor() {
                this.con = document.getElementById('three-canvas-container');
                this.tg = new TextureGen();
                this.mats = {};
                this.mx=0; this.my=0;
            }

            init() {
                this.scene = new THREE.Scene();
                this.cam = new THREE.PerspectiveCamera(45, window.innerWidth/window.innerHeight, 0.1, 100);
                this.cam.position.set(0,0,10);
                this.ren = new THREE.WebGLRenderer({ alpha: true, antialias: true });
                this.ren.setSize(window.innerWidth, window.innerHeight);
                this.ren.setPixelRatio(Math.min(window.devicePixelRatio, 2));
                this.ren.shadowMap.enabled = true;
                this.ren.shadowMap.type = THREE.PCFSoftShadowMap;
                this.con.appendChild(this.ren.domElement);

                // Textures
                const t1 = this.tg.pureTex();
                const t2 = this.tg.reverieTex();
                const t3 = this.tg.deepTex();

                // Materials
                this.mats.pure = new THREE.MeshStandardMaterial({
                    color: 0xffffff, roughness: 0.25, metalness: 0.9, bumpMap: t1, bumpScale: 0.05
                });
                this.mats.reverie = new THREE.MeshStandardMaterial({
                    color: 0x81d4fa, roughness: 0.1, metalness: 0.5, bumpMap: t2, bumpScale: 0.1,
                    transparent: true, opacity: 0.9
                });
                this.mats.deep = new THREE.MeshStandardMaterial({
                    color: 0x1a1a1a, roughness: 1.0, metalness: 0.0, bumpMap: t3, bumpScale: 0.5
                });
                this.mats.mute = new THREE.MeshBasicMaterial({
                    color: 0x333333, wireframe: true, transparent: true, opacity: 0.15
                });

                // Objects
                this.sphere = new THREE.Mesh(new THREE.IcosahedronGeometry(1.6, 30), this.mats.pure);
                this.sphere.castShadow = true; this.sphere.receiveShadow = true;
                this.scene.add(this.sphere);

                const planeMat = new THREE.MeshStandardMaterial({ 
                    color: 0x000000, roughness: 1, metalness: 0,
                    transparent: true, opacity: 0.6
                });
                this.plane = new THREE.Mesh(new THREE.PlaneGeometry(80,80), planeMat);
                this.plane.rotation.x = -Math.PI/2;
                this.plane.position.y = -3.5;

        /**
         * 0. NOISE EFFECT (Mono Grain)

        class NoiseLayer {
            constructor() {
                this.canvas = document.getElementById('noise-canvas');
                this.ctx = this.canvas.getContext('2d');
                this.resize();
                window.addEventListener('resize', () => this.resize());
                this.loop();
            }
            resize() {
                this.canvas.width = window.innerWidth;
                this.canvas.height = window.innerHeight;
            }
            loop() {
                const w = this.canvas.width;
                const h = this.canvas.height;
                const idata = this.ctx.createImageData(w, h);
                const buffer32 = new Uint32Array(idata.data.buffer);
                const len = buffer32.length;

                for (let i = 0; i < len; i++) {
                    if (Math.random() < 0.1) { // 雜訊密度
                        // 黑色雜訊，Alpha 隨機
                        buffer32[i] = ((Math.random() * 50)|0) << 24; 
                    }
                }
                this.ctx.putImageData(idata, 0, 0);
                requestAnimationFrame(() => this.loop());
            }
        }

        /**
         * 1. AUDIO ENGINE

        class AudioEngine {
            constructor() {
                this.ctx = null;
                this.baseGain = 0.5; // 音量衰減 50%
                this.vol = 0.5;
                this.tone = 0.7; this.space = 0.3;
                this.lfoNodes = [];
            }

            init() {
                if(this.ctx) return;
                const AC = window.AudioContext || window.webkitAudioContext;
                this.ctx = new AC();
                
                this.master = this.ctx.createGain();
                this.master.gain.value = 0;
                this.master.connect(this.ctx.destination);

                this.analyser = this.ctx.createAnalyser();
                this.analyser.fftSize = 256;
                this.analyser.smoothingTimeConstant = 0.85;
                this.analyser.connect(this.master);

                this.toneFilter = this.ctx.createBiquadFilter();
                this.toneFilter.type = 'lowpass';
                
                this.delay = this.ctx.createDelay();
                this.delay.delayTime.value = 0.4;
                const fb = this.ctx.createGain();
                fb.gain.value = 0.3;
                this.delay.connect(fb); fb.connect(this.delay);

                this.wet = this.ctx.createGain();
                this.updateParams();

                this.delay.connect(this.wet);
                this.wet.connect(this.analyser);
                this.toneFilter.connect(this.analyser);
                this.toneFilter.connect(this.delay);
            }

            play(type) {
                this.init();
                if(this.ctx.state === 'suspended') this.ctx.resume();
                
                if(this.source) { try{this.source.stop();}catch(e){} }
                this.lfoNodes.forEach(n=>{ try{n.stop();n.disconnect();}catch(e){} });
                this.lfoNodes=[];

                if(type === 'off') { this.mute(); return; }
                this.isMuted = false;

                const buf = this.ctx.createBuffer(1, 2*this.ctx.sampleRate, this.ctx.sampleRate);
                const d = buf.getChannelData(0);
                for(let i=0; i<d.length; i++) d[i] = Math.random()*2-1;

                this.source = this.ctx.createBufferSource();
                this.source.buffer = buf;
                this.source.loop = true;

                let out = this.source;
                
                if(type === 'pure') {
                    const hp = this.ctx.createBiquadFilter();
                    hp.type = 'highpass'; hp.frequency.value = 200;
                    this.source.connect(hp); out = hp;
                } else if(type === 'reverie') {
                    const pink = this.ctx.createBiquadFilter();
                    pink.type = 'lowshelf'; pink.frequency.value = 500; pink.gain.value = -12;
                    const dyn = this.ctx.createBiquadFilter();
                    dyn.type = 'lowpass'; dyn.frequency.value = 800;
                    
                    const lfo = this.ctx.createOscillator();
                    lfo.frequency.value = 0.12;
                    const lg = this.ctx.createGain(); lg.gain.value = 400;
                    lfo.connect(lg); lg.connect(dyn.frequency);
                    lfo.start(); this.lfoNodes.push(lfo, lg);

                    this.source.connect(pink); pink.connect(dyn); out = dyn;
                } else if(type === 'deep') {
                    const lp = this.ctx.createBiquadFilter();
                    lp.type = 'lowpass'; lp.frequency.value = 160;
                    this.source.connect(lp); out = lp;
                }

                out.connect(this.toneFilter);
                this.source.start();

                const t = this.ctx.currentTime;
                this.master.gain.cancelScheduledValues(t);
                this.master.gain.setValueAtTime(0, t);
                this.master.gain.linearRampToValueAtTime(this.vol * this.baseGain, t+2.0);
            }

            mute() {
                if(!this.master) return;
                this.isMuted = true;
                const t = this.ctx.currentTime;
                this.master.gain.cancelScheduledValues(t);
                this.master.gain.setValueAtTime(this.master.gain.value, t);
                this.master.gain.exponentialRampToValueAtTime(0.001, t+1.5);
                setTimeout(()=> { if(this.isMuted && this.source) this.source.stop(); }, 1600);
            }

            updateParams() {
                if(!this.toneFilter) return;
                const f = 150 * Math.pow(80, this.tone);
                this.toneFilter.frequency.setTargetAtTime(f, this.ctx.currentTime, 0.1);
                this.wet.gain.setTargetAtTime(this.space * 0.8, this.ctx.currentTime, 0.1);
            }
            
            set(k, v) {
                if(k==='vol') { 
                    this.vol = v; 
                    if(!this.isMuted && this.master) 
                        this.master.gain.setTargetAtTime(v * this.baseGain, this.ctx.currentTime, 0.1); 
                }
                else if(k==='tone') { this.tone=v; this.updateParams(); }
                else if(k==='space') { this.space=v; this.updateParams(); }
            }
            getEnergy() {
                if(!this.analyser || this.isMuted) return 0;
                const d = new Uint8Array(this.analyser.frequencyBinCount);
                this.analyser.getByteFrequencyData(d);
                let s = 0; for(let i=0; i<d.length; i++) s+=d[i];
                return (s/d.length)/255;
            }
        }

            resize() { this.canvas.width = window.innerWidth; this.canvas.height = window.innerHeight; }

                this.mode = m; this.elements = []; 
                if(m==='deep') for(let i=0; i<6; i++) this.elements.push(this.bar());
            }
            
            ripple() { return { x:Math.random()*this.canvas.width, y:Math.random()*this.canvas.height, r:0, s:Math.random()*0.4+0.1, a:Math.random()*0.3+0.1 }; }
            bar() {
                return { y: Math.random()*this.canvas.height, h: Math.random()*200+50, s: Math.random()*0.15+0.05, a: 0 };
            }

            render(e) {
                const w=this.canvas.width, h=this.canvas.height;
                
                // Clear background
                this.ctx.globalCompositeOperation = 'source-over';
                this.ctx.fillStyle = 'rgba(2,2,2,0.2)'; // 稍微快一點的清除，避免殘影過重
                this.ctx.fillRect(0,0,w,h);

                if(this.mode === 'pure') {
                    this.ctx.globalCompositeOperation = 'lighter';
                    this.ctx.strokeStyle = `rgba(200, 220, 255, ${0.05+e*0.15})`;
                    this.ctx.lineWidth = 1;
                    const t = performance.now()*0.001;
                    for(let i=0; i<3; i++) {
                        this.ctx.beginPath();
                        for(let x=0; x<w; x+=20) {
                            const y = h/2 + Math.sin(x*0.002 + t*(0.3+i*0.1) + i)* (60+e*100);
                            x===0 ? this.ctx.moveTo(x,y) : this.ctx.lineTo(x,y);
                        }
                        this.ctx.stroke();
                    }
                } else if(this.mode === 'reverie') {
                    this.ctx.globalCompositeOperation = 'screen';
                    if(Math.random() < 0.04+e*0.05) this.elements.push(this.ripple());
                    for(let i=this.elements.length-1; i>=0; i--) {
                        const p = this.elements[i];
                        p.r += p.s; p.a -= 0.003;
                        this.ctx.beginPath(); this.ctx.arc(p.x, p.y, p.r, 0, 6.28);
                        this.ctx.strokeStyle = `rgba(100, 210, 255, ${p.a})`;
                        this.ctx.stroke();
                        if(p.a<=0) this.elements.splice(i,1);
                    }
                } else if(this.mode === 'deep') {
                    this.ctx.globalCompositeOperation = 'screen';
                    if(Math.random()<0.015) this.elements.push(this.bar());
                    for(let i=this.elements.length-1; i>=0; i--) {
                        const b = this.elements[i];
                        b.y += b.s;
                        if(b.y < h*0.4) b.a = Math.min(b.a+0.002, 0.12);
                        else if(b.y > h*0.6) b.a -= 0.0015;
                        
                        this.ctx.shadowBlur = 50;
                        this.ctx.shadowColor = `rgba(200, 80, 20, ${b.a})`;
                        const g = this.ctx.createLinearGradient(0, b.y, 0, b.y+b.h);
                        g.addColorStop(0, `rgba(255, 100, 50, 0)`);
                        g.addColorStop(0.5, `rgba(255, 100, 50, ${b.a*(0.5+e)})`);
                        g.addColorStop(1, `rgba(100, 50, 20, 0)`);
                        this.ctx.fillStyle = g;
                        this.ctx.fillRect(0, b.y, w, b.h);
                        this.ctx.shadowBlur = 0;
                        if(b.a<=0) this.elements.splice(i,1);
                    }
                }
            }
