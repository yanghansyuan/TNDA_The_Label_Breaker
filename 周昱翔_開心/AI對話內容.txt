物理引擎 (Physics)：

慣性投擲：當你拖著球甩出去放開時，它會根據你最後甩動的速度繼續飛出去，而不是停在原地。

彈跳與摩擦：球撞到牆壁會反彈，而且隨著時間會慢慢減速（模擬空氣阻力），讓動作更自然。

互動與變色 (Interaction)：

拖曳變色：當你抓起一顆球時，它會變成高亮度的白色（代表能量聚集），放開後會慢慢變回原本的顏色。

融合效果：為了達成你說的「顏色融合」，我把每顆球設定了不同的顏色（黃、粉、青）。當它們靠在一起變成「融球」時，重疊的部分會自動混合出新的顏色（例如黃+藍=綠），視覺上非常漂亮！

<!DOCTYPE html>
<html lang="zh-TW">
<head>

    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>開心 - 物理互動 (V2)</title>
    <style>
        body {
            margin: 0;
            padding: 0;
            overflow: hidden;
            background-color: #f0f0f0;
            /* 禁止選取文字，避免拖曳時選到字 */
            user-select: none; 
        }

        /* 說明文字
        */
        #instruction {
            position: absolute;
            top: 20px;
            left: 20px;
            font-family: '微軟正黑體', sans-serif;
            color: #555;
            z-index: 10;
            pointer-events: none;
        }

        /* 融球濾鏡層
        */
        #canvas-container {
            width: 100vw;
            height: 100vh;
            background-color: #fff;
            /* 混合模式關鍵：
               blur: 讓邊緣模糊
               contrast: 把模糊的邊緣再次拉得銳利，產生沾黏感
            */
            filter: blur(15px) contrast(30);
        }

        canvas {
            display: block;
        }
    </style>
</head>
<body>

    <div id="instruction">
        <h1>2_add_shape.html</h1>
        <p>互動：滑鼠左鍵按住球可拖曳、甩動丟出<br>
           物理：球體撞擊反彈、融合混色<br>
           試試看把不同顏色的球「黏」在一起！</p>
    </div>

    <div id="canvas-container">
        <canvas id="emotionCanvas"></canvas>
    </div>

    <script>
        const canvas = document.getElementById('emotionCanvas');
        const ctx = canvas.getContext('2d');

        let width, height;
        let particles = [];
        
        // 滑鼠/觸控狀態
        let dragParticle = null;
        let mouseX = 0;
        let mouseY = 0;
        let lastMouseX = 0;
        let lastMouseY = 0;

        function resize() {
            width = canvas.width = window.innerWidth;
            height = canvas.height = window.innerHeight;
        }
        window.addEventListener('resize', resize);
        resize();

        class Particle {
            constructor(x, y, color) {
                this.x = x;
                this.y = y;
                this.baseRadius = Math.random() * 25 + 35; // 基礎大小
                this.radius = this.baseRadius;
                this.color = color; // 原始顏色
                this.currentColor = color; // 當前顏色
                
                // 物理速度
                this.vx = (Math.random() - 0.5) * 5;
                this.vy = (Math.random() - 0.5) * 5;
                
                this.friction = 0.98; // 摩擦力 (越小停越快)
                this.isDragging = false;
            }

            update() {
                if (this.isDragging) {
                    // 1. 拖曳模式：跟隨滑鼠，並計算「投擲速度」
                    // 速度 = 目前位置 - 上一次的位置 (這就是甩出去的力道)
                    this.vx = mouseX - lastMouseX;
                    this.vy = mouseY - lastMouseY;
                    this.x = mouseX;
                    this.y = mouseY;
                    
                    // 拖曳時變色 (變亮/變白)
                    this.currentColor = '#ffffff'; 
                    // 拖曳時稍微變大
                    this.radius = this.baseRadius * 1.2;

                } else {
                    // 2. 自由模式：應用物理
                    this.x += this.vx;
                    this.y += this.vy;

                    // 摩擦力減速
                    this.vx *= this.friction;
                    this.vy *= this.friction;

                    // 邊界反彈 (保持在畫面內)
                    if (this.x < this.radius) {
                        this.x = this.radius;
                        this.vx *= -1;
                    }
                    if (this.x > width - this.radius) {
                        this.x = width - this.radius;
                        this.vx *= -1;
                    }
                    if (this.y < this.radius) {
                        this.y = this.radius;
                        this.vy *= -1;
                    }
                    if (this.y > height - this.radius) {
                        this.y = height - this.radius;
                        this.vy *= -1;
                    }

                    // 恢復原本顏色與大小
                    this.currentColor = this.color;
                    // 讓大小有個像呼吸一樣的微小浮動，感覺更有生命
                    this.radius = this.baseRadius + Math.sin(Date.now() / 300) * 2;
                }
            }

            draw() {
                ctx.beginPath();
                ctx.fillStyle = this.currentColor;
                ctx.arc(this.x, this.y, this.radius, 0, Math.PI * 2);
                ctx.fill();
            }
        }

        function init() {
            particles = [];
            // 定義一些開心的顏色：熱情的紅、活力的黃、放鬆的藍
            const colors = ['#FFD700', '#FF6347', '#00BFFF', '#32CD32', '#FF69B4'];
            
            for (let i = 0; i < 6; i++) {
                particles.push(new Particle(
                    width / 2 + (Math.random() - 0.5) * 200,
                    height / 2 + (Math.random() - 0.5) * 200,
                    colors[i % colors.length]
                ));
            }
        }

        // --- 互動事件處理 ---

        function handleStart(x, y) {
            // 檢查有沒有點到球
            // 反向迴圈，確保先抓到最上面的球
            for (let i = particles.length - 1; i >= 0; i--) {
                const p = particles[i];
                const dx = x - p.x;
                const dy = y - p.y;
                // 距離公式檢查
                if (Math.sqrt(dx*dx + dy*dy) < p.radius) {
                    p.isDragging = true;
                    dragParticle = p;
                    break;
                }
            }
            mouseX = x;
            mouseY = y;
            lastMouseX = x;
            lastMouseY = y;
        }

        function handleMove(x, y) {
            lastMouseX = mouseX;
            lastMouseY = mouseY;
            mouseX = x;
            mouseY = y;
        }

        function handleEnd() {
            if (dragParticle) {
                dragParticle.isDragging = false;
                dragParticle = null;
            }
        }

        // 滑鼠監聽
        canvas.addEventListener('mousedown', e => handleStart(e.clientX, e.clientY));
        window.addEventListener('mousemove', e => handleMove(e.clientX, e.clientY));
        window.addEventListener('mouseup', handleEnd);

        // 觸控監聽 (手機版)
        canvas.addEventListener('touchstart', e => {
            e.preventDefault();
            handleStart(e.touches[0].clientX, e.touches[0].clientY);
        }, {passive: false});
        window.addEventListener('touchmove', e => {
            e.preventDefault(); // 防止滑動畫面
            handleMove(e.touches[0].clientX, e.touches[0].clientY);
        }, {passive: false});
        window.addEventListener('touchend', handleEnd);


        function animate() {
            ctx.clearRect(0, 0, width, height);
            
            particles.forEach(p => {
                p.update();
                p.draw();
            });

            requestAnimationFrame(animate);
        }

        init();
        animate();

    </script>
</body>

                    // [重點更新]：根據位置計算 RGB 調色盤顏色
                    
                    // 1. 計算色相 (Hue)：根據 X 軸位置，從 0 到 360 度循環
                    // 確保 x 在畫面內，避免計算出錯
                    let safeX = Math.max(0, Math.min(this.x, width));
                    const hue = (safeX / width) * 360;

                    // 2. 計算亮度 (Lightness)：根據 Y 軸位置
                    // 畫面最上方(y=0)亮度高(約70%)，最下方(y=height)亮度低(約30%)
                    let safeY = Math.max(0, Math.min(this.y, height));
                    const lightness = 70 - (safeY / height) * 40;

                    // 3. 設定新顏色 (使用 HSL 格式，飽和度固定 100% 最鮮豔)
                    this.currentColor = `hsl(${hue}, 100%, ${lightness}%)`;

                    // 拖曳時稍微變大
                    this.radius = this.baseRadius * 1.3;

                } else {
                    // --- 自由模式 ---
                    this.x += this.vx;
                    this.y += this.vy;
                    this.vx *= this.friction;
                    this.vy *= this.friction;

                    if (this.x < this.radius) { this.x = this.radius; this.vx *= -1; }
                    if (this.x > width - this.radius) { this.x = width - this.radius; this.vx *= -1; }
                    if (this.y < this.radius) { this.y = this.radius; this.vy *= -1; }
                    if (this.y > height - this.radius) { this.y = height - this.radius; this.vy *= -1; }

                    // 放開後，慢慢變回原本的顏色
                    this.currentColor = this.originalColor;
                    this.radius = this.baseRadius + Math.sin(Date.now() / 300) * 2;
                }
            }

            draw() {
                ctx.beginPath();
                ctx.fillStyle = this.currentColor;
                ctx.arc(this.x, this.y, this.radius, 0, Math.PI * 2);
                ctx.fill();
            }
        }

        function init() {
            particles = [];
            // 初始顏色我們用比較柔和的，這樣對比拖曳時的鮮豔感
            const colors = ['#FFD700', '#FF6347', '#00BFFF', '#32CD32', '#FF69B4'];
            for (let i = 0; i < 6; i++) {

            }

        function animate() {
            ctx.clearRect(0, 0, width, height);
            particles.forEach(p => {
                p.update();
                p.draw();
            });
            requestAnimationFrame(animate);
        }

        init();

調整重點分析：

球體太模糊 & 融球效果怪異：

原因：為了製造「黏糊糊」的效果，我們使用了很強的「模糊濾鏡」加上很強的「對比度濾鏡」。這雖然讓球黏在一起，但也犧牲了球體本身的清晰度，而且過高的對比有時會讓顏色看起來像「炸開」一樣不自然。

解法：我們降低濾鏡的強度。減少模糊量，讓球體邊緣更實在；同時降低對比度，讓顏色更柔和自然，但仍保留兩球靠近時的沾黏感。

融合時拖曳中斷：

原因：當兩顆小球 (A 和 B) 融合時，程式碼的邏輯是「刪除 A 和 B，然後生成一顆新的大球 C」。如果你正在拖曳 A，當 A 被刪除後，你的滑鼠手上就「抓空」了，所以拖曳會中斷。

解法：在融合發生的瞬間，我們需要告訴程式：「如果這兩顆舊球其中有一顆正在被拖曳，那麼新生成的大球也要立刻進入『被拖曳狀態』，並讓滑鼠繼續抓住它。」

這是一份優化後的版本，請存成 final_v2_fixed.html。請特別留意 CSS 中的濾鏡數值變化，以及 Javascript 中 checkCollisions 函式裡面的註解。

球體顏色的變化怪怪的，應該要是一旦當球體有發生移動變化時就會像RGB調色盤移動那樣漸變地改變顏色，然後融球的效果還是怪怪的

沒問題，我完全理解你的感覺。之前的版本中，顏色變化只有在「拖曳」時才發生，而且融球的邊緣因為濾鏡開太強，看起來會有鋸齒或雜訊（俗稱「炸掉」），確實不夠平滑。

這次的 V3 優化版 針對你提出的兩點做了核心手術：

全時動態色域（RGB 調色盤）：

現在不用拖曳，球體只要在移動，顏色就會根據它的位置自動漸變。

整個螢幕就是一個巨大的調色盤：

X 軸（左右）：決定色相（彩虹色光譜）。

Y 軸（上下）：決定光影亮度（上面亮、下面深）。

這樣球體在飛來飛去時，會像變色龍一樣不斷滑順地改變顏色。

高清融球算法（解決「怪怪的」）：

我加入了 DPR (Device Pixel Ratio) 支援。這會讓畫布解析度依照你的螢幕密度加倍，解決「模糊」和「鋸齒」的問題。

調整了濾鏡參數，讓融合的邊緣更圓潤，像水銀一樣，而不是糊糊的果凍。

            background-color: #000000;

    <div class="canvas-wrapper" id="goo-layer">
        <canvas id="gooCanvas"></canvas>

    <div class="canvas-wrapper" id="particle-layer">
        <canvas id="particleCanvas"></canvas>

    <script>
        const gooCanvas = document.getElementById('gooCanvas');
        const ctxGoo = gooCanvas.getContext('2d');

        const particleCanvas = document.getElementById('particleCanvas');
        const ctxParticle = particleCanvas.getContext('2d');

        let width, height;
        let dpr = window.devicePixelRatio || 1;

        let balls = [];
        let particles = [];
        let draggedBall = null;
        let isGameOver = false;

        let mouseX = 0, mouseY = 0;
        let lastMouseX = 0, lastMouseY = 0;

        function resize() {
            width = window.innerWidth;
            height = window.innerHeight;

            gooCanvas.width = width * dpr;
            gooCanvas.height = height * dpr;
            ctxGoo.scale(dpr, dpr);

            particleCanvas.width = width * dpr;
            particleCanvas.height = height * dpr;
            ctxParticle.scale(dpr, dpr);
            
            gooCanvas.style.width = `${width}px`;
            gooCanvas.style.height = `${height}px`;
            particleCanvas.style.width = `${width}px`;
            particleCanvas.style.height = `${height}px`;
        }
        window.addEventListener('resize', resize);
        resize();

        // --- 類別定義 ---

        class Ball {
            constructor(x, y, r) {
                this.x = x;
                this.y = y;
                this.radius = r;
                this.baseRadius = r;
                
                this.vx = (Math.random() - 0.5) * 6;
                this.vy = (Math.random() - 0.5) * 6;
                this.isDragging = false;
                this.isGiant = false; // 標記是否為終局大球
                
                // 初始色相隨機
                this.hue = Math.random() * 360;
                this.updateColor();
            }

            updateColor() {
                // 如果是終局大球，不使用位置變色邏輯
                if (this.isGiant) return;

                let safeX = Math.max(0, Math.min(this.x, width));
                this.hue = (safeX / width) * 360;

                // [修改 1] 移除白色：限制最高亮度
                // 亮度範圍控制在 40% (底部) 到 70% (頂部) 之間
                // 飽和度維持 100%，確保顏色鮮豔
                let safeY = Math.max(0, Math.min(this.y, height));
                let lightness = 70 - (safeY / height) * 30; 

                this.color = `hsl(${this.hue}, 100%, ${lightness}%)`;
            }

            update() {
                // [修改 2] 終局大球的特殊行為
                if (this.isGiant) {
                    // 自動彩虹循環變色
                    this.hue = (this.hue + 0.5) % 360; // 每幀增加色相
                    // 固定高亮度 65%，看起來很跳
                    this.color = `hsl(${this.hue}, 100%, 65%)`; 
                    
                    // 平滑移到中間
                    this.x += (width/2 - this.x) * 0.05;
                    this.y += (height/2 - this.y) * 0.05;
                    // 呼吸效果
                    this.radius = this.baseRadius + Math.sin(Date.now() / 300) * 20;
                    
                    // 大球不執行後面的移動邏輯
                    return;
                }

                if (this.isDragging) {
                    // --- 拖曳模式 (含邊界限制) ---
                    let targetX = mouseX;
                    let targetY = mouseY;
                    this.x = Math.max(this.radius, Math.min(targetX, width - this.radius));
                    this.y = Math.max(this.radius, Math.min(targetY, height - this.radius));
                    
                    this.vx = this.x - lastMouseX;
                    this.vy = this.y - lastMouseY;
                    this.radius = this.baseRadius * 1.15;
                } else {
                    // --- 自由模式 ---
                    this.x += this.vx;
                    this.y += this.vy;
                    this.vx *= 0.99;
                    this.vy *= 0.99;

                    if (this.x < this.radius) { this.x = this.radius; this.vx *= -1; }
                    if (this.x > width - this.radius) { this.x = width - this.radius; this.vx *= -1; }
                    if (this.y < this.radius) { this.y = this.radius; this.vy *= -1; }
                    if (this.y > height - this.radius) { this.y = height - this.radius; this.vy *= -1; }

                    if (this.radius > this.baseRadius) {
                        this.radius -= 0.5;
                    }
                }
                this.updateColor();
            }

            draw() {
                ctxGoo.beginPath();
                ctxGoo.fillStyle = this.color;
                ctxGoo.arc(this.x, this.y, this.radius, 0, Math.PI * 2);
                ctxGoo.fill();
            }
        }

        class Spark {
            // [修改 3] 建構子不再需要接收顏色，自己生成隨機顏色
            constructor(x, y) {
                this.x = x;
                this.y = y;
                const angle = Math.random() * Math.PI * 2;
                const speed = Math.random() * 10 + 5; // 速度更快
                this.vx = Math.cos(angle) * speed;
                this.vy = Math.sin(angle) * speed;
                this.life = 1.0;
                this.decay = Math.random() * 0.03 + 0.02;
                
                // 隨機生成鮮豔的五彩顏色
                const randomHue = Math.random() * 360;
                this.color = `hsl(${randomHue}, 100%, 60%)`;
                this.size = Math.random() * 6 + 3; // 粒子稍大
            }
            update() {
                this.x += this.vx;
                this.y += this.vy;
                this.vy += 0.2; // 重力
                this.life -= this.decay;
            }
            draw() {
                ctxParticle.globalAlpha = this.life;
                ctxParticle.fillStyle = this.color;
                
                // 繪製粒子核心
                ctxParticle.beginPath();
                ctxParticle.arc(this.x, this.y, this.size * 0.7, 0, Math.PI * 2);
                ctxParticle.fill();
                
                // 繪製外部光暈
                ctxParticle.globalAlpha = this.life * 0.5;
                ctxParticle.beginPath();
                ctxParticle.arc(this.x, this.y, this.size * 1.2, 0, Math.PI * 2);
                ctxParticle.fill();
                
                ctxParticle.globalAlpha = 1.0;
            }
        }

        // --- 邏輯函數 ---

        // [修改 3] 不再需要傳入顏色參數
        function createExplosion(x, y, count = 20) {
            for (let i = 0; i < count; i++) {
                particles.push(new Spark(x, y));
            }
        }

        function checkCollisions() {
            if (balls.length < 2) return;

            for (let i = 0; i < balls.length; i++) {
                for (let j = i + 1; j < balls.length; j++) {
                    let b1 = balls[i];
                    let b2 = balls[j];
                    let dx = b2.x - b1.x;
                    let dy = b2.y - b1.y;
                    let dist = Math.sqrt(dx*dx + dy*dy);

                    if (dist < (b1.radius + b2.radius) * 0.9) {
                        
                        let newArea = b1.radius * b1.radius + b2.radius * b2.radius;
                        let newRadius = Math.sqrt(newArea);
                        let totalR = b1.radius + b2.radius;
                        let newX = (b1.x * b1.radius + b2.x * b2.radius) / totalR;
                        let newY = (b1.y * b1.radius + b2.y * b2.radius) / totalR;

                        let newBall = new Ball(newX, newY, newRadius);
                        // 繼承色相，讓新球顏色不突兀
                        newBall.hue = (b1.hue + b2.hue) / 2;

                        if (b1.isDragging || b2.isDragging) {
                             newBall.isDragging = true;
                             draggedBall = newBall; 
                        }

                        // 融合時產生五彩火花 (數量增加)
                        createExplosion(newX, newY, 50);

                        balls.splice(j, 1);
                        balls.splice(i, 1);
                        balls.push(newBall);
                        return; 
                    }
                }
            }
        }

        function checkEndGame() {
            if (balls.length === 1) {
                let bigBall = balls[0];
                let limit = Math.min(width, height) * 0.42; 
                
                if (bigBall.radius > limit && !isGameOver) {
                    isGameOver = true;
                    bigBall.isGiant = true; // 標記為大球

                    if (draggedBall) {
                        draggedBall.isDragging = false;
                        draggedBall = null;
                    }
                }
            }
        }

        // --- 輸入處理 ---

        function handleInputStart(x, y) {
            let hitBall = false;
            for (let i = balls.length - 1; i >= 0; i--) {
                let b = balls[i];
                let dist = Math.sqrt((x - b.x)**2 + (y - b.y)**2);
                
                if (dist < b.radius) {
                    hitBall = true;
                    if (isGameOver) {
                        // [修改 3] 終極五彩大爆炸 (數量超多)
                        createExplosion(x, y, 300); 
                        balls = [];
                        isGameOver = false;
                    } else {
                        b.isDragging = true;
                        draggedBall = b;
                    }
                    break; 
                }
            }

            if (!hitBall && !isGameOver) {
                let newR = Math.random() * 25 + 35;
                balls.push(new Ball(x, y, newR));
                // 生成時產生五彩火花
                createExplosion(x, y, 15); 
            }
            mouseX = x; mouseY = y;
            lastMouseX = x; lastMouseY = y;
        }

        function handleInputMove(x, y) {
            lastMouseX = mouseX;
            lastMouseY = mouseY;
            mouseX = x;
            mouseY = y;
        }

        function handleInputEnd() {
            if (draggedBall) {
                draggedBall.isDragging = false;
                draggedBall = null;
            }
        }

        window.addEventListener('mousedown', e => handleInputStart(e.clientX, e.clientY));
        window.addEventListener('mousemove', e => handleInputMove(e.clientX, e.clientY));
        window.addEventListener('mouseup', handleInputEnd);
        
        window.addEventListener('touchstart', e => {
            e.preventDefault();
            handleInputStart(e.touches[0].clientX, e.touches[0].clientY);
        }, {passive: false});
        window.addEventListener('touchmove', e => {
            e.preventDefault();
            handleInputMove(e.touches[0].clientX, e.touches[0].clientY);
        }, {passive: false});
        window.addEventListener('touchend', handleInputEnd);

        // --- 繪圖循環 ---

        function animate() {
            ctxGoo.clearRect(0, 0, width, height);
            ctxParticle.clearRect(0, 0, width, height);

            balls.forEach(b => {
                b.update();
                b.draw();
            });

            checkCollisions();
            checkEndGame();

            for (let i = particles.length - 1; i >= 0; i--) {
                let p = particles[i];
                p.update();
                p.draw();
                if (p.life <= 0) {
                    particles.splice(i, 1);
                }
            }
            requestAnimationFrame(animate);
        }

        animate();

    </script>
