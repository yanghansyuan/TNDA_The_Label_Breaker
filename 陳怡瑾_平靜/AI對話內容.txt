這段程式碼如何滿足你的需求？

自適應網頁：CSS 中的 width: 100vw; height: 100vh; 與 background-size: cover; 確保了無論用手機或電腦螢幕開啟，畫面都會填滿且不會變形。

滑鼠左鍵點擊產生水波紋：interactive: true 設定讓套件自動偵測滑鼠點擊，並在你點擊的位置觸發 dropRadius: 20 大小的水滴擴散效果。

滑鼠移動產生隨風飄逸感：套件會自動追蹤滑鼠軌跡產生連貫的波紋。此外，我在程式碼底部加入了一個 setInterval 計時器，它會每隔兩秒在畫面上隨機產生微小的擾動，這樣即使滑鼠不動，畫面也會有微風吹過水面、波光粼粼的視覺效果。

以下是為你量身打造的純程式碼水波紋實作。請將以下內容完全取代你剛才 index.html 裡面的程式碼：

完整程式碼（純運算版）
HTML
<!DOCTYPE html>
<html lang="zh-TW">
<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>純程式碼運算 - 互動式水波紋</title>
    <style>
        /* 重置基本樣式，確保畫布填滿整個螢幕 */
        * {
            margin: 0;
            padding: 0;
            box-sizing: border-box;
        }

        body, html {
            width: 100%;
            height: 100%;
            overflow: hidden; /* 隱藏捲軸 */
            background-color: #000;
        }

        /* 透過 CSS 將畫布拉伸至全螢幕 (自適應) */
        #waterCanvas {
            width: 100vw;
            height: 100vh;
            display: block;
            cursor: pointer;
        }
    </style>
</head>
<body>

    <canvas id="waterCanvas"></canvas>

    <script>
        const canvas = document.getElementById('waterCanvas');
        const ctx = canvas.getContext('2d', { willReadFrequently: true });

        // 【效能優化設定】
        // 為了確保數學運算順暢，我們將內部解析度固定，再透過 CSS 放大至全螢幕。
        // 這樣既能保持水波的平滑感，又不會讓電腦/手機跑到卡頓。
        const width = 600;
        const height = 400;
        canvas.width = width;
        canvas.height = height;

        const halfWidth = width >> 1;
        const halfHeight = height >> 1;
        const size = width * (height + 2) * 2;

        // 建立兩個陣列來記錄水波目前的狀態與前一幀的狀態 (物理模擬)
        let rippleMap = new Int16Array(size);
        let oldIdx = width;
        let newIdx = width * (height + 3);
        let ripple, texture;

        // 1. 用程式碼畫出「漸層藍色」的水底背景 (取代圖片)
        function drawWaterBackground() {
            let gradient = ctx.createLinearGradient(0, 0, width, height);
            gradient.addColorStop(0, '#E0FFFF');   // 左上角：波光粼粼的淺藍色
            gradient.addColorStop(0.5, '#87CEEB'); // 中間：清澈的水藍色
            gradient.addColorStop(1, '#4682B4');   // 右下角：較深的海藍色
            
            ctx.fillStyle = gradient;
            ctx.fillRect(0, 0, width, height);
            
            // 將畫好的背景儲存成像素資料，以供後續折射運算使用
            texture = ctx.getImageData(0, 0, width, height);
            ripple = ctx.getImageData(0, 0, width, height);
        }

        // 2. 觸發水波的函數
        function drop(dx, dy, radius, strength) {
            // 將滑鼠的螢幕座標轉換為畫布的內部座標
            dx = Math.floor(dx / window.innerWidth * width);
            dy = Math.floor(dy / window.innerHeight * height);

            for (let j = dy - radius; j < dy + radius; j++) {
                for (let k = dx - radius; k < dx + radius; k++) {
                    if (j >= 0 && j < height && k >= 0 && k < width) {
                        rippleMap[oldIdx + (j * width) + k] += strength;
                    }
                }
            }
        }

        // 3. 核心運算：計算波浪擴散與光學折射
        function process() {
            let i = oldIdx;
            oldIdx = newIdx;
            newIdx = i;

            i = 0;
            let mapIdx = oldIdx;
            for (let y = 0; y < height; y++) {
                for (let x = 0; x < width; x++) {
                    // 物理波浪方程式：根據周圍水分子的高度計算當前高度
                    let data = (
                        rippleMap[mapIdx - width] +
                        rippleMap[mapIdx + width] +
                        rippleMap[mapIdx - 1] +
                        rippleMap[mapIdx + 1]
                    ) >> 1;
                    data -= rippleMap[newIdx + i];
                    data -= data >> 5; // 阻尼係數 (讓水波慢慢衰減平靜)

                    rippleMap[newIdx + i] = data;

                    // 光學折射效果 (利用波浪高度扭曲背景像素)
                    data = 1024 - data;
                    let a = ((x - halfWidth) * data / 1024) + halfWidth;
                    let b = ((y - halfHeight) * data / 1024) + halfHeight;

                    // 確保折射不會超出邊界
                    if (a >= width) a = width - 1;
                    if (a < 0) a = 0;
                    if (b >= height) b = height - 1;
                    if (b < 0) b = 0;

                    let newPixel = (Math.floor(a) + (Math.floor(b) * width)) * 4;
                    let curPixel = i * 4;

                    // 將折射後的背景像素寫入當前畫面
                    ripple.data[curPixel] = texture.data[newPixel];
                    ripple.data[curPixel + 1] = texture.data[newPixel + 1];
                    ripple.data[curPixel + 2] = texture.data[newPixel + 2];

                    mapIdx++;
                    i++;
                }
            }
            // 將運算完的畫面渲染出來
            ctx.putImageData(ripple, 0, 0);
        }

        // 4. 動畫迴圈
        function loop() {
            process();
            requestAnimationFrame(loop);
        }

        // --- 初始化與事件綁定 ---
        drawWaterBackground();

        // 滑鼠移動：產生小漣漪 (隨風飄逸感)
        canvas.addEventListener('mousemove', function(e) {
            drop(e.clientX, e.clientY, 2, 300); // 半徑 2, 強度 300
        });

        // 滑鼠左鍵點擊：產生大水花
        canvas.addEventListener('mousedown', function(e) {
            drop(e.clientX, e.clientY, 8, 1200); // 半徑 8, 強度 1200
        });

        // 手機觸控支援
        canvas.addEventListener('touchstart', function(e) {
            drop(e.touches[0].clientX, e.touches[0].clientY, 8, 1200);
        }, {passive: true});
        canvas.addEventListener('touchmove', function(e) {
            drop(e.touches[0].clientX, e.touches[0].clientY, 2, 300);
        }, {passive: true});

        // 模擬微風：每隔一段時間隨機在畫面上產生微小擾動
        setInterval(() => {
            drop(Math.random() * window.innerWidth, Math.random() * window.innerHeight, 3, 200);
        }, 1200);

        // 啟動動畫
        loop();
    </script>
</body>
</html>

真正的物理運算：這段程式碼建立了一個水波矩陣（rippleMap），利用波浪方程式來計算水分子的推擠與擴散，並針對像素點進行光學折射運算，這正是遊戲引擎裡模擬水面的基礎做法。

效能優化（自適應）：我將內部的物理運算解析度固定在 600x400，然後用 CSS 的 100vw / 100vh 將其拉伸到全螢幕。這種做法不僅讓波浪看起來更柔和，還能確保在效能較差的手機或舊電腦上也能順暢保持 60 幀（FPS）。

原本的程式碼只有做「折射」（扭曲背景），新的版本我加入了兩個關鍵技術：

程序化紋理生成 (Procedural Texture Generation)：為了模仿參考圖中那種深淺不一、類似大理石紋或墨水暈染的質感，我們不能只用單純的漸層。我在初始化時，用程式碼在畫布上繪製了數百個隨機大小、透明度的白色和淺藍色光暈，堆疊出一個具有有機深度的「水底」背景。

類比光照反射 (Fake Specular Highlight)：這是讓點擊變「特別亮」的關鍵。在運算水波時，程式會去計算水波的「斜率」（波浪有多陡）。當我們用力點擊時，波浪最陡峭的邊緣會被判定為「迎光面」，程式會強制在這些像素上疊加白光，製造出波光粼粼的效果。

        function generateTexture() {

            gradient.addColorStop(0, '#a7d3ed');   // 參考圖的淺藍

            ctx.fillRect(0, 0, width, height);

            // 【關鍵步驟】利用隨機堆疊的透明光暈，製造出類似墨水暈染的有機質感
            ctx.globalCompositeOperation = 'overlay'; // 使用疊加模式讓顏色融合
            for (let i = 0; i < 600; i++) {
                const x = Math.random() * width;
                const y = Math.random() * height;
                const radius = Math.random() * 100 + 20;
                const alpha = Math.random() * 0.15;
                
                // 隨機繪製白色或更淺藍色的柔和圓形
                let radialGrad = ctx.createRadialGradient(x, y, 0, x, y, radius);
                if (Math.random() > 0.5) {
                     radialGrad.addColorStop(0, `rgba(255, 255, 255, ${alpha})`);
                } else {
                     radialGrad.addColorStop(0, `rgba(200, 240, 255, ${alpha})`);
                }
                radialGrad.addColorStop(1, 'rgba(255, 255, 255, 0)');
                
                ctx.fillStyle = radialGrad;
                ctx.beginPath();
                ctx.arc(x, y, radius, 0, Math.PI * 2);
                ctx.fill();
            }
            // 恢復正常繪圖模式
            ctx.globalCompositeOperation = 'source-over';

            // 將生成好的背景材質儲存起來
            textureData = ctx.getImageData(0, 0, width, height);
            rippleData = ctx.createImageData(width, height);
        }

        // --- 2. 觸發波紋函數 ---
        function drop(x, y, radius, strength) {
            // 座標轉換
            const cx = Math.floor(x / window.innerWidth * width);
            const cy = Math.floor(y / window.innerHeight * height);
            const rSq = radius * radius;

            for (let j = cy - radius; j < cy + radius; j++) {
                for (let k = cx - radius; k < cx + radius; k++) {
                    if (j >= 0 && j < height && k >= 0 && k < width) {
                        // 產生圓形波紋
                        const dx = k - cx;
                        const dy = j - cy;
                        const distSq = dx * dx + dy * dy;
                        if (distSq < rSq) {
                            // 根據距離中心點給予不同的強度，讓波形更圓潤
                            const power = (1 - distSq / rSq) * strength;
                            rippleMap[oldIdx + (j * width) + k] -= power;
                        }
                    }
                }
            }
        }

        // --- 3. 核心物理與光影運算 ---
        function process() {
            const temp = oldIdx; oldIdx = newIdx; newIdx = temp;
            let i = 0;
            let mapIdx = oldIdx;
            const pixelData = rippleData.data;
            const baseData = textureData.data;

            for (let y = 0; y < height; y++) {
                for (let x = 0; x < width; x++) {
                    // 邊界檢查
                    if (x === 0 || x === width - 1 || y === 0 || y === height - 1) {
                         rippleMap[newIdx + i] = 0;
                         mapIdx++; i++; continue;
                    }

                    // 物理波浪公式：計算周圍高度平均值
                    let data = (
                        rippleMap[mapIdx - width] +
                        rippleMap[mapIdx + width] +
                        rippleMap[mapIdx - 1] +
                        rippleMap[mapIdx + 1]
                    ) >> 1;
                    data -= rippleMap[newIdx + i];
                    data *= damping; // 衰減

                    // 計算波浪的「斜率」(Slope)，用來判斷哪裡是迎光面

                    const yoffset = (rippleMap[mapIdx - width] - rippleMap[mapIdx + width]);

                    // 計算高光 (Specular Highlight)：斜率越陡，光越亮
                    // 這裡簡化計算，將 X 和 Y 的斜率絕對值相加
                    let highlight = (Math.abs(xoffset) + Math.abs(yoffset)) * lightStrength;
                    // 限制亮度不要爆掉
                    if (highlight > 200) highlight = 200;
                    if (highlight < 20) highlight = 0; // 太平緩的地方不要有光

                    // 計算折射後的背景座標
                    let newX = x + xoffset * 0.05; // 0.05 是折射強度係數
                    let newY = y + yoffset * 0.05;
                    // 防止折射超出畫面

                    if (newY >= height) newY = height - 1; else if (newY < 0) newY = 0;

                    const newPixelIdx = (Math.floor(newY) * width + Math.floor(newX)) * 4;
                    const curPixelIdx = i * 4;

                    // 【最終著色】 背景顏色 + 高光亮度
                    pixelData[curPixelIdx] = baseData[newPixelIdx] + highlight;     // R
                    pixelData[curPixelIdx + 1] = baseData[newPixelIdx + 1] + highlight; // G
                    pixelData[curPixelIdx + 2] = baseData[newPixelIdx + 2] + highlight; // B
                    pixelData[curPixelIdx + 3] = 255; // Alpha

                    mapIdx++;
                    i++;
                }
            }
            ctx.putImageData(rippleData, 0, 0);
            requestAnimationFrame(process);
        }

        // --- 初始化與事件 ---
        generateTexture();
        process();

        // 滑鼠移動：小而頻繁的波紋 (隨風感)

                lastMove = now;
            }
        });

        // 滑鼠點擊：大而強烈的波紋 (會觸發明顯光亮)
        canvas.addEventListener('mousedown', function(e) {
            // 強度設為 2000，這會產生很陡的斜率，進而觸發強烈高光
            drop(e.clientX, e.clientY, 12, 2000); 
        });

        // 手機支援
        canvas.addEventListener('touchstart', (e) => drop(e.touches[0].clientX, e.touches[0].clientY, 12, 2000), {passive: true});
        canvas.addEventListener('touchmove', (e) => drop(e.touches[0].clientX, e.touches[0].clientY, 3, 150), {passive: true});

質感背景 (Texture Generation)：
請看程式碼中 generateTexture() 這個函數。我不再只是畫一個簡單的漸層，而是像畫水彩一樣，在畫布上隨機疊加了 600 個不同大小、不同透明度的白色與淺藍色光暈。這創造出了你想要的圖片中那種不均勻、有深度的「大理石紋」或「水墨」質感。

點擊時的強烈光感 (Specular Highlight)：
請看 process() 函數中間的註解 // --- 計算光影與折射 ---。

我計算了每個像素點波浪的「斜率」（xoffset 和 yoffset）。

當你用滑鼠點擊時（mousedown 事件），我給了一個非常大的強度值 2000，這會製造出非常陡峭的波浪。

程式偵測到陡峭的斜率，就會在 highlight 變數中加入亮度值。

最後著色時：pixelData[...] = baseData[...] + highlight;，背景色加上亮度，就形成了那種水面反光的效果。

現在試試看，滑鼠滑過時是柔和的波紋，而用力點擊時，波浪邊緣是不是會閃爍出明亮的白光了呢？

升級一：讓滑鼠點擊更「Q 彈」柔和

你提到的「生硬」是因為原本的程式碼在點擊瞬間，只觸發了一次非常強烈的波浪運算（強度 2000）。這就像是用硬梆梆的鐵鎚敲打水面，波浪會瞬間產生尖銳的邊緣。

解決方案： 我們要模擬真實物體（例如一顆石頭）落入水中的物理過程。石頭入水不是一瞬間完成的，而是會在極短的時間內持續推開水面。

升級二：背景加入明暗點點（波光與雜質）

為了增加質感，我們不能只有大範圍的顏色暈染。真實的水體中會有懸浮微粒、氣泡，以及水面細微凹凸捕捉到的強烈光點。

解決方案： 在 generateTexture（生成背景材質）的階段，我在原本的雲霧效果之上，再疊加了一層**「高頻細節層」**。

我用程式碼隨機撒下了約 2000 個極小的點。

其中約 30% 是高亮度的白色（模擬水面閃爍的波光）。

另外 70% 是深藍色的暗點（模擬水底深處的雜質或陰影）。
這些點點在靜止時提供了豐富的細節，而當水波經過並產生折射時，它們的晃動會讓畫面看起來更有「水感」。

            for (let j = cy - radius; j < cy + radius; j++) {
                for (let k = cx - radius; k < cx + radius; k++) {
                    if (j >= 0 && j < physicsHeight && k >= 0 && k < physicsWidth) {
                        const dx = k - cx;
                        const dy = j - cy;
                        const distSq = dx * dx + dy * dy;
                        if (distSq < rSq) {
                            const power = Math.cos(distSq / rSq * Math.PI / 2) * strength;
                            rippleMap[oldIdx + (j * physicsWidth) + k] -= power;
                        }
                    }
                }
            }
        }

        // --- 2. 核心運算與繪製迴圈 ---
        function loop() {
            // A. 物理運算階段 (計算波浪起伏)
            const temp = oldIdx; oldIdx = newIdx; newIdx = temp;
            let i = 0;
            let mapIdx = oldIdx;

            for (let y = 0; y < physicsHeight; y++) {
                for (let x = 0; x < physicsWidth; x++) {
                    if (x === 0 || x === physicsWidth - 1 || y === 0 || y === physicsHeight - 1) {
                         rippleMap[newIdx + i] = 0; mapIdx++; i++; continue;
                    }
                    let data = (
                        rippleMap[mapIdx - physicsWidth] + rippleMap[mapIdx + physicsWidth] +
                        rippleMap[mapIdx - 1] + rippleMap[mapIdx + 1]
                    ) / 2; // 使用 Float32 改用除法
                    data -= rippleMap[newIdx + i];
                    data *= damping;
                    rippleMap[newIdx + i] = data;
                    mapIdx++; i++;
                }
            }

            // B. 繪製階段 (用點點表現波浪)
            // 清空畫面
            ctx.fillStyle = '#051626'; // 深海藍背景
            ctx.fillRect(0, 0, physicsWidth, physicsHeight);

            // 遍歷網格來畫點
            for (let y = 0; y < physicsHeight; y += dotSpacing) {
                for (let x = 0; x < physicsWidth; x += dotSpacing) {
                    // 取得該位置的波浪高度
                    const mapIndex = newIdx + (y * physicsWidth) + x;
                    let waveHeight = rippleMap[mapIndex];

                    // --- 視覺化核心：將高度映射到點點的屬性 ---
                    
                    // 1. 計算半徑：波峰(正值)變大，波谷(負值)變小
                    // waveHeight 通常在 -500 到 500 之間跳動，我們將其縮小比例
                    let radius = baseRadius + (waveHeight / 80); 
                    if (radius < 0.8) radius = 0.8; // 最小尺寸
                    if (radius > 7) radius = 7;     // 最大尺寸

                    // 2. 計算顏色與亮度：波峰趨近白色，波谷趨近深藍
                    let r, g, b, a;
                    // 亮度基於高度 (越高越亮)
                    let brightness = (waveHeight + 100) / 200; // 歸一化到 0~1 之間
                    if (brightness < 0) brightness = 0;
                    if (brightness > 1) brightness = 1;

                    // 顏色插值：從深藍 -> 青藍 -> 亮白
                    // 使用 exponential 讓亮部更集中在最高點
                    const sharpBrightness = Math.pow(brightness, 3); 

                    r = Math.floor(50 + sharpBrightness * 205);  // 偏藍的基底 -> 白
                    g = Math.floor(150 + sharpBrightness * 105); // 青色基底 -> 白
                    b = Math.floor(200 + sharpBrightness * 55);  // 亮藍基底 -> 白
                    a = 0.4 + sharpBrightness * 0.6; // 越亮越不透明

                    // 繪製圓點
                    ctx.beginPath();
                    ctx.arc(x, y, radius, 0, Math.PI * 2);
                    ctx.fillStyle = `rgba(${r}, ${g}, ${b}, ${a})`;
                    ctx.fill();
                }
            }

            requestAnimationFrame(loop);
        }

        // --- 初始化與事件 ---
        loop();

        // 滑鼠移動：小波紋
        let lastMove = 0;
        canvas.addEventListener('mousemove', function(e) {
            const now = Date.now();
            if (now - lastMove > 30) { 
                // 強度降低，讓移動時的點點變化比較柔和
                drop(e.clientX, e.clientY, 4, 100);
                lastMove = now;
            }
        });

        // 滑鼠點擊：柔和的連續觸發
        function softDrop(x, y) {
            let count = 0;
            let strength = 800; // 較強的初始波浪
            const radius = 15;
            const interval = setInterval(() => {
                drop(x, y, radius, strength);
                strength -= 150;
                count++;
                if (count > 4 || strength <= 0) clearInterval(interval);
            }, 40);
        }
        canvas.addEventListener('mousedown', (e) => softDrop(e.clientX, e.clientY));
        canvas.addEventListener('touchstart', (e) => softDrop(e.touches[0].clientX, e.touches[0].clientY), {passive: true});
        canvas.addEventListener('touchmove', (e) => drop(e.touches[0].clientX, e.touches[0].clientY, 4, 100), {passive: true});

        // 自動隨機波紋
        setInterval(() => {
            if (Math.random() > 0.5) {
                 drop(Math.random() * window.innerWidth, Math.random() * window.innerHeight, 3, 80);
            }
        }, 1200);

    </script>

透明畫布 (Canvas)：

只在波紋處畫點 (JS)：

這次升級的亮點：

櫻色落花：用程式碼繪製了半透明的粉色花瓣，它們會自帶非常緩慢的旋轉與漂浮感，模擬落在水面上隨波逐流的樣子。

        const canvas = document.getElementById('waterCanvas');
        const ctx = canvas.getContext('2d'); 

        // --- 設定參數 ---
        const physicsWidth = 500;  
        const physicsHeight = 350;
        
        // 水波紋點點設定
        const dotSpacing = 6;    
        const maxRadius = 2.5;   
        const colorStrength = 1.2; 
        const damping = 0.99; 
        
        canvas.width = physicsWidth;
        canvas.height = physicsHeight;
        const size = physicsWidth * physicsHeight;
        let rippleMap = new Float32Array(size * 2);
        let oldIdx = size;
        let newIdx = 0;

        // --- 新增：滑鼠軌跡與風力追蹤 ---
        let mouseX = -100;
        let mouseY = -100;
        let mouseVx = 0; // 滑鼠X軸速度 (風力)
        let mouseVy = 0; // 滑鼠Y軸速度 (風力)

        // --- 新增：花瓣粒子系統 ---
        class Petal {
            constructor() {
                // 隨機散佈在畫布上
                this.x = Math.random() * physicsWidth;
                this.y = Math.random() * physicsHeight;
                // 基礎微風漂浮速度 (往右下角緩慢飄)
                this.vx = Math.random() * 0.2 + 0.1;
                this.vy = Math.random() * 0.2 + 0.1;
                this.size = Math.random() * 3 + 4; // 花瓣大小
                this.angle = Math.random() * Math.PI * 2; // 初始旋轉角度
                this.spin = (Math.random() - 0.5) * 0.02; // 自身旋轉速度
                // 粉嫩的花瓣顏色
                const r = 255;
                const g = Math.floor(180 + Math.random() * 40);
                const b = Math.floor(200 + Math.random() * 30);
                this.color = `rgba(${r}, ${g}, ${b}, 0.85)`;
            }

            update() {
                // 1. 計算滑鼠產生的風力推動
                let dist = Math.hypot(this.x - mouseX, this.y - mouseY);
                if (dist < 60) { // 如果滑鼠靠近花瓣 (半徑 60 內)
                    let force = (60 - dist) / 60; // 越靠近力量越大
                    // 將滑鼠的移動速度轉化為推動花瓣的風力
                    this.vx += mouseVx * force * 0.08; 
                    this.vy += mouseVy * force * 0.08;
                }

                // 2. 水面摩擦力 (讓被推動的花瓣慢慢停下來)
                this.vx *= 0.95;
                this.vy *= 0.95;

                // 3. 基礎微風牽引 (即使不動滑鼠，也會有微微的自然漂流)
                this.vx += 0.005; 
                this.vy += 0.005;

                // 4. 更新位置與旋轉
                this.x += this.vx;
                this.y += this.vy;
                this.angle += this.spin;

                // 5. 邊界循環 (飄出畫面後從另一邊回來)
                if (this.x > physicsWidth + 20) this.x = -20;
                if (this.x < -20) this.x = physicsWidth + 20;
                if (this.y > physicsHeight + 20) this.y = -20;
                if (this.y < -20) this.y = physicsHeight + 20;
            }

            draw(ctx) {
                ctx.save();
                ctx.translate(this.x, this.y);
                ctx.rotate(this.angle);
                ctx.beginPath();
                // 繪製水滴狀的花瓣
                ctx.moveTo(0, -this.size);
                ctx.quadraticCurveTo(this.size, -this.size/2, 0, this.size);
                ctx.quadraticCurveTo(-this.size, -this.size/2, 0, -this.size);
                ctx.fillStyle = this.color;
                ctx.fill();
                ctx.restore();
            }
        }

        // 產生 20 片花瓣 (數量可以自己調整)
        const petals = [];
        for (let i = 0; i < 20; i++) {
            petals.push(new Petal());
        }

        // --- 核心運算 ---
        function drop(x, y, radius, strength) {
            const cx = Math.floor(x / window.innerWidth * physicsWidth);
            const cy = Math.floor(y / window.innerHeight * physicsHeight);
            const rSq = radius * radius;

            for (let j = cy - radius; j < cy + radius; j++) {
                for (let k = cx - radius; k < cx + radius; k++) {
                    if (j >= 0 && j < physicsHeight && k >= 0 && k < physicsWidth) {
                        const dx = k - cx;
                        const dy = j - cy;
                        const distSq = dx * dx + dy * dy;
                        if (distSq < rSq) {
                            const power = Math.cos(distSq / rSq * Math.PI / 2) * strength;
                            rippleMap[oldIdx + (j * physicsWidth) + k] -= power;
                        }
                    }
                }
            }
        }

        function loop() {
            const temp = oldIdx; oldIdx = newIdx; newIdx = temp;
            let i = 0;
            let mapIdx = oldIdx;
            
            // 水波物理運算
            for (let y = 0; y < physicsHeight; y++) {
                for (let x = 0; x < physicsWidth; x++) {
                    if (x === 0 || x === physicsWidth - 1 || y === 0 || y === physicsHeight - 1) {
                         rippleMap[newIdx + i] = 0; mapIdx++; i++; continue;
                    }
                    let data = (
                        rippleMap[mapIdx - physicsWidth] + rippleMap[mapIdx + physicsWidth] +
                        rippleMap[mapIdx - 1] + rippleMap[mapIdx + 1]
                    ) / 2;
                    data -= rippleMap[newIdx + i];
                    data *= damping;
                    rippleMap[newIdx + i] = data;
                    mapIdx++; i++;
                }
            }

            ctx.clearRect(0, 0, physicsWidth, physicsHeight);

            // 繪製水波點點
            for (let y = 0; y < physicsHeight; y += dotSpacing) {
                for (let x = 0; x < physicsWidth; x += dotSpacing) {
                    const mapIndex = newIdx + (y * physicsWidth) + x;
                    let waveHeight = rippleMap[mapIndex];
                    let intensity = Math.abs(waveHeight);
                    
                    if (intensity < 8) continue; 
                    
                    let alpha = (intensity / 200) * colorStrength;
                    if (alpha > 0.75) alpha = 0.75; 
                    if (alpha < 0) alpha = 0;

                    let radius = (intensity / 150) * maxRadius;
                    if (radius < 0.5) radius = 0.5; 
                    if (radius > maxRadius) radius = maxRadius;

                    ctx.beginPath();
                    ctx.arc(x, y, radius, 0, Math.PI * 2);
                    ctx.fillStyle = `rgba(255, 255, 255, ${alpha})`;
                    ctx.fill();
                }
            }

            // --- 新增：更新與繪製花瓣 ---
            petals.forEach(petal => {
                petal.update();
                petal.draw(ctx);
            });

            // 風力衰減 (當滑鼠停止移動時，風力慢慢平息)
            mouseVx *= 0.9;
            mouseVy *= 0.9;

            requestAnimationFrame(loop);
        }

        loop();

        // --- 事件綁定 ---
        let lastMove = 0;
        canvas.addEventListener('mousemove', function(e) {
            const now = Date.now();
            
            // 計算滑鼠在畫布中的座標
            let currentMx = e.clientX / window.innerWidth * physicsWidth;
            let currentMy = e.clientY / window.innerHeight * physicsHeight;
            
            // 計算滑鼠移動的速度 (風力)
            mouseVx = currentMx - mouseX;
            mouseVy = currentMy - mouseY;
            
            // 更新滑鼠位置
            mouseX = currentMx;
            mouseY = currentMy;

            if (now - lastMove > 30) { 
                drop(e.clientX, e.clientY, 4, 100); 
                lastMove = now;
            }
        });

        // 點擊觸發較大的水波
        function softDrop(x, y) {
            let count = 0;
            let strength = 600; 
            const radius = 16;
            const interval = setInterval(() => {
                drop(x, y, radius, strength);
                strength -= 100;
                count++;
                if (count > 5 || strength <= 0) clearInterval(interval);
            }, 40);
        }
        canvas.addEventListener('mousedown', (e) => softDrop(e.clientX, e.clientY));
        canvas.addEventListener('touchstart', (e) => softDrop(e.touches[0].clientX, e.touches[0].clientY), {passive: true});
        canvas.addEventListener('touchmove', (e) => {
            let currentMx = e.touches[0].clientX / window.innerWidth * physicsWidth;
            let currentMy = e.touches[0].clientY / window.innerHeight * physicsHeight;
            mouseVx = currentMx - mouseX;
            mouseVy = currentMy - mouseY;
            mouseX = currentMx;
            mouseY = currentMy;
            drop(e.touches[0].clientX, e.touches[0].clientY, 4, 100);
        }, {passive: true});

        // 隨機微風波紋
        setInterval(() => {
            if (Math.random() > 0.6) {
                 drop(Math.random() * window.innerWidth, Math.random() * window.innerHeight, 3, 90);
            }
        }, 1500);

        canvas.height = physicsHeight;
        const size = physicsWidth * physicsHeight;

        let oldIdx = size;
        let newIdx = 0;

        let mouseX = -100;
        let mouseY = -100;
        let mouseVx = 0; 
        let mouseVy = 0; 

        const activePetals = []; 
        const fallenPetals = []; 
        const clickLights = [];

        class Petal {
            constructor(isNew = false) {
                this.x = Math.random() * physicsWidth;
                this.y = isNew ? -20 : Math.random() * physicsHeight * 0.8; 
                this.vx = Math.random() * 0.2 - 0.1;
                this.vy = Math.random() * 0.4 + 0.3; 
                this.size = Math.random() * 3 + 4; 
                this.angle = Math.random() * Math.PI * 2; 
                this.spin = (Math.random() - 0.5) * 0.04; 
                
                // 【調整】：改為更深、更飽和的緋櫻色 (深粉紅)，確保不會被綠色背景吃掉
                const r = Math.floor(200 + Math.random() * 40); // 偏紅
                const g = Math.floor(70 + Math.random() * 30);  // 降低綠色
                const b = Math.floor(100 + Math.random() * 30); // 降低藍色
                this.color = `rgba(${r}, ${g}, ${b}, 0.95)`;    // 提高不透明度到 0.95

                this.targetY = physicsHeight - Math.random() * 40 - this.size;
                this.isFallen = false; 
            }

            updateFalling(index, array) {
                let dist = Math.hypot(this.x - mouseX, this.y - mouseY);
                if (dist < 60) { 
                    let force = (60 - dist) / 60; 
                    this.vx += mouseVx * force * 0.08; 
                    this.vy += mouseVy * force * 0.08;
                }

                this.vx *= 0.95;
                this.vy *= 0.95;
                this.vx += 0.005; 
                this.vy += 0.015; 

                this.x += this.vx;
                this.y += this.vy;
                this.angle += this.spin;

                if (this.x > physicsWidth + 20) this.x = -20;
                if (this.x < -20) this.x = physicsWidth + 20;

                if (this.y >= this.targetY) {
                    this.y = this.targetY;
                    this.isFallen = true;
                    this.vx = 0; 
                    this.vy = 0;
                    array.splice(index, 1);
                    fallenPetals.push(this);
                    if (fallenPetals.length > 250) fallenPetals.shift();

                }

            updateFallen() {
                let dist = Math.hypot(this.x - mouseX, this.y - mouseY);
                if (dist < 80) { 
                    let force = (80 - dist) / 80; 
                    this.vx += mouseVx * force * 0.06;
                    this.vy += mouseVy * force * 0.03;
                }

                this.vx *= 0.85;
                this.vy *= 0.85;

                this.x += this.vx;
                this.y += this.vy;
                this.angle += this.spin * 0.5 + this.vx * 0.1; 

                if (this.y < this.targetY - 30) {
                     this.y = this.targetY - 30;
                     this.vy += 0.5; 
                }
                if (this.y > physicsHeight + 10) this.y = physicsHeight + 10;
                if (this.x > physicsWidth + 20) this.x = -20;
                if (this.x < -20) this.x = physicsWidth + 20;
            }

                ctx.translate(this.x, this.y);
                ctx.rotate(this.angle);
                ctx.beginPath();

                ctx.fill();
                ctx.restore();

            activePetals.push(new Petal());
        }

        function drop(x, y, radius, strength) {
            const cx = Math.floor(x / window.innerWidth * physicsWidth);
            const cy = Math.floor(y / window.innerHeight * physicsHeight);
            const rSq = radius * radius;

            for (let j = cy - radius; j < cy + radius; j++) {
                for (let k = cx - radius; k < cx + radius; k++) {
                    if (j >= 0 && j < physicsHeight && k >= 0 && k < physicsWidth) {
                        const dx = k - cx;
                        const dy = j - cy;
                        const distSq = dx * dx + dy * dy;
                        if (distSq < rSq) {
                            const power = Math.cos(distSq / rSq * Math.PI / 2) * strength;
                            rippleMap[oldIdx + (j * physicsWidth) + k] -= power;
                        }
                    }
                }
            }
        }

        function loop() {
            const temp = oldIdx; oldIdx = newIdx; newIdx = temp;
            let i = 0;
            let mapIdx = oldIdx;
            
            for (let y = 0; y < physicsHeight; y++) {
                for (let x = 0; x < physicsWidth; x++) {
                    if (x === 0 || x === physicsWidth - 1 || y === 0 || y === physicsHeight - 1) {
                         rippleMap[newIdx + i] = 0; mapIdx++; i++; continue;
                    }
                    let data = (
                        rippleMap[mapIdx - physicsWidth] + rippleMap[mapIdx + physicsWidth] +
                        rippleMap[mapIdx - 1] + rippleMap[mapIdx + 1]
                    ) / 2;
                    data -= rippleMap[newIdx + i];
                    data *= damping;
                    rippleMap[newIdx + i] = data;
                    mapIdx++; i++;
                }

                    continue;

                // 【調整】：改為同色系 (嫩綠色) 的漸層光芒
                gradient.addColorStop(0, `rgba(255, 255, 255, ${light.alpha})`);      // 核心依舊是高亮白
                gradient.addColorStop(0.3, `rgba(200, 240, 160, ${light.alpha * 0.7})`); // 中段改為明亮的嫩綠色
                gradient.addColorStop(1, `rgba(150, 210, 120, 0)`);                   // 邊緣以綠色透明消失

                ctx.globalCompositeOperation = 'lighter'; 
                ctx.beginPath();
                ctx.arc(light.x, light.y, light.radius, 0, Math.PI * 2);
                ctx.fillStyle = gradient;
                ctx.fill();
                ctx.restore();
            }

            for (let y = 0; y < physicsHeight; y += dotSpacing) {
                for (let x = 0; x < physicsWidth; x += dotSpacing) {
                    const mapIndex = newIdx + (y * physicsWidth) + x;
                    let waveHeight = rippleMap[mapIndex];
                    let intensity = Math.abs(waveHeight);
                    if (intensity < 8) continue; 
                    let alpha = (intensity / 200) * colorStrength;
                    if (alpha > 0.75) alpha = 0.75; 
                    if (alpha < 0) alpha = 0;
                    let radius = (intensity / 150) * maxRadius;
                    if (radius < 0.5) radius = 0.5; 
                    if (radius > maxRadius) radius = maxRadius;
                    ctx.beginPath();
                    ctx.arc(x, y, radius, 0, Math.PI * 2);
                    ctx.fillStyle = `rgba(255, 255, 255, ${alpha})`;
                    ctx.fill();
                }
            }

            fallenPetals.forEach(petal => {
                petal.updateFallen(); 
                petal.draw(ctx);
            });

            for (let j = activePetals.length - 1; j >= 0; j--) {
                activePetals[j].updateFalling(j, activePetals);
                activePetals[j].draw(ctx);
            }

            mouseVx *= 0.9;
            mouseVy *= 0.9;

            requestAnimationFrame(loop);
        }

        loop();

            let currentMy = e.clientY / window.innerHeight * physicsHeight;

            mouseVy = currentMy - mouseY;
            mouseX = currentMx;
            mouseY = currentMy;

                lastMove = now;
            }
        });

最後一個想要微調的：櫻花的顏色兩個可以再淺一點嗎？兩者顏色有一點差異

這真是畫龍點睛的需求！細微的色彩差異能讓整個畫面更有層次感和生命力。

我已經針對你的需求完成了調整：

整體變淺：我大幅提高了 RGB 的數值，讓花瓣從原本深沉的「緋櫻色」變成了粉嫩、輕盈的「淡櫻色」，在黃綠色的背景上會顯得更柔和雅緻。

新舊差異：我在程式碼中區分了「新生花瓣」（從天飄落）和「舊有花瓣」（初始分佈或已落地）。

新生花瓣：顏色最淺、最粉嫩、稍微不透明一點，彷彿剛從樹上落下。

舊有花瓣：顏色稍微深一點點、帶一點點水氣濕潤的感覺，透明度也稍微高一些，模擬泡在水裡一段時間的樣子。

這細微的差別會讓畫面看起來更豐富真實。請使用這段最終版的程式碼：

            let currentMx = e.touches[0].clientX / window.innerWidth * physicsWidth;
            let currentMy = e.touches[0].clientY / window.innerHeight * physicsHeight;
            mouseVx = currentMx - mouseX;
            mouseVy = currentMy - mouseY;
            mouseX = currentMx;
            mouseY = currentMy;
            drop(e.touches[0].clientX, e.touches[0].clientY, 4, 100);
        }, {passive: true});
